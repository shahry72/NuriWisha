بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ الْجُمُعَةِ 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبەبەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


يُسَبِّحُ تەسبیحات دەکات

لِلَّـهِ بۆ خودا

مَا هەرچی

فِي لە

السَّمَاوَاتِ لە ئاسمانەکان

وَمَا و هەرچی

فِي الْأَرْضِ لە زەویە

الْمَلِكِ پادشای

الْقُدُّوسِ پاک و پیرۆز

الْعَزِيزِ دەسەڵاتدار و باڵادەستی خاوەن عیززەت

الْحَكِيمِ کاربەجێی دانا

(١)

هُوَ الَّذِي ئەو خۆی

بَعَثَ ناردی

فِي لەناو

الْأُمِّيِّينَ نەخوێنەوارەکان

رَسُولًا پێغەمبەرێک

مِّنْهُمْ لەخۆیان

يَتْلُو دەخوێنتەوە

عَلَيْهِمْ بەسەریاندا

آيَاتِهِ ئایەتەکانی

وَيُزَكِّيهِمْ وپاکیان دەکاتەوە

وَيُعَلِّمُهُمُ و فێری-

الْكِتَابَ کیتاب (قورئان)

وَالْحِكْمَةَ وحیکمەت و دانایی (سوننەت) -یان دەکات

وَإِن كَانُوا و ئەوان-

مِن لە

قَبْلُ پێشدا

لَفِي هەر لەناو

ضَلَالٍ گومڕاییەکی

مُّبِينٍ دیار وئاشکرادا -بوون

(٢)

وَآخَرِينَ و هیتریش

مِنْهُمْ لەوان

لَمَّا هێشتا

يَلْحَقُوا نەگەیشتوون

بِهِمْۚ  بەوان

وَهُوَ وەهەر ئەویش

الْعَزِيزُ دەسەڵاتداری

الْحَكِيمُ کارلەجێی دانایە

(٣)

ذَٰلِكَ ئەوە

فَضْلُ فەزڵ و بەهرەی

اللَّـهِ خودایە

يُؤْتِيهِ ئەیداتە

مَن هەرکەسێ

يَشَاءُۚ   (خۆی) بیەوێت

وَاللَّـهُ و خوداش خاوەنی

ذُو الْفَضْلِ خاوەنی فەزڵ (بەهرە وبەخششی)

الْعَظِيمِ زۆر مەزنە

(٤)

مَثَلُ نمونەی

الَّذِينَ ئەوانەی

حُمِّلُوا -

التَّوْرَاةَ تەورات -درا بەسەریان

ثُمَّ پاشان

لَمْ يَحْمِلُوهَا هەڵیان نەگرت

كَمَثَلِ  (نمونەکەیان وایە) وەک نمونەی

الْحِمَارِ کەر

يَحْمِلُ هەلبگرێت

أَسْفَارًاۚ چەندین کتێب

بِئْسَ خراپترینە

مَثَلُ نمونەی

الْقَوْمِ الَّذِينَ ئەو قەومەی

كَذَّبُوا بەدرۆیان دانا و بڕوایان نەکرد

بِآيَاتِ و بە ئایاتەکانی

اللَّـهِۚ خودا

وَاللَّـهُ و خوداش

لَا يَهْدِي ڕێنمونی و هیدایەت نادات

الْقَوْمَ ئەو قەومەی

الظَّالِمِينَ ستەمکارن

(٥)

قُلْ بڵێ

يَا أَيُّهَا الَّذِينَ هۆئەی ئەوانەی

هَادُوا لەناو یەهودیەت (جولەکایەتی) گەورەبوون

إِن زَعَمْتُمْ ئەگەر لاف لێدەدەن

أَنَّكُمْ کەئێوە

أَوْلِيَاءُ دۆست و لایەنگرانن

لِلَّـهِ بۆ خودا

مِن دُونِ بێجگە لە

النَّاسِ خەلک

فَتَمَنَّوُا دەی ئاواتەخوز و تەمەننای-

الْمَوْتَ مردن -بکەن

إِن ئەگەر

كُنتُمْ ئێوە-

صَادِقِينَ ڕاستگۆ /بووبن

(٦)

وَلَا يَتَمَنَّوْنَهُ و تەمەننای (مردن) ناکەن

أَبَدًا هەرگیز

بِمَا بەهۆی ئەوەی

قَدَّمَتْ پێشر کردوویەتی

أَيْدِيهِمْۚ دەستەکانیان

وَاللَّـهُ و خوداش

عَلِيمٌ زانایە

بِالظَّالِمِينَ بە ستەمکاران

(٧)

قُلْ بڵێ

إِنَّ بەدڵنیایی

الْمَوْتَ الَّذِي ئەو مردنەی کە

تَفِرُّونَ ڕادەکەن

مِنْهُ لێی

فَإِنَّهُ ئەوە بەدڵنیایی

مُلَاقِيكُمْ پێتان دەگات

ثُمَّ پاشان

تُرَدُّونَ دەگێردرێنەوە

إِلَىٰ بۆ لای

عَالِمِ زانای

الْغَيْبِ غەیب (نادیارەکان)

وَالشَّهَادَةِ وبینراوەکان

فَيُنَبِّئُكُم جا هەواڵی ڕاستتان پێدەدات

بِمَا بەوەی

كُنتُمْ جاران

تَعْمَلُونَ دەتانکرد

(٨)

يَا أَيُّهَا الَّذِينَ هۆ ئەی ئەوانەی

آمَنُوا بڕواتان هێناوە

إِذَا نُودِيَ کاتێ بانگدرا

لِلصَّلَاةِ بۆ نوێژ

مِن لە

يَوْمِ ڕۆژی

الْجُمُعَةِ کۆبوونەوە (هەینی)

فَاسْعَوْا ئەوە خێرا بەڕێ کەون

إِلَىٰ بۆ

ذِكْرِ یادی

اللَّـهِ خودا

وَذَرُوا و واز بهێنن لە

الْبَيْعَۚ کڕین و فرۆشتن

ذَٰلِكُمْ ئەویان

خَيْرٌ چاکترە

لَّكُمْ بۆتان

إِن ئەگەر

كُنتُمْ تَعْلَمُونَ دەتانزانی

(٩)

فَإِذَا جا کاتێ

قُضِيَتِ کۆتای هات و بڕایەوە

الصَّلَاةُ نوێژ

فَانتَشِرُوا ئەوە پەرت بن بڵاوە بکەن

فِي الْأَرْضِ لە زەوی

وَابْتَغُوا و هەوڵ بدەن وداوا بکەن

مِن فَضْلِ لە فەزڵ و بەخششی

اللَّـهِ خودا

وَاذْكُرُوا ویادی-

اللَّـهَ خودا -بکەنەوە

كَثِيرًا زۆر

لَّعَلَّكُمْ بەڵکو

تُفْلِحُونَ سەرکەوتوو سەرفراز دەبن

(١٠)

وَإِذَا جا کاتێ

رَأَوْا بینیان

تِجَارَةً بازرگانیەک

أَوْ یان

لَهْوًا ڕابواردنێک

انفَضُّوا چۆڵیانکرد

إِلَيْهَا بەرەو ئەوێ

وَتَرَكُوكَ وتۆیان جێهێشت

قَائِمًاۚ ڕاوەستاو بە پێوە

قُلْ بڵێ

مَا ئەوەی

عِندَ لای

اللَّـهِ خودایە

خَيْرٌ چاکترە

مِّنَ لە

اللَّـهْوِ ڕبواردن

وَمِنَ و لە

التِّجَارَةِۚ بازرگانی

وَاللَّـهُ و خوداش

خَيْرُ چاکترینی

الرَّازِقِينَ ڕزق و ڕۆزی دەرەکانە

(١١)
