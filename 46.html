تَسۡتَكۡبِرُون خۆتان بەزل دادەنا

فِي لە

ٱلۡأَرۡض زەوی

بِغَيۡر بەبێ

ٱلۡحَقّ حەق

وَبِمَا و بەوەی

كُنتُمۡ پێشان

تَفۡسُقُون لەڕێی ڕاست دەردەچوون

(٢٠)

وَٱذۡكُرۡ و یادبکەرەوە

أَخَا برای

عَادٍ  (قەومی) عاد

إِذۡ کە

أَنذَر وریایکردنەوە و ترساندنی

قَوۡمَهُۥ نەتەوەکەی خۆی

بِٱلۡأَحۡقَاف لە ئەحقاف

وَقَدۡ و لەکاتێ

خَلَت تێپەيڕی

ٱلنُّذُر ترسێنەرەکان

مِنۢ لە

بَيۡن نێوان

يَدَيۡه دەستەکانی (پێشی وی)

وَمِنۡ و لە

خَلۡفِهِۦٓ دووای وی

أَلَّا تَعۡبُدُوٓا کەوا نەپەرستن

إِلَّا شتێ جگەلە

ٱللَّه خودا

إِنِّي بەدڵنیایی من

أَخَاف ئەترسم

عَلَيۡكُمۡ لەسەرتان

عَذَاب عەزاب و ئەشکەنجەیی

يَوۡمٍ ڕۆژێ

عَظِيمٖ زۆر گەورە و مەزن

(٢١)

قَالُوٓا گوتیان

أَجِئۡتَنَا ئایا هاتی بۆمان

لِتَأۡفِكَنَا تا وەرمانگێڕی

عَنۡ لە ، لێی

ءَالِهَتِنَا پەرستراو و خوداکانمان

فَأۡتِنَا دە بۆمان بهێنە

بِمَا ئەوەی

تَعِدُنَآ بەڵێنمان پێدەدەی

إِن ئەگەر

كُنت تۆ-

مِن لە

ٱلصَّٰدِقِين ڕاستگۆکان (-بووی)

(٢٢)

قَال گوتی

إِنَّمَا لەڕاستیدا تەنها

ٱلۡعِلۡم زانین و زانست

عِند لای

ٱللَّه خودا یە

وَأُبَلِّغُكُم و پێتان ڕادەگەیەنم

مَّآ ئەوەی

أُرۡسِلۡت من هەناردراوم

بِهِۦ پێی

وَلَٰكِنِّيٓ بەڵکوو من

أَرَىٰكُمۡ دەتانبینم

قَوۡمٗا قەومێ

تَجۡهَلُون نەزانن و نەفامی دەکەن

(٢٣)

فلَمَّا جا کاتێ

رَأَوۡه بینیان

عَارِضٗا هەورێ دەرکەوت

مُّسۡتَقۡبِل ڕووی کردبووە

أَوۡدِيَتِهِمۡ دۆڵەکانیان

قَالُوا گوتیان

هَٰذَا ئەوەیە

عَارِضٞ هەورێک ە

مُّمۡطِرُنَا بارانمان بۆ دەبارێنێ ،

بَلۡ بەلکو

هُو ئەوە

مَا ٱسۡتَعۡجَلۡتُم ئەوەیە کە پەلەتان کرد

بِهِۦ پێی ،

رِيحٞ ڕەشەبایێکە

فِيهَا تیایدایە

عَذَاب عەزاب و ئەشکەنجەیەکی

أَلِيمٞ زۆر بە ئازار

(٢٤)

تُدَمِّر وێراندەکات

كُلّ هەموو

شَيۡءِۢ شتێ

بِأَمۡر بەفەرمانی

رَبِّهَا خاوەن و خوداوەندی

فَأَصۡبَحُوا ئیتر وایانلێهات

لَايُرَىٰٓ نابینرێ

إِلَّا شتێ جگەلە

مَسَٰكِنُهُمۡ جێگاکی نیشتەجێبوونەکانیان

كَذَٰلِك ئاوا

نَجۡزِي سزادەدەین

ٱلۡقَوۡم ئەو خەلک و قەومەی

ٱلۡمُجۡرِمِين ستەمکارەن

(٢٥)

وَلَقَدۡ و بەدڵنیایی

مَكَّنَّٰهُمۡ جێگیرمانکردبوون

فِيمَآ لەوەی

إِن کە

مَّكَّنَّٰكُمۡ ئێوەمان جێگیرکردبوو

فِيه تیایدا

وَجَعَلۡنَا وداماننا

لَهُمۡ بۆیان

سَمۡعٗا بیستن

وَأَبۡصَٰرٗا و بیناییان

وَأَفۡ‍ِٔدَة و دڵ و دەرونێ

فَمَآ جا-

أَغۡنَى سودی (-نە) بەخشی

عَنۡهُمۡ پێیان

سَمۡعُهُمۡ بیستنیان

وَلَآ أَبۡصَٰرُهُمۡ و نە بیناییەکانیان

وَلَآ أَفۡ‍ِٔدَتُهُم و نە دڵ و دەرونیان

مِّن (هیچ)

شَيۡءٍ شتێ

إِذۡ کە

كَانُوا وابوون

يَجۡحَدُون نکۆڵیاندەکرد وکوفریاندەکرد

بِ‍َٔايَٰت بە ئایەتەکانی

ٱللَّه خودا

وَحَاق و گیربوو

بِهِم لێیان

مَّا ئەوەی

كَانُوا ئەوان پێشان

بِهِۦ -

يَسۡتَهۡزِءُون گەمەو گاڵتەیان دەکرد (-پێی)

(٢٦)

وَلَقَدۡ و بەدڵنیایی

أَهۡلَكۡنَا لەناومانبرد

مَا حَوۡلَكُم ئەوەی دەورووبەری ئێوە

مِّن لە

ٱلۡقُرَى نشینگەکان (گوند و شارەکان)

وَصَرَّفۡنَا و جۆراوجۆرمانکرد

ٱلۡأٓيَٰت ئایەتەکان ، بەڵگە و نیشانە گەورەکان ، موعجیزاتەکان

لَعَلَّهُمۡ بەڵکو ئەوان

يَرۡجِعُون بگەڕێنەوە

(٢٧)

فَلَوۡلَانَصَرَهُم باشە بۆ سەریان نەخستن

ٱلَّذِین ئەوانەی

ٱتَّخَذُوا دیاریانکردوە و دایانناوە

مِن لە

دُون غەیری

ٱللَّه خودا

قُرۡبَانًا نزیکخەرەوانێک

ءَالِهَةَۢ پەرستراوانێک وگەلێ خودای درۆیینە

بَلۡ بەلکو

ضَلُّوا وونبوون

عَنۡهُمۡ لێیان

وَذَٰلِك و ئەوەش

إِفۡكُهُمۡ هەڵبەستراوەی ئەوان

وَمَا ئەوەی

كَانُوا پێشان

يَفۡتَرُون بەدرۆهەڵیاندەبەست

(٢٨)

وَإِذۡ و کە

صَرَفۡنَآ  (هەناردمان)

نَفَرٗا کۆمەڵێک

مِّن لە

ٱلۡجِنّ جنۆکان

فلَمَّا جا کاتێ

حَضَرُوه ئامادەی بوون

قَالُوٓا گوتیان

أَنصِتُوا بێدەنگ گوێبگرن

فلَمَّا ئیتر کاتێ

قَالُوا گوتیان

يَٰقَوۡمَنَآ ئەی قەومی خۆمان

سَمِعۡنَا بیستمان
