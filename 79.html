بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ النَّازِعَاتِ 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبە بەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


وَ سوێند بە

النَّازِعَاتِ توند ڕاکێشەرانی

غَرْقًا ئەوپەڕی توند ڕاکێش لە قۆڵای

(١)

وَالنَّاشِطَاتِ و سوێند بە چالاکانی

نَشْطًا چالاكن چالاکیێ

(٢)

وَالسَّابِحَاتِ و سویند بەخیراگەڕۆکانی (یان تەسبیحکارەکان)

سَبْحًا خۆشڕەون خۆشڕەویێ (دەکەن تەسبیحێ..)

(٣)

فَالسَّابِقَاتِ هەروا سویند بە پێشکەوتوان

سَبْقًا پێش ئەکەون پێشکەوتنێ

(٤)

فَالْمُدَبِّرَاتِ و سوێند بە سازدەران ی

أَمْرًا کارێ .. (بە فرمانی پەروەردگاریان)

(٥)

يَوْمَ ئەو ڕۆژەی

تَرْجُفُ ڕادەهەژێ

الرَّاجِفَةُ ڕاهەژێنەکە

(٦)

تَتْبَعُهَا بەدوای دادێت

الرَّادِفَةُ بەدواداهاتوەکە

(٧)

قُلُوبٌ دڵانێ ..

يَوْمَئِذٍ ئەوڕۆژە

وَاجِفَةٌ لەوپەڕی لەرزین و دڵەکوتەی و نائارامی دان

(٨)

أَبْصَارُهَا بینایی و چاوەکانیان

خَاشِعَةٌ شۆڕ و زەبونە (بەجۆرێ..)

(٩)

يَقُولُونَ دەڵێن

أَ ئەرێ

إِنَّا ئێمە

لَمَرْدُودُونَ هەربەڕاستی دەگێردرێینەوە؟

فِي بۆ

الْحَافِرَةِ باری جاران

(١٠)

أَ ئەرێ

إِذَا ئەگەر

كُنَّا بوینە

عِظَامًا چەند ئیسقانیکی

نَّخِرَةً پووچ و بۆش !!

(١١)

قَالُوا گوتیان

تِلْكَ ئەوە

إِذًا (کەوابێت)

كَرَّةٌ گەڕانەوەێکی (کەڕەتێکی)

خَاسِرَةٌ دۆڕاوی زیان بەخشە

(١٢)

فَإِنَّمَا جا -هەر تەنها

هِيَ -ئەوە

زَجْرَةٌ دەنگێکی گەورەی (ترسناکی) ە

وَاحِدَةٌ یەک (دەنگ)

(١٣)

فَإِذَا هەرکە وابوو ئیتر

هُم ئەوان

بِالسَّاهِرَةِ لە مەیدانی مەحشەر ن

(١٤)

هَلْ ئایا

أَتَاكَ بۆت هات

حَدِيثُ قسەو وباسی

مُوسَىٰ موسی

(١٥)

إِذْ کە

نَادَاهُ بانگی کرد

رَبُّهُ پەروەردگاری

بِالْوَادِ لە دۆڵی

الْمُقَدَّسِ پیرۆزکراو

طُوًى طوا

(١٦)

اذْهَبْ بڕۆ بۆ

إِلَىٰ لای

فِرْعَوْنَ فیرعەون

إِنَّهُ بەڕاستی ئەو

طَغَىٰ سنووری بەزاند و لە حەد دەرچوو

(١٧)

فَقُلْ جا بڵێ

هَل لَّكَ ئایا (ناتەوێ؟)

إِلَىٰ بەرەو (ئەوە بێی)

أَن کە

تَزَكَّىٰ خۆ پاك بکەیتەوە

(١٨)

وَأَهْدِيَكَ و ڕێنماییت بکەم

إِلَىٰ بۆ لای

رَبِّكَ پەروەردیگارت

فَتَخْشَىٰ ئەوسا کە بترسی

(١٩)

فَأَرَاهُ جا نیشانیدا

الْآيَةَ ئەو بەڵگەو و موعجیزەی

الْكُبْرَىٰ گەورەترینە

(٢٠)

فَكَذَّبَ ئەوەبوو (فیرعەون) بەدرۆی دانا

وَعَصَىٰ و یاخی بوو

ثُمَّ پاشان

أَدْبَرَ پشتی هەڵکرد

يَسْعَىٰ (بەپەلە ڕۆیی -کەوتە کۆشش کردن)

(٢٢)

فَحَشَرَ جا -کۆی کرد و (لەئاقارێک)

فَنَادَىٰ ئینجا بانگی ڕاهێڵا

(٢٣)

فَقَالَ ئەوەبوو گوتی

أَنَا من

رَبُّكُمُ پەروەردگاری

الْأَعْلَىٰ هەرە بەرزو بڵند (ی ئێوەم)

(٢٤)

فَأَخَذَهُ جا ڕاپێچیکرد

اللَّـهُ خودای گەورە

نَكَالَ (وەک) سزا و کردنە پەند ی

الْآخِرَةِ داویی (قیامەت)

وَالْأُولَىٰ (دونیا) و یەکەمی

(٢٥)

إِنَّ بەڕاستی

فِي ذَٰلِكَ لەوەدا (هەیە)

لَعِبْرَةً پەندێ (چۆن پەندێ) ..

لِّمَن بۆ ئەوەی

يَخْشَىٰ بترسێ

(٢٦)

أَأَنتُمْ ئەرێ ئێوە

أَشَدُّ توندتر وبەهێزترن

خَلْقًا لە پێکهاتە و دروستبووندا

أَمِ یا خود

السَّمَاءُ ۚ

بَنَاهَا (خودای پەروەردیگار) بنیادیناوە

(٢٧)

رَفَعَ-ی بەرز کردۆتەوە

سَمْكَهَا -بەرزی بنمێچەکەی

فَسَوَّاهَا جا ڕێکی کردوە و لەباری کردوە

(٢٨)

وَأَغْطَشَ و تاریکیداهێناوە

لَيْلَهَا شەوەکەی

وَأَخْرَجَ و دەریخستوە

ضُحَاهَا چێشتانگاکەی

(٢٩)

وَالْأَرْضَ و زەوییش

بَعْدَ پاش

ذَٰلِكَ ئەوە

دَحَاهَا ڕایخستوە (درێژی کردۆتەوە)

(٣٠)

أَخْرَجَ دەریکردوە

مِنْهَا لێی

مَاءَهَا ئاوەکەی

وَمَرْعَاهَا و لەوەڕگاکانی

(٣١)

وَالْجِبَالَ و چیاکانیش

أَرْسَاهَا لەنگەرگرتووی کردوون (لە زەویی چەقاندن)

(٣٢)

مَتَاعًا (ئەوانە هەمووی) مایەی رابوێرینە و سودوەرگرتنە (بۆکاتێکی کەم و دیاریراو)

لَّكُمْ بۆ ئێوە

وَلِأَنْعَامِكُمْ و بۆ ئاژەڵەکانتان

(٣٣)

فَإِذَا جا کاتێ

جَاءَتِ هات

الطَّامَّةُ بەڵا وموسیبەتی گشتگیری

الْكُبْرَىٰ هەرەگەورە

(٣٤)

يَوْمَ ئەوڕۆژە یە

يَتَذَكَّرُ وەبیری دێنتەوە

الْإِنسَانُ مرۆڤ

مَا ئەوەی-هەرچی

سَعَىٰ هەوڵی بۆداوە

(٣٥)

وَبُرِّزَتِ و دەرخسترا

الْجَحِيمُ دۆزەخ

لِمَن بۆ ئەوەی-هەرکەسێ

يَرَىٰ دەبینێ -ببینێ

(٣٦)

فَأَمَّا مَن جا هەر کەسێک

طَغَىٰ تێیپەراند و لە حەد دەرچوو

(٣٧)

وَآثَرَ و زیاتر وەلای

الْحَيَاةَ ژیانی

الدُّنْيَا دونیا

(٣٨)

فَإِنَّ ئەوە بەدڵنیایی

الْجَحِيمَ دۆزەخ

هِيَ (خۆیەتی)

الْمَأْوَىٰ ماڵ و شوێنی گەڕانەوە و حەوانەوە ی

(٣٩)

وَأَمّا بەڵام

مَنْ هەرکەسێ ئەوەی

خَافَ ترسا لە

مَقَامَ وەستانی لە بەردەستی پایە و شکۆیی

رَبِّهِ پەروەردگاری

وَنَهَى و قەدەغە و ڕێگری کرد

النَّفْسَ نەفس و خود

عَنِ لە

الْهَوَىٰ کەوتن لە ئارەزوبازی

(٤٠)

فَإِنَّ ئەوە بێگومان

الْجَنَّةَ بەهەشت

هِيَ خۆی

الْمَأْوَىٰ شوێنی گەڕانەوە و حەوانەوە یەتی

(٤١)

يَسْأَلُونَكَ پرسیارت لێدەکەن

عَنِ دەربارەی

السَّاعَةِ رۆژی قیامەت

أَيَّانَ کەیە

مُرْسَاهَا جێگیربوونی (کاتی هاتنی)

(٤٢)

فِيم لەکوێ-لەو کاتەی

أَنتَ تۆ

مِن لە

ذِكْرَاهَا یاد و باسی ئەودای

(٤٣)

إِلَىٰ (تەنها) بۆ

رَبِّكَ پەروەردیگارتە

مُنتَهَاهَا کۆتاییەکەی

(٤٤)

إِنَّمَا أَنتَ تۆ هەر تەنها

مُنذِرُ ترسێنەری

مَن کەسێک ی

يَخْشَاهَا لێی بترسێ

(٤٥)

كَأَنَّهُم وەک بڵێی

يَوْمَ ئەو ڕۆژەی

يَرَوْنَهَا دەیبینن

لَمْ يَلْبَثُوا (لە ژیانی دوونیا) نەمابووبن

إِلَّا جگە لە

عَشِيَّةً ئێوارە و شەوێ

أَوْ یان

ضُحَاهَا بەیانی و کێشتانەکەی

(٤٦)
