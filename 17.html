وَٱخۡفِضۡ و دابگرە و نزم بکەوە

لَهُمَا بۆیان

جَنَاح باڵی

ٱلذُّلّ زەلیلی و ملکەچی

مِن لە

ٱلرَّحۡمَة ڕەحمەت و بەزەییەوە

وَقُل و بڵێ

رَّبّ پەروەردیگارم

ٱرۡحَمۡهُمَا بەزەییت بێتەوە پێیان

كَمَا هەرچۆن

رَبَّيَانِي ئەوان منیان پەروەردەکرد

صَغِيرٗا بە بچووکی

(٢٤)

رَّبُّكُمۡ خودای پەروەردیگارتان

أَعۡلَم زاناتر و ئاگادارترە

بِمَا بەوەی

فِي لە ناو

نُفُوسِكُمۡ نەفسی خۆتانە

إِن ئەگەر

تَكُونُوا ئێوە -

صَٰلِحِين چاک و صالح (-بن)

فَإِنَّهُۥ ئیتر بەدڵنیایی ئەو

كَان -

لِلۡأَوَّٰبِين بۆ هەمیشە گەڕاوەکان (بۆلای خودا)

غَفُورٗا زۆر لێبووردەیە

(٢٥)

وَءَات و بدە

ذَا ئەوی خاوەن

ٱلۡقُرۡبَى نزیکی و خزمایەتیە

حَقَّهُۥ هەقەکەی

وَٱلۡمِسۡكِين و (بدە) نەدار

وَٱبۡن و کوڕی

ٱلسَّبِيل ڕێگا (ئەوەی سەفەریکردوە و لێی بڕاوە)

وَلَاتُبَذِّرۡ و زێدەڕی وپارەڕێژی مەکە

تَبۡذِيرًا زێدەڕەوییەک

(٢٦)

إِنّ بەدڵنیایی

ٱلۡمُبَذِّرِين پارەڕێژان

كَانُوٓا هەر-

إِخۡوَٰن برایانی

ٱلشَّيَٰطِين شەیطانەکان (-بوون)

وَكَان و-

ٱلشَّيۡطَٰن شەیطان

لِرَبِّهِۦ بۆ پەروەدیگاری

كَفُورٗا زۆر ناشوکر (-بوو)

(٢٧)

وَإِمَّا جا یان

تُعۡرِضَنّ پشت دەکەی

عَنۡهُم لییان

ٱبۡتِغَآء بۆخواستی

رَحۡمَةٖ بەزەیی و میهرەبانیە ک

مِّن لە

رَّبِّك پەروەردیگارتەوە

تَرۡجُوهَا هیوات پێیە

فَقُل ئەوە بڵێ

لَّهُمۡ پێیان

قَوۡلٗا قسە و وتارێکی

مَّيۡسُورٗا رۆن و ئاسان

(٢٨)

وَلَاتَجۡعَلۡ و -

يَدَك دەستت

مَغۡلُولَة بەستراو (-مەکە)

إِلَى بۆ/ بە

عُنُقِك گەردەنت

وَلَاتَبۡسُطۡهَا و واڵاو و کراوەشی مەکە

كُلّ ئەوپەڕی

ٱلۡبَسۡط واڵا و کراوەیی

فَتَقۡعُد ئیتر دابنیشی

مَلُومٗا بە لۆمەکراوی

مَّحۡسُورًا پەشیمان و پەریشانی

(٢٩)

إِنّ بەدڵنیایی

رَبَّك پەروەردیگارت

يَبۡسُط فرەوان و زۆر دەکات

ٱلرِّزۡق رۆزی و بژێوی

لمَن ئەو کەسەی

يَشَآء ئەو بیەوێت

وَيَقۡدِر و تەنگی دەکات

إِنَّهُۥ بەدڵنیایی ئەوە

كَان هەمیشە

بِعِبَادِهِۦ بە بەندەکانی

خَبِيرَۢا تەواو زانای ورد و درشتی

بَصِيرٗا تەواو بینەر بە هەموو شتێکە

(٣٠)

وَلَاتَقۡتُلُوٓا و مەکوژن

أَوۡلَٰدَكُمۡ مناڵەکانتان

خَشۡيَة لە ترسی

إِمۡلَٰقٖ نەبوونی

نَّحۡن ئێمە خۆمان

نَرۡزُقُهُمۡ ڕزق و ڕۆزیان دەدەین

وَإِيَّاكُمۡ لەگەڵ ئێوەش

إِنّ بەدڵنیایی

قَتۡلَهُمۡ کوشتنی ئەوان

كَان هەمیشە -

خِطۡ‍ٔٗا هەڵە و تاوانێکی

كَبِيرٗا گەورە (-بووە)

(٣١)

وَلَاتَقۡرَبُوا و نزیک مەکەونەوە لە

ٱلزِّنَىٰٓ زینا

إِنَّهُۥ بەدڵنیایی ئەوە

كَان هەر

فَٰحِشَة گوناهیکی دزێو و قێزەوەن

وَسَآء و-

سَبِيلٗا ڕێگایەکی (-زۆر خراپە)

(٣٢)

وَلَاتَقۡتُلُوا و مەکوژن

ٱلنَّفۡس ئەونەفس و کەسە

ٱلَّتِي ئەوەی

حَرَّم -

ٱللَّه خودا (-قەدەغەی کردوە)

إِلَّا مەگەر

بِٱلۡحَقّ بە حەق و ڕەوایی

وَمَن و هەرکەسێ

قُتِل بکوژرێ

مَظۡلُومٗا بە ستەملێکراوی

فَقَدۡ ئەوە بەدڵنیایی

جَعَلۡنَا دامانناوە

لِوَلِيِّهِۦ بۆ کەس و کارەکەی

سُلۡطَٰنٗا حەق و دەسەڵاتێ

فَلَايُسۡرِف جا با زێدەڕەوی نەکات

فِّي لە

ٱلۡقَتۡل کوشتن

إِنَّهُۥ بەدڵنیایی ئەو

كَان بوو

مَنصُورٗا پشتیوانکراوە

(٣٣)

وَلَاتَقۡرَبُوا و نزیک مەکەونەوە لە

مَال ماڵی

ٱلۡيَتِيم هەتیوو

إِلَّا ئیللا

بِٱلَّتِي بەوەی

هِي ئەو

أَحۡسَن چاکترین ە

حَتَّى تاکو

يَبۡلُغ دەگات بە

أَشُدَّهُۥ توند و تۆڵی جەستە و هۆشی خۆی

وَأَوۡفُوا تەواو پابەند بن

بِٱلۡعَهۡد بە بەڵێن و پەیمان

إِنّ بەدڵنیایی

ٱلۡعَهۡد پەیمان و بەڵێن

كَان هەر

مَسۡ‍ُٔولٗا جێی پرسیار لێکرانە

(٣٤)

وَأَوۡفُوا و مافی تەواوبدەن ە

ٱلۡكَيۡل کێش و پێوانە

إِذَا ئەگەر

كِلۡتُمۡ شتتان کێشا

وَزِنُوا و بکێشن و بپێون

بِٱلۡقِسۡطَاس بە میزان و پێوانەی

ٱلۡمُسۡتَقِيم ڕاست و دروست

ذَٰلِك ئەویان

خَيۡرٞ خێر و چاکەیە

وَأَحۡسَن و باشترین

تَأۡوِيلٗا لێکدانەوەیە

(٣٥)

وَلَاتَقۡف و مەکەوە دوای

مَالَيۡس نیە

لَك بۆتۆ

بِهِۦ دەربارەی وی

عِلۡم زانست و زانیاری

إِنّ بەدڵنیایی

ٱلسَّمۡع بیستن و گوێ

وَٱلۡبَصَر و بینین

وَٱلۡفُؤَاد و دڵ و هۆش

كُلّ هەموو

أُوْلَٰٓئِك ئەوانە

كَان هەمیشە

عَنۡه -

مَسۡ‍ُٔولٗا بەرپرسن (-لێی)

(٣٦)

وَلَاتَمۡش و مەگەڕێ و مەڕۆ

فِي لەسەر

ٱلۡأَرۡض زەوی

مَرَحًا بەفیز و لەخۆبای بوونەوە

إِنَّك بەدڵنیایی تۆ

لَن تَخۡرِق هەرگیز -

ٱلۡأَرۡض زەوی (-کون ناکەی)

وَلَن وهەرگیز -

تَبۡلُغ ناگەی بە

ٱلۡجِبَال چیاکان

طُولٗا لەدرێژیدا

(٣٧)

كُلّ هەموو

ذَٰلِك ئەوانە

كَان -

سَيِّئُهُۥ خراپەکەیان

عِند لای

رَبِّك پەروەردیگارت

مَكۡرُوهٗا ناحەز و نەویستراوە

(٣٨)

ذَٰلِك ئەویان

مِمَّآ لەوەیە

أَوۡحَىٰٓ کە سروشی کردوە و وەحی کرد وە

إِلَيۡك بۆتۆ

رَبُّك پەروەردیگارت

مِن لە

ٱلۡحِكۡمَة حیکمەت و لێزانی و کاربەجێیی

وَلَاتَجۡعَلۡ و دامەنێ

مَع لەگەڵ

ٱللَّه خودا

إِلَٰهًا پەرستراوێکی

ءَاخَر دیکە

فَتُلۡقَى ئەوساکە تێهەڵدەدرێیتە

فِي ناو

جَهَنَّم دۆزەخ

مَلُومٗا بەلۆمەکراوی

مَّدۡحُورًا و دوورخراوی (لەڕەحمەتی خودا)

(٣٩)

أَفَأَصۡفَىٰكُمۡ ئایا ئێوەی تایبەتکردوە

رَبُّكُم پەروەردیگارتان

بِٱلۡبَنِين بەکوڕان

وَٱتَّخَذ و هەڵیبژاردووە

مِن لە

ٱلۡمَلَٰٓئِكَة فریشتەکان

إِنَٰثًا مێیان

إِنَّكُمۡ بەدڵنیایی ئێوە

لَتَقُولُون -

قَوۡلًا قسەیەکی

عَظِيمٗا زۆر زۆر گەورە و مەزن (-دەڵێن)

(٤٠)

وَلَقَدۡ و بەدڵنیایی

ٱلۡقُرۡءَان قورئانە

لِيَذَّكَّرُوا تا یادبکەنەوە و پەند وەرگرن

إِلَّا شتێ جگەلە

نُفُورٗا ڕەوینەوە و دورکەوتنەوە

(٤١)

إِلَى بۆ لای

ذِي خاوەن

ٱلۡعَرۡش عەرش و تەختی فەرمانڕەوایی

سَبِيلٗا ڕێگا و هۆکارێک (بۆ ڕکابەری)

يَقُولُون ئەوان دەڵێن

كَبِيرٗا گەورە

إِنَّهُۥ بەدڵنیایی ئەو

ءَايَٰتِۢ بەلگە و نیشانەی گەورە

بَيِّنَٰتٖ هەمووی ڕۆن و ئاشکرا

فَسۡ‍َٔلۡ جا پرسیار بکە لە

بَنِيٓ ڕۆڵە و هۆزی

إِسۡرَٰٓءِيل پێغەمبەر یەعقووب (جولەکەکان)

إِذۡ کە

جَآءَهُمۡ ئەو بۆیان هات

فَقَال جا -

لَهُۥ پێی (-گوت)

فِرۡعَوۡن فیرعەون

إِنِّي بەڕاستی من

لَأَظُنُّك واگومانت پێدەبەم ـ

يَٰمُوسَى ئەی موسا

مَسۡحُورٗا جادوولێکراو (ـبیت)

(١٠١)

قَال گوتی

لَقَدۡ بەدڵنیایی

عَلِمۡت تۆ زایوتە

مَآ ـ

أَنزَل دای (-نە) بەزاندوە

هَٰٓؤُلَآء ئەوانێهێ

إِلَّا یەکێک جگەلە

رَبّ پەروەردیگار وخاوەنی

ٱلسَّمَٰوَٰت ئاسمانەکان

و ٱلۡأَرۡض و زەوی

بَصَآئِر چەندین بەڵگەی ڕوونە

وَإِنِّي و بەڕاستی من

لَأَظُنُّك وا گومانت پێدەبەم

يَٰفِرۡعَوۡن ئەی فیرعەون

مَثۡبُورٗا بەهیلاکچووی

(١٠٢)

فَأَرَاد جا ئەو ویستی

أَن کە

يَسۡتَفِزَّهُم بەسوکایەتی پێکردنەوە دەریان بکات

مِّن لە

ٱلۡأَرۡض زەوی

فَأَغۡرَقۡنَٰه جا نوقممانکرد

وَمَن و ئەوەی

مَّعَهُۥ لەگەڵی بوو

جَمِيعٗا هەمووی

(١٠٣)

وَقُلۡنَا و گوتمان

مِنۢ لە

بَعۡدِهِۦ دوایی ئەو

لِبَنِيٓ بە نەوەی

إِسۡرَٰٓءِيل یەعقوب پێغەمبەر (جولەکەکان)

ٱسۡكُنُوا نیشتەجێ بن لە

ٱلۡأَرۡض زەوی

فَإِذَا جاکاتێ

جَآء هات

وَعۡد بەڵێنی

ٱلۡأٓخِرَة ڕۆژی دوایی

جِئۡنَا ـ

بِكُمۡ ئێوەمان (-هێنا)

لَفِيفٗا کۆمەڵ کۆمەڵ

(١٠٤)

وَبِٱلۡحَقّ و بەهەق و ڕاستی

أَنزَلۡنَٰه دامانبەزاند

وَبِٱلۡحَقّ و بەهەق و ڕاستی

نَزَل دابەزی

وَمَآ و-

أَرۡسَلۡنَٰك تۆمان (-نە) ناردوە

إِلَّا بەدەر لەوەی -

مُبَشِّرٗا موژدەبەخش

وَنَذِيرٗا و بێدار کەر و ترسێنەر (-بی)

(١٠٥)

وَقُرۡءَانٗا و قوڕئانێک

فَرَقۡنَٰه بەش بەشمانکرد

لِتَقۡرَأَهُۥ تا بیخوێنیەوە

عَلَى بەسەر

ٱلنَّاس خەلک

عَلَى بە

مُكۡثٖ هێواشی

وَنَزَّلۡنَٰه و دامانبەزاند

تَنزِيلٗا دابەزینێکی یەک لەدوا یەک

(١٠٦)

قُلۡ بڵێ

ءَامِنُوا بڕوا بهێنن

بِهِۦٓ پێی

أَوۡ یان

لَاتُؤۡمِنُوٓا بڕوانەهێنن

إِنّ بەدڵنیایی

ٱلَّذِین ئەوانەی

أُوتُوا پێیاندراوە

ٱلۡعِلۡم زانین و زانست

مِن لە

قَبۡلِهِۦٓ پێش ئەو

إِذَا کاتێ

يُتۡلَى دەخوێندرێتەوە

عَلَيۡهِمۡ بەسەریان

يَخِرُّونَۤ خێرا دەکەون

لِلۡأَذۡقَانِۤ بەسەر دەم و چاویان

سُجَّدٗاۤ بەسوژدەبردن

(١٠٧)

وَيَقُولُون و دەڵێن

سُبۡحَٰن پاکی و بێگەردی بۆ

رَبِّنَآ پەروەردیگارمان

إِن (بەڕاستی)

كَان -

وَعۡد بەڵێنی

رَبِّنَا پەروەردیگارمان

لَمَفۡعُولٗا (هەمیشە) کردرا (-بوو)

(١٠٨)

وَيَخِرُّون و خێرا دەکەون

لِلۡأَذۡقَان بۆ سەر چەناگە و دەم و چاوان

يَبۡكُون دەگرین

وَيَزِيدُهُمۡ و بۆیان زیاد دەکات

خُشُوعٗا 🕌 ملکەچی وکپی و داماوی

(١٠٩)

قُل بڵێ

ٱدۡعُوا بپاڕێنەوە لە

ٱللَّه خودا

أَو یان

ٱدۡعُوا بپاڕێنەوە لە

ٱلرَّحۡمَٰن ڕەحمان (ناوێکی پیرۆزی خودایە مانای زۆر میهرەبانی زۆربەبەزەیی)

أَيّٗا هەرکامیان بێ

مَّا ئەوە

تَدۡعُوا لێی دەپاڕێنەوە و هاواری لۆدەکەن

فَلَه ئەوە هەر هی ئەوە

ٱلۡأَسۡمَآء ناوە

ٱلۡحُسۡنَى چاکترین و پیرۆزترینەکان

وَلَاتَجۡهَرۡ و دەنگ هەڵمەبڕە و بەرز مەکەوە

بِصَلَاتِك بە پاڕانەوەت

وَلَاتُخَافِتۡ و نە کپ و بێدەنگیشی بکە

بِهَا پێی

وَٱبۡتَغ و بخوازە (دەنگێک)

بَيۡن نێوان

ذَٰلِك ئەوانە

سَبِيلٗا بەڕێگا و جۆرێکی (مام ناوەند)

(١١٠)

وَقُل و بڵێ

ٱلۡحَمۡد هەرچی سوپاس و ستایشە

لِلَّه بۆ ئەو خودایە

ٱلَّذِي ئەوەی

لَمۡ يَتَّخِذۡ هەڵی نەبژاردوە و دیاری نەکردوە

وَلَدٗا (هیچ) کوڕێک

وَلَمۡ يَكُن وهەرگیز نەبووە

لَّهُۥ بۆ ئەو

شَرِيكٞ هاوبەشێک

فِي لە

ٱلۡمُلۡك دەسەڵات و خاوەنداری

وَلَمۡ يَكُن وهەرگیز نەبووە

لَّهُۥ بۆ ئەو

وَلِيّٞ دۆست و پشتیوانێک

مِّن لە

ٱلذُّلّ بێدەسەڵاتی و لاوازی

وَكَبِّرۡه و گەورەی ڕابگرە

تَكۡبِيرَۢا بە گەورەکرن و گوتنی الله اکبر

(١١١)
