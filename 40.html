لَّمۡ -

نَقۡصُصۡ (-نە) مانگێڕاوەتەوە چیڕۆکیان

عَلَيۡك بەسەرت ، لەسەرت

وَمَا و نە

كَان بووە

لِرَسُولٍ بۆ پەیامبەرێک

أَن کە

يَأۡتِي بێنێت

بِ‍َٔايَةٍ ئایەت وموعجیزەیەک

إِلَّا تەنها

بِإِذۡن بەئیزن و ڕێپێدانی

ٱللَّه خودا

فَإِذَا جاکاتێ

جَآء -

أَمۡر فەرمانی

ٱللَّه خودا (-هات)

قُضِي بڕایەوە

بِٱلۡحَقّ بە حەق و ڕاستی و ڕەوایی

وَخَسِر و خەسارەتمەند بوون

هُنَالِك ئا لەوێ

ٱلۡمُبۡطِلُون ناحەق و پڕووپووچ بێژان

(٧٨)

ٱللَّه الله یە

ٱلَّذِي ئەوەی

جَعَل دروستیکرد

لَكُم بۆتان

ٱلۡأَنۡعَٰم مەڕ و ماڵات و ئاژەڵانێک

لِتَرۡكَبُوا تا سوارببن

مِنۡهَا لە(سەر) هەندێکیان

وَمِنۡهَا و لە(هەندێکیانیش)

تَأۡكُلُون دەخۆن

(٧٩)

وَلَكُمۡ بۆ ئێوە هەیە

فِيهَا تیایدا

مَنَٰفِع چەندین سوود

وَلِتَبۡلُغُوا و تا بگەینن

عَلَيۡهَا بەسەری دا

حَاجَة حاجەت و پێویستیەک

فِي لە

صُدُورِكُمۡ سینگ و دڵتاندایە

وَعَلَيۡهَا و لە سەر ئەوان

وَ عَلَى و لەسەر

ٱلۡفُلۡك کەشتی

تُحۡمَلُون هەڵدەگیرێن

(٨٠)

وَيُرِيكُمۡ و پیشانتان دەدات

ءَايَٰتِهِۦ ئایەتەکانی (بەلگە و نیشانە و موعجیزاتی)

فَأَيّ جا کام

ءَايَٰت ئایەتەکانی قورئان ، نیشانە گەورەکانی ، موعجیزاتی

ٱللَّه خودا

تُنكِرُون ئینکار دەکەن و نکۆڵی لێدەکەن

(٨١)

أَفَلَمۡ يَسِيرُوا ئایا نەگەڕاون

فِي لە ، لەناو

ٱلۡأَرۡض زەوی

فَيَنظُرُوا ئەوسا سەیر بکەن و تێڕابمینێن

كَيۡف چۆن

كَان بوو

عَٰقِبَة سەرئەنجامی

ٱلَّذِین ئەوانەی

مِن لە

قَبۡلِهِم پێش ئەوان (بوون)

كَانُوٓا ئەوان-

أَكۡثَر زۆرتر

مِنۡهُمۡ لەوان

وَأَشَدّ و تووند تۆڵتر (-بوو)

قُوَّة هێز و قووەتی

وَءَاثَارٗا و شوێنەوار و کاریگەریان

فِي لە

ٱلۡأَرۡض زەوی

فَمَآ جا-

أَغۡنَى سوودی نەبەخشێ

عَنۡهُم (پێیان)

مَّا ئەوەی

كَانُوا پێشان

يَكۡسِبُون بەدەستیان دەهێنا

(٨٢)

فلَمَّا جا کاتێ

جَآءَتۡهُمۡ بۆیان هات

رُسُلُهُم پەیامبەرانیان بە

بِٱلۡبَيِّنَٰت به‌ڵگه‌ی ڕوون و ئاشکراو موعجیزات

فَرِحُوا دڵیانخۆش بوو

بِمَا بەوەی

عِندَهُم لایان هەبوو

مِّن لە

ٱلۡعِلۡم زانین و زانست

وَحَاق و گیربوو

بِهِم لێیان

مَّا ئەوەی

كَانُوا ئەوان پێشان

بِهِۦ -

يَسۡتَهۡزِءُون گەمەو گاڵتەیان دەکرد (-پێی)

(٨٣)

فلَمَّا جا کاتێ

رَأَوۡا بینیان

بَأۡسَنَا زەبر و زەنگی ئێمە

قَالُوٓا گوتیان

ءَامَنَّا بڕوامان هێنا

بٱللَّه بەخودا

وَحۡدَهُۥ بە تەنهایی

وَكَفَرۡنَا و کوفرمانکرد

بِمَا بەوەی

كُنَّا پێشان-

بِهِۦ پێی

مُشۡرِكِين موشریک و بێبڕوا (-بوون)

(٨٤)

فَلَمۡ يَك جا ئەوە نەبوو

يَنفَعُهُمۡ سودیان پێبگەیەنێ

إِيمَٰنُهُمۡ بڕواهێنانیان

لَمَّا کاتێ

رَأَوۡا بۆیان هات

بَأۡسَنَا زەبر و زەنگی ئێمە

سُنَّت سونەت و یاسای

ٱللَّه خودا یە

ٱلَّتِي ئەوەی

قَدۡ بێگومان

خَلَتۡ (هەروابووە)

فِي لە

عِبَادِهِۦ بەندەکانی

وَخَسِر و خەسارەتمەند بوون

هُنَالِك ئا لەوێ

ٱلۡكَٰفِرُون کافر و بێبڕواکان

(٨٥)
