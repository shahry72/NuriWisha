أُوْلَٰٓئِك ئەوان

ٱلَّذِین ئەوانەن

لَعَنَهُم لەعنەتی لێکردوون و بێبەشی کردوون لەڕەحمەتی خۆی

ٱللَّه خودا

فَأَصَمَّهُمۡ جا کەڕیکردوون

وَأَعۡمَىٰٓ و کوێریکردوە

أَبۡصَٰرَهُمۡ بیناییەکانیان

(٢٣)

أَفَلَايَتَدَبَّرُون ئایا رانامینێن و وردنابنەوە لە

ٱلۡقُرۡءَان قورئان و کەلامی خودا

أَمۡ یان

عَلَى لەسەر

قُلُوبٍ دڵەکانیان (هەیە)

أَقۡفَالُهَآ کۆت و زنجیرەکانیان

(٢٤)

إِنّ بەدڵنیایی

ٱلَّذِین ئەوانەی

ٱرۡتَدُّوا هەڵگەڕانەوە

عَلَى ٰٓ لەسەر

أَدۡبَٰرِهِم دوواوەیان

مِّنۢ لە

بَعۡد دوای

مَا تَبَيَّن ئەوەی ڕوونبووەوە

لَهُم لۆیان

ٱلۡهُدَى ڕێی ڕاست و هیدایەت

ٱلشَّيۡطَٰن  (ئەوە) شەیطان

سَوَّل ڕازاندیەوە وئاسانیکرد

لَهُمۡ بۆیان

وَأَمۡلَى و هیوای تەمەندرێژی داوە

لَهُمۡ بۆیان

(٢٥)

ذَٰلِك ئەویان

بِأَنَّهُمۡ بەهۆی ئەوەی

قَالُوا گوتیان

لِلَّذِين بەئەوانەی

كَرِهُوا رقیان بوو لە

مَا نَزَّل ئەوەی دایبەزاندووە

ٱللَّه خودا

سَنُطِيعُكُمۡ لەوزووانە گوێڕایەڵتان دەبین

فِي لە

بَعۡض هەندێ

ٱلۡأَمۡر کارو فەرمایشتان

وَٱللَّه و خودا

يَعۡلَم دەزانێ

إِسۡرَارَهُمۡ شاردنەوە و پەنهانکردنەکانیان

(٢٦)

فكَيۡف جا چۆنە

إِذَا کاتێ

تَوَفَّتۡهُم تەواویانیکرد ژیانیان (ڕۆحیان کێشان)

ٱلۡمَلَٰٓئِكَة فریشتەکان

يَضۡرِبُون لێدەدەن لە

وُجُوهَهُمۡ دەموو چاوەکانیان

وَأَدۡبَٰرَهُمۡ و دوواوەیان

(٢٧)

ذَٰلِك ئەویان

بِأَنَّهُم بەهۆی ئەوەی

ٱتَّبَعُوا ئەوان کەوتنە دوای

مَآ شتێ

أَسۡخَط تووڕەیکردوە

ٱللَّه خودا

وَكَرِهُوا و ئەوان رقیان بوو لە

رِضۡوَٰنَهُۥ ڕەزامەندی ئەو

فَأَحۡبَط ئیتر (خودا) پوچیکردەوە

أَعۡمَٰلَهُمۡ کار و کردەوەکانیان

(٢٨)

أَمۡ یان

حَسِب وایان زانیوە

ٱلَّذِین ئەوانەی

فِي لەناو

قُلُوبِهِم دڵەکانیان (هەیە)

مَّرَض نەخۆشیێک

أَن کە

لَّن هەرگیز-

يُخۡرِج دەر (-نا) هێنێ و بەدەر (-نا) خا

ٱللَّه خودا

أَضۡغَٰنَهُمۡ ڕق و قینەیان

(٢٩)

وَلَوۡ وئەگەر

نَشَآء بمانەوێ

لَأَرَيۡنَٰكَهُمۡ ئەوە بەدڵنیایی ئەوانت نیشان دەدەیین

فَلَعَرَفۡتَهُم ئیتر بە دڵنیایی دەیانناسی

بِسِيمَٰهُمۡ بە سیمایان

وَلَتَعۡرِفَنَّهُمۡ و هەردەشیانناسی

فِي لە

لَحۡن ئاواز و چۆنیەتی

ٱلۡقَوۡل گوفتار

وَٱللَّه و خودا

يَعۡلَم دەزانێ و ئاگادارە بە

أَعۡمَٰلَكُمۡ کار و کردەوەکانتان

(٣٠)

وَلَنَبۡلُوَنَّكُمۡ و بەدڵنیایی تاقیتان دەکەینەوە

حَتَّى تاکو

نَعۡلَم بزانین و بناسێنین

ٱلۡمُجَٰهِدِين موجاهید و کۆشش کارەکان

مِنكُمۡ لەنگۆ

وَٱلصَّٰبِرِين و ئارامگرەکان

وَنَبۡلُوَا و تاقی بکەینەوە ودەربخەین

أَخۡبَارَكُمۡ هەواڵەکانتان

(٣١)

إِنّ بەدڵنیایی

ٱلَّذِین ئەوانەی

كَفَرُوا کوفریانکرد و بێ بڕوابوون

وَصَدُّوا و بەربەست وکۆسپیان دانا

عَن لە

سَبِيل ڕێگای

ٱللَّه خودا

وَشَآقُّوا و ڕکابەریانکرد دژی

ٱلرَّسُول پێغەمبەر

مِنۢ لە

بَعۡد دوای

مَا تَبَيَّن ئەوەی ڕوونبووەوە

لَهُم لۆیان

ٱلۡهُدَى ڕێی ڕاست و هیدایەت

لَن يَضُرُّوا هەرگیز -زەرەر (-نا) دەن لە

ٱللَّه خودا

شَيۡ‍ٔٗا بە (هیچ) جۆرێ

وَسَيُحۡبِط و لەوزوانەش پووچ دەکاتەوە

أَعۡمَٰلَهُمۡ هەوڵ و کار و کردەوەکانیان

(٣٢)

يَٰٓأَيُّهَا هۆ ئەی

ٱلَّذِین ئەوانەی

ءَامَنُوٓا بڕواتان هێناوە

أَطِيعُوا گوێڕایەڵی بکەن لە

ٱللَّه خودا

وَأَطِيعُوا و گوێڕایەڵی بکەن لە

ٱلرَّسُول پێغەمبەر

وَلَاتُبۡطِلُوٓا و بەتاڵ مەکەنەوە

أَعۡمَٰلَكُمۡ کار و کردەوەکانتان

(٣٣)

إِنّ بەدڵنیایی

ٱلَّذِین ئەوانەی

كَفَرُوا کوفریانکرد و بێ بڕوابوون

وَصَدُّوا و بەربەستیان دانا

عَن لە ، لێ

سَبِيل ڕێگای

ٱللَّه خودا

ثُمّ و پاشان

مَاتُوا مردن

وَهُمۡ لەحاڵێ ئەوان

كُفَّارٞ کافر بوون

فَلَن ئەوەهەرگیز -

يَغۡفِر  (-نا) بورێ

ٱللَّه خودا

لَهُمۡ لێیان

(٣٤)

فَلَاتَهِنُوا جا بێهێزی و لاوازی نەکەن

وَتَدۡعُوٓا و بانگهێشە (مە) کەن

إِلَى بۆ

ٱلسَّلۡم ئاشتی

وَأَنتُم و ئێوە

ٱلۡأَعۡلَوۡن هەرەبەرز و بڵندن

وَٱللَّه و خودا

مَعَكُمۡ لەگەڵ ئێوەیە

وَلَن وهەرگیز -

يَتِرَكُمۡ کەم (-نا) کاتەوە

أَعۡمَٰلَكُمۡ کار و کردەوەکانتان

(٣٥)

إِنَّمَا لەڕاستیدا-

ٱلۡحَيَوٰة ژیانی

ٱلدُّنۡيَا دونیا

لَعِبٞ  (-هەر تەنها) گاڵتەکردن

وَلَهۡوٞ و رابوێری و فەرامۆشیە

وَإِن و ئەگەر

تُؤۡمِنُوا بڕواداری بکەن

وَتَتَّقُوا و لەخودابترسن و پارێزکاری بکەن

يُؤۡتِكُمۡ پێتان دەدات

أُجُورَكُمۡ پاداشتی خۆتان

وَلَايَسۡ‍َٔلۡكُمۡ و لێتان داوا ناکا

أَمۡوَٰلَكُمۡ ماڵ و سامانەکانتان

(٣٦)

إِن ئەگەر

يَسۡ‍َٔلۡكُمُوهَا لێتان داوا بکات

فَيُحۡفِكُمۡ جا پێداگری بکات لەسەرتان

تَبۡخَلُوا  (ئەوە) رژدی و رەزیلی دەکەن

وَيُخۡرِجۡ و بەدەردێخێ

أَضۡغَٰنَكُمۡ رک و قینەکانتان

(٣٧)

هَٰٓأَنتُمۡ ئەوە ئێوەن

هَٰٓؤُلَآء ئەوانێهێ

تُدۡعَوۡن بانگدەکرێن

لِتُنفِقُوا تا ببەخشن

فِي لە

سَبِيل ڕێگای

ٱللَّه خودا

فَمِنكُم ئیتر لەئێوەهەیە

مَّن کەسانێک

يَبۡخَل رژدی و ڕەزیلی دەکات

وَمَن و هەرکەسێ

يَبۡخَلۡ رژدی و ڕەزیلی بکا

فَإِنَّمَا ئیتر هەر ئەوە

يَبۡخَل رژدی و ڕەزیلی دەکات

عَن لەڕاست

نَّفۡسِهِۦ نەفسی خۆی

وَٱللَّه و خودا

ٱلۡغَنِيّ تەواو دەوڵەمەندی بێ نیازە

وَأَنتُم و ئێوەن

ٱلۡفُقَرَآء فەقیر و نەدار

وَإِن و ئەگەر

تَتَوَلَّوۡا پشتهەڵکەن

يَسۡتَبۡدِلۡ دەتانگۆڕێتەوە بە

قَوۡمًا قەومێکی

غَيۡرَكُمۡ غەیری ئێوە

ثُمّ پاشان

لَايَكُونُوٓا -

أَمۡثَٰلَكُم وەک ئێوە (-نابن)

(٣٨)
