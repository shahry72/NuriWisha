سُورَةُ المعارج 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی بەبەزەیی و میهرەبان (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


سَأَلَ -

سَائِلٌ داواکار و پرسیارکەرێک -داوا وپرسیاری کرد

بِعَذَابٍ بۆ عەزاب وئەشکەنجەیەک (کە)

وَاقِعٍ هاتووە رویداوە

(١)

لِّلْكَافِرِينَ بۆکافر ویەکخوانەپەرستان

لَيْسَ نیە

لَهُ بۆ ئەو (عەزاب) ە

دَافِعٌ(هیچ) پاڵنەر ولادەرێ

(٢)

مِّنَ لە

اللَّـهِ خودای

ذِي خاوەن

الْمَعَارِجِ بەرزی و بڵند بوونەکان

(٣)

تَعْرُجُ -

الْمَلَائِكَةُ فریشتەکان

وَالرُّوحُ و روح -بەرزدەبنەوە

إِلَيْهِ بۆلای ئەو

فِي لە

يَوْمٍ ڕۆژێک

كَانَ -

مِقْدَارُهُ بڕەکەی

خَمْسِينَ پەنجا

أَلْفَ هەزار

سَنَةٍ ساڵ -بووە

(٤)

فَاصْبِرْ جا ئارام بگرە

صَبْرًا ئارامگرتنێکی

جَمِيلًا جوان

(٥)

إِنَّهُمْ بێ گومان ئەوان

يَرَوْنَهُ وای دەبینن

بَعِيدًا(کە) دوورە

(٦)

وَنَرَاهُ و ئیمە دەیبینین

قَرِيبًا نزیکە

(٧)

يَوْمَ ڕۆژێک

تَكُونُ -

السَّمَاءُ ئاسمان

كَالْمُهْلِ وەک مسی تواوە -دەبێت

(٨)

وَتَكُونُ -

الْجِبَالُ وەچیاکان

كَالْعِهْنِ وەک خوری پەڕەکراو -دەبن

(٩)

وَلَايَسْأَلُ و پرسیارناکات

حَمِيمٌ (هیچ) دۆستێک (ی گەرم وگوڕ)

حَمِيمًا(لە) دۆستێکی (گەرم وگوڕی)

(١٠)

يُبَصَّرُونَهُمْ نیشانیان دەدرێن (ودەشیانبینن)

يَوَدُّ -

الْمُجْرِمُ ئەوەی تاوانکارە (-حەزدەکات و ئاوات دەخوازێ)

لَوْ ئەگەر ببوایە

يَفْتَدِي فیدیە بدا و بەرامبەری بکا بۆڕزگارکردنی (خۆی)

مِنْ لە

عَذَابِ عەزابی

يَوْمِئِذٍ ئەو ڕۆژەیە

بِبَنِيهِ بە كوڕەکانی

(١١)

وَصَاحِبَتِهِ وهاوسەرەکەی

وَأَخِيهِ و برایەکەی

(١٢)

وَفَصِيلَتِهِ و (کەسە نزیکەکانی) هۆزەکەی

الَّتِي ئەوەی

تُؤْوِيهِ لەخۆی دەگرن و ئەیحەوێنتەوە

(١٣)

وَمَن و هەر کەسێکە

فِي لە

الْأَرْضِ زەوی دا

جَمِيعًا هەموویان

ثُمَّ ئیتر

يُنجِيهِ دەربازی دەکا؟

(١٤)

كَلَّا نا نەخێر

إِنَّهَا بێگومان ئەوە

لَظَىٰ بڵێسەی (ئاگر) ێکە

(١٥)

نَزَّاعَةً داماڵینەرە

لِّلشَّوَىٰ بۆپێستی سەران (کە برژاو وهەلقرچاون)

(١٦)

تَدْعُو بانگی -

مَنْ ئەو کەسەی -دەکات کە

أَدْبَرَ پشتی هەڵکردوە

وَتَوَلَّىٰ و ڕووی وەرگێراوە

(١٧)

وَجَمَعَ و کۆیکردۆتەوە

فَأَوْعَىٰ ئینجا خستویەتیە ناو ئامان و دەفرێک

(١٨)

إِنَّ بێگومان

الْإِنسَانَ ئادەمیزاد

خُلِقَ دروستکراوە

هَلُوعًا بەهەلوعی (واتە:)

(١٩)

إِذَا ئەگەر

مَسَّهُ وێی کەوت

الشَّرُّ شەر و زیان

جَزُوعًا زۆر بێ هیوا پەککەوتە دەبێ

(٢٠)

وَإِذَا و ئەگەر

مَسَّهُ -

الْخَيْرُ چاکەی -وێی کەوت (ئەوە)

مَنُوعًا زۆر لێگرەوە ونەبەخش دەبێ

(٢١)

إِلَّا تەنها (ئەوانە نەبن:)

الْمُصَلِّينَ ئەو نوێژخوێنانەی

(٢٢)

الَّذِينَ کەوا

هُمْ خۆیان

عَلَىٰ لەسەر

صَلَاتِهِمْ نوێژەکانیان

دَائِمُونَ بەردەوامن

وَالَّذِينَ و ئەوانەی

فِي لە

أَمْوَالِهِمْ لە ماڵ و سامانیان دا هەیە

حَقٌّ مافێک ی

مَّعْلُومٌ دیاریکراو

(٢٤)

لِّلسَّائِلِ بۆئەوەی داواکارە

وَالْمَحْرُومِ وئەوەی لێبڕاو وبێ بەشە

(٢٥)

وَالَّذِينَ و ئەوانەی

يُصَدِّقُونَ بەهێز باوەڕدەکەن

بِيَوْمِ بە ڕۆژی

الدِّينِ پاداشت و سزا (ڕۆژی قیامەتە)

(٢٦)

وَالَّذِينَ هُم وەئەوانەی خۆیان

مِّنْ لە

عَذَابِ ئەشکەنجە و سزای

رَبِّهِم پەروەردگاریان

مُّشْفِقُونَ ئاگارداروترساوون

(٢٧)

إِنَّ بەڕاستی

عَذَابَ عەزابی

رَبِّهِمْ پەروەردگاریان

غَيْرُ -

مَأْمُونٍ جێی دڵنیایی ومتمان -نیە

(٢٨)

وَالَّذِينَ و ئەوانەی

هُمْ خۆیان

لِفُرُوجِهِمْ بۆ داوێنیان

حَافِظُونَ پارێزکارن

(٢٩)

إِلَّا تەنها

عَلَىٰ بۆ سەر

أَزْوَاجِهِمْ خێزنەکانیان

أَوْ یان

مَا مَلَكَتْ أَيْمَانُهُمْ کەنیزەک جاریەکانیان (نەبێ)

فَإِنَّهُمْ ئەوە ئەوان

غَيْرُ -

مَلُومِينَ سەرزەنشتکراو ولۆمەکراو -نین

(٣٠)

فَمَنِ جا هەرکەسێ

ابْتَغَىٰ(شتێ بخوازێ)

وَرَاءَ لەدوای

ذَٰلِكَ ئەوانە

فَأُولَـٰئِكَ هُمُ ئەوە هەر ئەوانن کە

الْعَادُونَ سنوور بەزین و دەستدرێژیکاری تەواو

(٣١)

وَ هەروەها

الَّذِينَ هُمْ ئەوانەی خۆیان

لِأَمَانَاتِهِمْ لۆ سپاردەو

وَعَهْدِهِمْ و پەیمانەکانیان

رَاعُونَ چاودێر و پارێزەرن

(٣٢)

وَالَّذِينَ هُم و ئەوانەی خۆیان

بِشَهَادَاتِهِمْ بەشایەتی دانەکانیان

قَائِمُونَ هەڵساون

(٣٣)

وَالَّذِينَ هەروەها ئەوانەی

هُمْ خۆیان

عَلَىٰ لەسەر

صَلَاتِهِمْ نوێژەکانیان

يُحَافِظُونَ پارێزگاری دەکەن

(٣٤)

أُولَـٰئِكَ ئەوان

فِي لەناو

جَنَّاتٍ باغ وباغاتێکدان

مُّكْرَمُونَ ڕێزلێنراون

(٣٥)

فَمَالِ جا چیانە ؟

الَّذِينَ ئەوانەی

كَفَرُوا بڕوایان نەهێناوە و کافرن

قِبَلَكَ بەرەو ڕووی تۆ

مُهْطِعِينَ خێرا هاتوون وچاویان لە سەر تۆیە

(٣٦)

عَنِ لە

الْيَمِينِ ڕاست

وَعَنِ و لە

الشِّمَالِ چەپ

عِزِينَ پەرت وبەشبەش بوون

(٣٧)

أَ ئایا

يَطْمَعُ -

كُلُّ هەموو

امْرِئٍ کەسێک

مِّنْهُمْ لەوان -بەتەمایە و وا دەخوازێ

أَن کە

يُدْخَلَ بخرێتە

جَنَّةَ بەهەشت ی

نَعِيمٍ پڕ لە ناز و نیعمەت

(٣٨)

كَلَّاۖ نا نەخێر (وانابێت)

إِنَّا ئێمە خۆمان

خَلَقْنَاهُم دروستمان کردوون

مِّمَّا لەوەی

يَعْلَمُونَ دەیزانن

(٣٩)

فَلَاأُقْسِمُ جا سوێند نەبێ

بِرَبِّ بە پەروەردگاری

الْمَشَارِقِ ڕۆژهەڵاتەکان

وَالْمَغَارِبِ و ڕۆژئاوایەکان

إِنَّا بێگومان ئێمە

لَقَادِرُونَ هەمیشە توانادارین

(٤٠)

عَلَىٰ لەسەر ئەوەی

أَن کە

نُّبَدِّلَ -

خَيْرًا چاكتر

مِّنْهُمْ لەوان (-بهێنین و بگۆڕینەوە)

وَمَانَحْنُ و ئێمەش -

بِمَسْبُوقِينَ(هەرگیز دەستەوەسان و عاجز) (-نین)

(٤١)

فَذَرْهُمْ جا وازبهێنە لێیان

يَخُوضُوا بەناوبکەون

وَيَلْعَبُوا و یاری بکەن

حَتَّىٰ هەتاكوو

يُلَاقُوا تووش دێن و دەگەن بە

يَوْمَهُمُ ڕۆژیان

الَّذِي ئەوەی

يُوعَدُونَ بەڵێنیان پێدەدرێت

(٤٢)

يَوْمَ ڕۆژێ

يَخْرُجُونَ دەردەچن

مِنَ لە

الْأَجْدَاثِ گۆڕەکان

سِرَاعًا خێراخێرا

كَأَنَّهُمْ هەردەڵێی ئەوان

إِلَىٰ بەرەو

نُصُبٍ بەیداغ وسەکۆگەلێک (بت و پەرستراوەکانیان)

يُوفِضُونَ خۆش ڕەوی دەکەن

(٤٣)

خَاشِعَةً -

أَبْصَارُهُمْ(لە حاڵێک کە) بینایی وچاوەکانیان (-شۆڕکردوە وداماو بووە)

تَرْهَقُهُمْ دایگرتوون وماندووی کردوون

ذِلَّةٌۚ زەلیلی و ڕیسواییەک (بەلام چ ڕیسواییەک)

ذَٰلِكَ ئا ئەوە

الْيَوْمُ ئەو ڕۆژەیە
