سَبِيلٍ ڕێیێک

(٤٦)

ٱسۡتَجِيبُوا بەدەنگ وەرن

لِرَبِّكُم بۆ پەروەدیگارتان

مِّن لە

قَبۡل پێش

أَن ئەوەی کە

يَأۡتِي بێت

يَوۡمٞ ڕۆژێ

لَّا -

مَرَدّ گەڕانەوەی (-نیە)

لَهُۥ له

مِن لە

ٱللَّه خودا

مَا لَكُم بۆتان نییە

مِّن  (هیچ)

مَّلۡجَإٖ پەنایەک

يَوۡمَئِذٖ ئەوڕۆژە

وَمَا ونیە

لَكُم بۆتان

مِّن  (هیچ)

نَّكِيرٖ نَّكِير

(٤٧)

فَإِنۡ جا ئەگەر

أَعۡرَضُوا ئەوان پشتیان هەڵکرد

فَمَآ ئەوە -

أَرۡسَلۡنَٰك  (-نە) مان ناردووە تۆ

عَلَيۡهِمۡ بۆ سەریان ، لەسەریان

حَفِيظًا پارێزەر (بیت)

إِنۡ نیە

عَلَيۡك لەسەرت ، لۆسەرت

إِلَّا جگەلە ، تەنها ، بەدەرلە

ٱلۡبَلَٰغ ڕاگەیاندن

وَإِنَّآ و بەدڵنیایی ئێمە

إِذَآ کاتێ

أَذَقۡنَا چەشتمان بە

ٱلۡإِنسَٰن مرۆڤ

مِنَّا لە ئێمەوە

رَحۡمَة بەزەیی و میهرەبانی

فَرِح دڵخۆش بوو

بِهَا پێی

وَإِن و ئەگەر

تُصِبۡهُمۡ توشیان بێت و بەریانکەوێت

سَيِّئَةُۢ خراپەیێ

بِمَا بەوەی

قَدَّمَتۡ پێشیخست و کردی

أَيۡدِيهِمۡ دەستەکانیان

فَإِنّ ئیتر ئەوە

ٱلۡإِنسَٰن مرۆڤ

كَفُورٞ زۆر ناشوکر و بێ ئەمەکە

(٤٨)

لِّلَّه هەرهی خودایە

مُلۡك خاوەنداری و پادشایەتی

ٱلسَّمَٰوَٰت ئاسمانەکان

و ٱلۡأَرۡض و زەوی

يَخۡلُق دروستی دەکات

مَا يَشَآء ئەوەی بیەوێت

يَهَب دەبەخشێت

لمَن بە ئەوەی

يَشَآء ئەو بیەوێت

إِنَٰثٗا مێینان

وَيَهَب و دەبەخشێت

لمَن بە ئەوەی

يَشَآء ئەو بیەوێت

ٱلذُّكُور نێرینان

(٤٩)

أَوۡ یان

يُزَوِّجُهُمۡ پێکەوەیان دەکات

ذُكۡرَانٗا نێرینان

وَإِنَٰثٗا و مێینان

وَيَجۡعَل و -

مَن ئەوەی

يَشَآء ئەو بیەوێت

عَقِيمًا نەزۆکی (-دەکات)

إِنَّهُۥ بەدڵنیایی ئەوە

عَلِيمٞ تەواو زانا و ئاگادار

قَدِيرٞ تەواو توانادارە

(٥٠)

وَمَا نە

كَان بووە

لِبَشَرٍ بۆ (هیچ) مرۆڤێک

أَن کە

يُكَلِّمَه بیدووێنێ

ٱللَّه خودا

إِلَّا مەگەر ، تەنها

وَحۡيًا بە وەحی ونیگا

أَوۡ یان

مِن لە

وَرَآيٕ لەپشت

حِجَابٍ پەردە و حیجابێک

أَوۡ یان

يُرۡسِل بنێرێ

رَسُولٗا پەیامبەرێک

فَيُوحِي ئیتر نیگا و سروش بکا

بِإِذۡنِهِۦ بە ئیزن و ڕێپێدانی ئەو

مَا يَشَآء ئەوەی بیەوێت

إِنَّهُۥ بەدڵنیایی ئەوە

عَلِيّ بەرز و باڵایی

حَكِيمٞ تەواو کاربەجێ (یە)

(٥١)

وَكَذَٰلِك و ئاوا ش

أَوۡحَيۡنَآ سروش و نیگامانکرد

إِلَيۡك بۆتۆ

رُوحٗا رۆح و گیانێک

مِّنۡ لە

أَمۡرِنَا فەرمانی ئێمە

مَا كُنت تۆ نە

تَدۡرِي  (نەت) دەزانی

مَا ٱلۡكِتَٰب چی یە نوسراو و نامە (ی ئاسمانی)

وَلَاٱلۡإِيمَٰن و نە بڕوا و ئیمان

وَلَٰكِن بەڵام

جَعَلۡنَٰه وامان لێکرد

نُورٗا نوور و ڕۆناکییەک (بێت)

نَّهۡدِي ڕێنمایی دەکەین

بِهِۦ پێی

مَن وما

نَّشَآء ئێمە بمانەوێ

مِنۡ لە

عِبَادِنَا بەندەکانمان

وَإِنَّك و بەدڵنیایی تۆ

لَتَهۡدِيٓ هەر رێنمونی دەکەی

إِلَى بۆ ، بۆلای

صِرَٰطٖ شا ڕێگایێکی

مُّسۡتَقِيمٖ ڕاست و ڕەوان

(٥٢)

صِرَٰط شاڕێگای

ٱللَّه خودا

ٱلَّذِي ئەوەی

لَهُۥ بۆئەو ، هی ئەوە

مَا فِي لە ، لەناو

ٱلسَّمَٰوَٰت ئاسمانەکان

وَمَا ئەوەی

فِي لە ، لەناو

ٱلۡأَرۡض زەوی

أَلَآ دەبزانن

إِلَى هەر بۆ ، بۆلای

ٱللَّه خودا

تَصِير دێتەبەربار

ٱلۡأُمُور کار و فەرمانەکان

(٥٣)
