سُورَةُ الطَّارِقِ 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبە بەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


وَالسَّمَاءِ (سوێند بە) ئاسمان

وَالطَّارِقِ و (سوێند بە) دەرکەوتەی شەو

(١)

وَمَا أَدْرَاكَ و چوزانی

مَا چی یە

الطَّارِقُ دەرکەوتەی شەو

(٢)

النَّجْمُ ئەوئەستێرەی

الثَّاقِبُ كونکەرە

(٣)

إِن نیە

كُلُّ هەموو

نَفْسٍ نفسێک

لَّمَّا الا

عَلَيْهَا لەسەریەتی

حَافِظٌ پارێزەرێک (و چاودێرێک)

(٤)

فَلْيَنظُرِ جا با سەرنج بدات

الْإِنسَانُ ئادەمیزاد

مِمَّ لە چی

خُلِقَ دروستکراوە

(٥)

خُلِقَ دروستکراوە من لە

مَّاءٍ ئاوێکی

دَافِقٍ دەرپەڕیوو

(٦)

يَخْرُجُ دەردەچێ

مِن لە

بَيْنِ نێوان

الصُّلْبِ بڕبڕەی پشت

وَالتَّرَائِبِ پەراسوەکان

(٧)

إِنَّهُ بێگومان ئەو

عَلَىٰ لەسەر

رَجْعِهِ گێڕانەوەی

لَقَادِرٌ بەتاکید توانادارە

(٨)

يَوْمَ ئەو ڕۆەی

تُبْلَى دەردەخرێت و ئاشکرا دەکرێت

السَّرَائِرُ نهێنەکان

(٩)

فَمَالَهُ ئەوسا بۆی نییە

مِن (هیچ)

قُوَّةٍ هێزێک

وَلَا و نیەتیشی

نَاصِرٍ (هیچ) پشتیوان وسەرخەرێک

(١٠)

وَالسَّمَاءِ سوێند بە ئاسمانی

ذَاتِ خاوەن

الرَّجْعِ ئەوەی دەگەڕێتەوە (باران)

(١١)

وَالْأَرْضِ و سوێند بە زەویی

ذَاتِ خاوەن

الصَّدْعِ قڵیش و درز

(١٢)

إِنَّهُ بەڕاستی ئەو (قورئان)

لَقَوْلٌ گومان نیە کە قسەیەکی

فَصْلٌ یەکلاکەرەوەیە

(١٣)

وَمَا هُوَ بِالْهَزْلِ و(بەهیچ جۆرێک) گاڵتە نیە

(١٤)

إِنَّهُمْ بێگومان ئەوان

يَكِيدُونَ پیلان دەگێڕن

كَيْدًا (چۆن) پیلانێک

(١٥)

وَأَكِيدُ و پیلان دەگێرم

كَيْدًا (چۆن) پیلانێک

(١٦)

فَمَهِّلِ جا مۆڵەت بدە

الْكَافِرِينَ بێ بڕوا ویەکخوانەپەرستان

أَمْهِلْهُمْ مۆڵەتیان بدە

رُوَيْدًا بۆ کاتێکی كورت

(١٧)
