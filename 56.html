الْحُلْقُومَ گەروو (ڕۆحی کەسەکەی لەسەرەمەرگە)

(٨٣)

وَأَنتُمْ و ئێوەش

حِينَئِذٍ ئەوکاتەی

تَنظُرُونَ سەیردەکەن

(٨٤)

وَنَحْنُ و ئێمە

أَقْرَبُ نزیکتریین

إِلَيْهِ لێی

مِنكُمْ لەنگۆ

وَلَـٰكِن بەڵام

لَّا تُبْصِرُونَ ئێوە نابینن

(٨٥)

فَلَوْلَا إِن كُنتُمْ ئەگەر ئێوە وانەبوا

غَيْرَ مَدِينِينَ لێپرسراو و بەندکراو -نەبوان

(٨٦)

تَرْجِعُونَهَا دەیگێڕنەوە

إِن ئەگەر

كُنتُمْ ئێوە-

صَادِقِينَ ڕاستگۆ(-بووبن)

(٨٧)

فَأَمَّا إِن كَانَ جا ئەگەر هاتوو ئەو-

مِنَ لە

الْمُقَرَّبِينَ نزیککراوان -بێت

(٨٨)

فَرَوْحٌ ئەوە فێنکی وفەڕاحێ و خۆشی و کامەرانی ڕاحەت

وَرَيْحَانٌ و ڕەیحان و ئیسراحەتی

وَجَنَّتُ و بەهەشتی

نَعِيمٍ پڕ ناز و نیعمەت

(٨٩)

وَأَمَّا هەرچی

إِن ئەگەر

كَانَ -

مِنْ لە

أَصْحَابِ یار و یاوەرانی(دەستەو کۆمەڵی)

الْيَمِينِ ڕاستەکان(-بوو)

(٩٠)

فَسَلَامٌ ئەوە دروود و سەلامێک

لَّكَ لەتۆ

مِنْ لە

أَصْحَابِ یار و یاوەرانی(دەستە و کۆمەڵی)

الْيَمِينِ ڕاستەکان

(٩١)

وَأَمَّا هەرچی

إِن ئەگەر

كَانَ -

مِنَ لە

الْمُكَذِّبِينَ بەدرۆ دانەرو بێ بڕوا

الضَّالِّينَ گومڕاکان -بێت

(٩٢)

فَنُزُلٌ ئەوە پێشوازی لێکردنێ ومیوانداریێ

مِّنْ بە

حَمِيمٍ ئاگراوێک

(٩٣)

وَتَصْلِيَةُ وخرانە ناو فڕێدرانێ بۆناو

جَحِيمٍ دۆزەخێک

(٩٤)

إِنَّ بەڕاستی

هَـٰذَا ئەوە

لَهُوَ هەر خۆی

حَقُّ -

الْيَقِينِ یەقینی -ڕاستی یە

(٩٥)

فَسَبِّحْ ئیتر تەسبیح بکە

بِاسْمِ بە(هێنانی) ناوی

رَبِّكَ پەروەردگاری-

الْعَظِيمِ مەزنی -خۆت

(٩٦)
