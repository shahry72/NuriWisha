بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ نوح 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕیبە بەزەیی و میهرەبان (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


إِنَّا هەر ئێمە بوین

أَرْسَلْنَا -

نُوحًا نوح (-مان ڕەوانکرد)

إِلَىٰ بۆلای

قَوْمِهِ گەلەکەی

أَنْ کەوا

أَنذِرْ بێداربکەوە و بترسێنە

قَوْمَكَ گەلەکەت

مِن لە

قَبْلِ پێش ئەوەی

أَن کە

يَأْتِيَهُمْ بۆیان بێت

عَذَابٌ عەزاب و ئەشکەنجەیەکی

أَلِيمٌ زۆربە ئازار

(١)

قَالَ (نوح) گوتی

يَا ئەی

قَوْمِ گەلەکەم

إِنِّي دڵنیابن من

لَكُمْ بۆئێوە

نَذِيرٌ ترسێنەر و بێدارکەرەوەیەکی

مُّبِينٌ ڕۆن و ئاشكرام

(٢)

أَنِ کە

اعْبُدُوا -

اللَّـهَ خودا (-بپەرستن)

وَاتَّقُوهُ و لەو بترسن و خۆ بپارێزن لێی

وَأَطِيعُونِ و گوێڕایەڵ بن بۆم

(٣)

يَغْفِرْ (ئەوساکە خودا) خۆشدەبێت

لَكُم بۆتان

مِّن لە

ذُنُوبِكُمْ گوناهەکانتان

وَيُؤَخِّرْكُمْ و دواتان دەخات

إِلَىٰ تاكوو

أَجَلٍ ماوەیەک ی

مُّسَمًّىۚ دیاریکراو

إِنَّ بێگومان

أَجَلَ ماوەی (دیاریکراوی)

اللَّـهِ خودا

إِذَا کە

جَاءَ هات

لَايُؤَخَّرُ دواناخرێت

لَوْ كُنتُمْ تَعْلَمُونَ ئەگەر ئێوە بزانن

(٤)

قَالَ (نوح) گوتی

رَبِّ پەروەردیگارە

إِنِّي ئەمن

دَعَوْتُ بانگم کرد

قَوْمِي گەلەکەم

لَيْلًا بەشەوێ

وَنَهَارًا و بەڕٶژێ

(٥)

فَلَمْ يَزِدْهُمْ جا چی بۆ زیادنەکردن

دُعَائِي بانگکردنم

إِلَّا تەنها

فِرَارًا ڕاکردن ودورکەوتنەوە (نەبێ)

(٦)

وَإِنِّي و بەڕاستی من

كُلَّمَا هەرکاتێ

دَعَوْتُهُمْ ئەوانم بانگکرد

لِتَغْفِرَ تاكو خۆشبی

لَهُمْ لێیان

جَعَلُوا -

أَصَابِعَهُمْ پەنجەکانیان (-کردە)

فِي ناو

آذَانِهِمْ گوێکانیان

وَاسْتَغْشَوْا و دایان پۆشی وبەسەرخۆیان دادا

ثِيَابَهُمْ جلەکانیان

وَأَصَرُّوا و پێداگریان کرد

وَاسْتَكْبَرُوا و خۆیان بە زل زانی و لووتبەرزبون

اسْتِكْبَارًا (چۆن) لووت بەرزیێ

(٧)

ثُمَّ پاشان

إِنِّي من خۆم

دَعَوْتُهُمْ بانگاوازم کرد و بانگم کردن

جِهَارًا بەئاشكرایی

(٨)

ثُمَّ پاشان

إِنِّي هەرخۆم

أَعْلَنتُ ئاشكرام کرد (بانگەوازەکەم)

لَهُمْ بۆیان

وَأَسْرَرْتُ و پەنهانیشم کرد

لَهُمْ بۆیان

إِسْرَارًا پەنهایەک (چۆن پەنهانیێ)

(٩)

فَقُلْتُ جا گوتم

اسْتَغْفِرُوا داوای لێخۆشبوون بکەن لە

رَبَّكُمْ پەروەردیگارتان

إِنَّهُ بەدڵنیایی ئەو

كَانَ -

غَفَّارًا زۆر لیبوردە و لێخۆشبوو (-بووە)

(١٠)

يُرْسِلِ -

السَّمَاءَ ئاسمان (-دەنێرێ)

عَلَيْكُم بەسەرتانا

مِّدْرَارًا باران لەدوای باران

(١١)

وَيُمْدِدْكُم وەبۆتان دەنێرێ

بِأَمْوَالٍ لە ماڵ

وَبَنِينَ ومناڵ وكوڕان

وَيَجْعَل و -

لَّكُمْ بۆتان (-فەراهەمدێنێ)

جَنَّاتٍ باخاتێک (چۆن باخاتێ)

وَيَجْعَل و -

لَّكُمْ بۆتان (-فەراهەمدێنێ)

أَنْهَارًا چەندەها ڕووبار

(١٢)

مَّالَكُمْ ئەوەچیتانە بۆ

لَاتَرْجُونَ ناخوازن

لِلَّـهِ (لە -بۆ) خودا

وَقَارًا ڕێزو شکۆیەک

(١٣)

وَقَدْ خَلَقَكُمْ لەو کاتەی کە ئێوەی -

أَطْوَارًا قۆناغ بە قۆناغ (- خولقاندوە)

(١٤)

أَلَمْ تَرَوْا ئەی نەوبینیوە

كَيْفَ چۆن

خَلَقَ -

اللَّـهُ خودا

سَبْعَ حەفت

سَمَاوَاتٍ ئاسمانان ی (- خولقاندوە)

طِبَاقًا چین لەسەرچین

(١٥)

وَجَعَلَ -

الْقَمَرَ و مانگی

فِيهِنَّ تیایاندا ((-کردۆتە))

نُورًا نووڕ و ڕۆناکی

وَجَعَلَ -

الشَّمْسَ وەخۆری ((-کردۆتە))

سِرَاجًا چرایەکی داگیرساو

(١٦)

وَاللَّـهُ و خودا

أَنبَتَكُم ئێوەی -

مِّنَ لە

الْأَرْضِ زەوی (-ڕواندووە)

نَبَاتًا ڕواندنێ

(١٧)

ثُمَّ پاشان

يُعِيدُكُمْ دەتانگێڕتەوە

فِيهَا لێی/لۆی

وَيُخْرِجُكُمْ و دەرتان دەهێنتەوە

إِخْرَاجًا دەرهێنانێ

(١٨)

وَاللَّـهُ و خودا

جَعَلَ -

لَكُمُ بۆتان

الْأَرْضَ زەوی (-کردۆتە)

بِسَاطًا ڕایەخێ

(١٩)

لِّتَسْلُكُوا تا بگرنەبەر

مِنْهَا لێی

سُبُلًا چەندین ڕێ ی

فِجَاجًا فرەوان و کراوە

(٢٠)

قَالَ -

نُوحٌ نوح (-گوتی)

رَّبِّ پەروەردیگارە

إِنَّهُمْ بەڕاستی ئەوان

عَصَوْنِي بەقسەیان نەکردم و لێم یاخی بوون

وَاتَّبَعُوا وکەوتنە دوای

مَن ئەوکەسەی

لَّمْ يَزِدْهُ چی بۆ زیاد نەکردوە

مَالُهُ ماڵەکەی

وَوَلَدُهُ و مناڵەکانی

إِلَّا تەنها

خَسَارًا خەسارەتمەندی(نەبێ)

(٢١)

وَمَكَرُوا و پیلانیان گێڕا

مَكْرًا پیلانێکی

كُبَّارًا هەرەگەورە

(٢٢)

وَقَالُوا و گوتیان

لَا تَذَرُنَّ قەت وازمەهێنن لە

آلِهَتَكُمْ پەرستراو و خوداکانتان

وَلَاتَذَرُنَّ وهەرگیز واز مەهێنن (نە) لە

وَدًّا وەدد

وَلَاسُوَاعًا و نە سواع

وَلَايَغُوثَ ونە یەغوث

وَيَعُوقَ ویەعوق

وَنَسْرًا ونەسر

(٢٣)

وَقَدْ و بەڕاستی

أَضَلُّوا -

كَثِيرًا زۆرێک (لە خەلک) یان (-گومڕاکرد)

وَلَاتَزِدِ و (خودایە) زیاد نەکەیت بۆ

الظَّالِمِينَ ستەمکارەکان هیچ

إِلَّا تەنها

ضَلَالًا گومڕایێ

(٢٤)

مِّمَّا خَطِيئَاتِهِمْ لەناو تاوانەکانیاندا و بەهۆیەوە

أُغْرِقُوا نقم كران

فَأُدْخِلُوا جا خرانە ناو

نَارًا ئاگرێ (چۆن ئاگرێک)

فَلَمْ يَجِدُوا جا نەیاندۆزیەوە و دەستیان نەکەوەت

لَهُم بۆخۆیان

مِّن لە

دُونِ غەیری

اللَّـهِ خودا

أَنصَارًا (هیچ) پشتیوانانێ

(٢٥)

وَقَالَ -

نُوحٌ نوح (-گوتی)

رَّبِّ پەروەردیگارە

لَاتَذَرْ -

عَلَى لەسەر

الْأَرْضِ زەوی (-مەهێڵەوە)

مِنَ لە

الْكَافِرِينَ کافرەکان

دَيَّارًا هاموشۆکارێک (هیچ کەسێک)

(٢٦)

إِنَّكَ بە دڵنیایی تۆ

إِن ئەگەر

تَذَرْهُمْ وازیان لێبینی و بیانهێڵیەوە

يُضِلُّوا ئەوە گومڕا دەکەن

عِبَادَكَ بەندەکانت

وَلَايَلِدُوا و نایانبێت ولێیان ناکەوێتەوە

إِلَّا تەنها

فَاجِرًا تاوانکار و خراپەکارێکی

كَفَّارًا زۆر کافر

(٢٧)

رَّبِّ پەروەردیگارە

اغْفِرْ خۆش ببە

لِي لێم

وَ لِوَالِدَيَّ و لە هەردوو باوانم (دایک و باوكم)

وَلِمَن و لەهەر کەسێک

دَخَلَ هاتبێتە

بَيْتِيَ ماڵم

مُؤْمِنًا بە ئیمانداری

وَلِلْمُؤْمِنِينَ وَالْمُؤْمِنَاتِ ولە بڕوادارانی نێرو مێ

وَلَاتَزِدِ و زیاد مەکە بۆ

الظَّالِمِينَ زاڵم و ستەمکارەکان (هیچ)

إِلَّا تەنها

تَبَارًا تیاچوون وخەسارەتمەندی و وێرانکارێ (وێنەی نەبێت)

(٢٨)
