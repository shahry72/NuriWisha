سُورَةُ الطلَاق 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبە بەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


يَا أَيُّهَا هۆ ئەی

النَّبِيُّ پێغەمبەر

إِذَا کاتێک

طَلَّقْتُمُ طەڵاقتاندا

النِّسَاءَ ئافرەتەکان

فَطَلِّقُوهُنَّ ئەوە طەڵاقیانبدەن

لِعِدَّتِهِنَّ لە عیددەیان

وَأَحْصُوا و ئەژمار بکەن

الْعِدَّةَ ۖ عیددەکە

وَاتَّقُوا و پارێزکاری بکەن و بترسن لە

اللَّـهَ خودای

رَبَّكُمْ ۖ پەروەردیگارتان ،

لَا تُخْرِجُوهُنَّ دەریان مەکەن

مِن لە

بُيُوتِهِنَّ ماڵەکانیان

وَلَا يَخْرُجْنَ و با ئەوانیش دەرنەچن

إِلَّا مەگەر

أَن ئەوەی کە

يَأْتِينَ -

بِفَاحِشَةٍ کارێکی قێزەوەنی

مُّبَيِّنَةٍ ئاشكرا -بهێنن (بکەن)

وَتِلْكَ وئەوەش

حُدُودُ سنوورەکانی

اللَّـهِ خودا یە

وَمَن وهەرکەسێ

يَتَعَدَّ ببەزێنێ

حُدُودَ سنوورەکانی

اللَّـهِ خودا

فَقَدْ ئەوە بەدڵنیایی و بە تەحقیق

ظَلَمَ ستەم و زوڵمی کردوە لە

نَفْسَهُ خۆی ،

لَا تَدْرِي تۆ نازانی

لَعَلَّ بەڵكو

اللَّـهَ خودا

يُحْدِثُ وابکات-

بَعْدَ پاش

ذَٰلِكَ ئەوە

أَمْرًا شتێک -ڕوبدات

(١)

فَإِذَا جا کاتێک

بَلَغْنَ (ئافرەتەکان) گەشتنە

أَجَلَهُنَّ ماوە دیاریکراوەکەیان

فَأَمْسِكُوهُنَّ ئەوە ڕایانگرن و بیانهێڵنەوە

بِمَعْرُوفٍ بە باشی

أَوْ یان

فَارِقُوهُنَّ لێیان جیا ببنەوە

بِمَعْرُوفٍ بە باشی

وَأَشْهِدُوا وبکەنە شایەت

ذَوَيْ دوو (کەسی) خاوەن

عَدْلٍ دادپەروەری

مِّنكُمْ لەخۆتان

وَأَقِيمُوا -

الشَّهَادَةَ وشایەتیەکە -بەڕێکی ئەنجام بدەن

لِلَّـهِ بۆ خودا ،

ذَٰلِكُمْ ئەوەیە

يُوعَظُ ئامۆژگاری دەکرێت

بِهِ پێی

مَن کەسێک

كَانَ -

يُؤْمِنُ بڕوای -بوبێت

بِاللَّـهِ بە خودا

وَالْيَوْمِ و ڕۆژی

الْآخِرِ دوایی

وَمَن و هەرکەسێ

يَتَّقِ پارێزکاری بکاو بترسێت لە

اللَّـهَ خودا

يَجْعَل (ئەوە خودا) دادەنێت

لَّهُ بۆی

مَخْرَجًا دەروویەک (دەرچەیەک)

(٢)

وَيَرْزُقْهُ و ڕیزق وبژێوی دەدات

مِنْ لە

حَيْثُ لایەک و بەجۆرێک

لَا يَحْتَسِبُ خەیاڵی بۆ نەچووبێت

وَمَن و هەرکەس

يَتَوَكَّلْ پشت ببەستێ وخۆ بسپێرێ

عَلَى بە

اللَّـهِ خودا

فَهُوَ ئەوە خودا

حَسْبُهُ بەسیەتی ،

إِنَّ بەدڵنیایی

اللَّـهَ خودا

بَالِغُ -

أَمْرِهِ کاری -ئەنجامدراوە

قَدْ بەدڵنیایی و بە تەحقیق

جَعَلَ -

اللَّـهُ خودا

لِكُلِّ بۆ هەموو

شَيْءٍ شتێک

قَدْرًا ئه ندازە و بڕێکی -داناوە

(٣)

وَاللَّائِي و ئەو ئافرەتانەی

يَئِسْنَ بێ ئومێد بوینە

مِنَ لە

الْمَحِيضِ حەیز

مِن لە

نِّسَائِكُمْ ژنەکانتان

إِنِ ئەگەر

ارْتَبْتُمْ گومانتانکرد و دڵنیا نەبوون

فَعِدَّتُهُنَّ ئەوە عیددەیان

ثَلَاثَةُ سێ

أَشْهُرٍ مانگە

وَاللَّائِي وەئەوانەی

لَمْ يَحِضْنَ حەیزیان نەکردوە

وَأُولَاتُ و ئەوانەشی خاوەن

الْأَحْمَالِ سکپڕینە (دووگیانن)

أَجَلُهُنَّ کاتی دیاریکراویان ئەوەیە

أَن کە

يَضَعْنَ دانێن

حَمْلَهُنَّ سکەکانیان

وَمَن وهەرکەسێ

يَتَّقِ پارێزکاربێت و بترسێ لە

اللَّـهَ خودا

يَجْعَل (ئەوە خودا) دابین دەکات

لَّهُ بۆی

مِنْ لە

أَمْرِهِ کارەکەی

يُسْرًا ئاسانیەک

(٤)

ذَٰلِكَ ئەوە

أَمْرُ فەرمانی

اللَّـهِ خودایە

أَنزَلَهُ دایبەزاندووە

إِلَيْكُمْ بۆتان

وَمَن و هەر کەسێک

يَتَّقِ پارێزکاری بکات و بترسێ لە

اللَّـهَ خودا

يُكَفِّرْ (ئەوە خودا) چاوپۆشی-

عَنْهُ لێ-دەکات

سَيِّئَاتِهِ لە خراپەکانی

وَيُعْظِمْ و گەورە و زیاد دەکات

لَهُ بۆی

أَجْرًا پاداشت

(٥)

أَسْكِنُوهُنَّ نیشتەجێیان بکەن

مِنْ لە

حَيْثُ هەرشوێنێ

سَكَنتُم (خۆتان) نیشتەجێی بوون

مِّن لەوەی

وُجْدِكُمْ دەستانکەوتووە

وَلَا تُضَارُّوهُنَّ و زەرەریان لێمەدەن

لِتُضَيِّقُوا تا تەنگ بکەن

عَلَيْهِنَّ لەسەریان

وَإِن و ئەگەر

كُنَّ ئەوان

أُولَاتِ خاوەنی

حَمْلٍ سکپڕی بوون

فَأَنفِقُوا ئەوە مەسرەف بکەن

عَلَيْهِنَّ لێیان

حَتَّىٰ هەتاكوو

يَضَعْنَ دادەنێن

حَمْلَهُنَّ سکەکانیان

فَإِنْ جا ئەگەر

أَرْضَعْنَ (مناڵەکانیان) شیردا

لَكُمْ بۆتان

فَآتُوهُنَّ ئەوە بیاندەنێ

أُجُورَهُنَّ ۖ پاداشتی خۆیان

وَأْتَمِرُوا و فەرمان بکەن

بَيْنَكُم لەنێوخۆتاندا

بِمَعْرُوفٍ ۖ بە چاکەیەک

وَإِن و ئەگەر

تَعَاسَرْتُمْ زەحمەتی کەوتە نێوانتان و ناکۆک بوون

فَسَتُرْضِعُ لَهُ ئەوە-

أُخْرَىٰ ئافرەتێکی تر -شیری پێدەدات

(٦)

لِيُنفِقْ با مەسرەف بکات

ذُو سَعَةٍ خاوەن هەبوونی (دەوڵەمەندە)

مِّن لە

سَعَتِهِ ۖ هەبونیەکەی

وَمَن و ئەوەی

قُدِرَ تەنگکراوە

عَلَيْهِ لەسەری

رِزْقُهُ بژێویەکەی

فَلْيُنفِقْ با مەسرەف بکا

مِمَّا لەوەی

آتَاهُ پێیداوە

اللَّـهُ خودا

لَا يُكَلِّفُ -

اللَّـهُ خودا -ئەرك ناخاتە سەر

نَفْسًا هیچ کەسێک

إِلَّا تەنها

مَا ئەوەندە (نەبێ کە)

آتَاهَا پێیداوە

سَيَجْعَلُ بەدڵنیایی-

اللَّـهُ خودا

بَعْدَ پاش

عُسْرٍ (هەر) زەحمەتیەک

يُسْرًا ئاسانیەک -دێنێ

(٧)

وَكَأَيِّن چەندەها

مِّن قَرْيَةٍ دێهات

عَتَتْ یاخی بوون

عَنْ لە

أَمْرِ فەرمانی

رَبِّهَا پەروەردگاریان

وَرُسُلِهِ وپێغەمبەرانی

فَحَاسَبْنَاهَا ئەوەبوو پرسینەوەمان لەگەڵ کردن

حِسَابًا پرسینەوە

شَدِيدًا زۆر تووند

وَعَذَّبْنَاهَا و عەزاب و ئەشکەنجەماندان

عَذَابًا عەزاب و ئەشکەنجەیەکی

نُّكْرًا زۆر سەخت ودژوار (شتی وا نەبینرابێت) .

(٨)

فَذَاقَتْ ئەوسا چەشتی

وَبَالَ ئۆباڵی

أَمْرِهَا کارەکانی

وَكَانَ وە-

عَاقِبَةُ سەرئەنجامی

أَمْرِهَا کارەکانی

خُسْرًا زیان و خەسارەتمەندی(-بوو)

(٩)

أَعَدَّ -

اللَّـهُ خودا -ئامادەی کردوە

لَهُمْ بۆیان

عَذَابًا عەزاب و ئەشکەنجەیەکی

شَدِيدًا ۖ زۆر تووند ،

فَاتَّقُوا ئیتر پارێزکاربن و بترسن لە

اللَّـهَ خودا

يَا ئەی

الْأَلْبَابِ عەقل و هۆشەکان

آمَنُوا بڕواتان هێناوە

قَدْ ئەوەتە

أَنزَلَ -

اللَّـهُ خودا -دایبەزاندووە

إِلَيْكُمْ بۆتان

ذِكْرًا یادکەرەوەیەک (کە قورئانە)

يَتْلُو دەخوێنتەوە

عَلَيْكُمْ بەسەرتاندا

آيَاتِ بەڵگەو و ئایاتەکانی

مِنَ لە

جَنَّاتٍ باخاتێک

تَجْرِي تێدەپەڕێ

مِن لە

تَحْتِهَا ژێریدا

خَالِدِينَ هەتاهەتایێ

فِيهَا تیایدان
