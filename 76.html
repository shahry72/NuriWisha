بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ الإنسان 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی بەبەزەیی و میهرەبان (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


هَلْ ئایا

أَتَىٰ هاتۆتە

عَلَى سەر

الْإِنسَانِ ئینسان

حِينٌ کاتێک

مِّنَ لە

الدَّهْرِ ڕۆژگاروزەمانە

لَمْ يَكُن -

شَيْئًا شتێک -نەبوو

مَّذْكُورًا باسکراو بێت

(١)

إِنَّا بێگومان ئێمە

خَلَقْنَا وەدیمان هێنا

الْإِنسَانَ ئینسان

مِن لە

نُّطْفَةٍ تنۆکێکی

أَمْشَاجٍ ئاوێتە

نَّبْتَلِيهِ ژێرئەزموونی دەخەین

فَجَعَلْنَاهُ جا کردمان بە

سَمِيعًا بیسەرێکی

بَصِيرًا بینەر

(٢)

إِنَّا ئێمە

هَدَيْنَاهُ ڕێنمونمان کرد بۆ

السَّبِيلَ ڕێگاکە

إِمَّا یا

شَاكِرًا سوپاس گوزارە

وَإِمَّا و یا

كَفُورًا نا شوكری سپڵەیە

(٣)

إِنَّا بەدڵنیایی ئێمە

أَعْتَدْنَا ئامادەمان کردوە

لِلْكَافِرِينَ بۆ کافر و یەکخودانەپەرستان

سَلَاسِلَ چەندەها زنجیر

وَأَغْلَالًا وچەندها کۆت

وَسَعِيرًا و ئاگرێکی جۆشدراو

(٤)

إِنَّ بە تەکید

الْأَبْرَارَ چاك وپاکان

يَشْرَبُونَ دەنۆشن

مِن لە

كَأْسٍ کاسەیەک

كَانَ -

مِزَاجُهَا تێکەڵکراوەکەی

كَافُورًا کافوورێک (-بوو)

(٥)

عَيْنًا کانیەک

يَشْرَبُ دەخۆنەوە

بِهَا لێی

عِبَادُ بەندەکانی

اللَّـهِ خودا

يُفَجِّرُونَهَا دەیتەقێننەوە و هەڵیدەقوڵێنن

تَفْجِيرًا (چۆن) تەقاندنەوەیەک

(٦)

يُوفُونَ بەتەواوی بەجێدەگەیەنن ووەفا دەکەن بە

بِالنَّذْرِ نەزرەکان (لەخۆگرتنەکان)

وَيَخَافُونَ و دەترسن لە

يَوْمًا ڕۆژێک

كَانَ شَرُّهُ شەڕ وبەڵاکەی -

مُسْتَطِيرًا بەئاسمانا چوو (-بوو)

(٧)

وَيُطْعِمُونَ الطَّعَامَ و خواردن دەدەن-

عَلَىٰ سەرەڕای

حُبِّهِ حەز لێکردنی -بە

مِسْكِينًا کەم دەرامەت

وَيَتِيمًا وهەتیوو

وَأَسِيرًا و بەدیل گیراو

(٨)

إِنَّمَا -

نُطْعِمُكُمْ .. ئەوەی کە .. خواردنتان دەدەین -هەر تەنها

لِوَجْهِ اللَّـهِ لە ڕاهی خودایە

لَانُرِيدُ نامانەوێت

مِنكُمْ لە ئێوە

جَزَاءً (نە) پاداشتێک

وَلَا و نە

شُكُورًا (هیچ) شوكرانەبژێریەکیش

(٩)

إِنَّا ئێمە

نَخَافُ دەترسین

مِن لە

رَّبِّنَا پەروەردیگارمانەوە

يَوْمًا ڕۆژێکی

عَبُوسًا گرژی

قَمْطَرِيرًا ترش وتاڵ و دژوار

(١٠)

فَوَقَاهُمُ -

اللَّـهُ جا خودا -پاراستنی وبەدووری گرتن لە

شَرَّ ناخۆشی ونەهامەتی

ذَٰلِكَ ئەوێ

الْيَوْمِ ڕٶژێ

وَلَقَّاهُمْ پێی گەیاندن

نَضْرَةً گەشاوەیەک

وَسُرُورًا ودڵخۆشیەک (هەرخودا دەزانێت چ دڵ خۆشیەکە)

(١١)

وَجَزَاهُم و پاداشتی دانەوە

بِمَا بەهۆی ئەوەی

صَبَرُوا ئارامیانگرت-

جَنَّةً بە بەهەشتێک

وَحَرِيرًا و ئاوریشمێک (هەرخودا دەزانێت چەند خۆش ونایابە)

(١٢)

مُّتَّكِئِينَ پاڵیانداوەتەوە

فِيهَا تیایدا

عَلَى لەسەر

الْأَرَائِكِ ۖ تەخت و قەنەفان

لَايَرَوْنَ نابینن

فِيهَا تیایدا

شَمْسًا (نە(هیچ)) ڕۆژێک

وَلَا و نە

زَمْهَرِيرًا سەرماوسۆڵی زەمهەریر

(١٣)

وَدَانِيَةً و نزیک و نزمە

عَلَيْهِمْ بە سەریاندا

ظِلَالُهَا سایە و سێبەری

وَذُلِّلَتْ و ئاسان وشۆڕ کراوە

قُطُوفُهَا هێشووی میوەکانی

تَذْلِيلًا (بەڵام) شۆڕبوونەوە

(١٤)

وَيُطَافُ و دەگێردرێ

عَلَيْهِم بە سەریاندا

بِآنِيَةٍ دەفرو سوراحیەک

مِّن لە

فِضَّةٍ زیووێک (چۆن زیوێک)

وَأَكْوَابٍ و گەلێک كووپ

كَانَتْ -

قَوَارِيرَا لەشوشە وكریستالێک (-بوو)

(١٥)

قَوَارِيرَ شوشەیەک

مِن لە

فِضَّةٍ زیووێک

قَدَّرُوهَا ئەندازیانکرد و سازاندویانە

تَقْدِيرًا سازدانێک (هەرخودا بزانێت چۆنە)

(١٦)

وَيُسْقَوْنَ و پێیان دەنۆشن

فِيهَا تیایدا

كَأْسًا کاسەیەک

كَانَ -

مِزَاجُهَا ئاویتەکەی

زَنجَبِيلًا زەنجەبیلێک (-بوو)

(١٧)

عَيْنًا (لە) کانیاوێک

فِيهَا تیایدا

تُسَمَّىٰ پێی دەوترێ

سَلْسَبِيلًا سەلسەبیل

(١٨)

وَيَطُوفُ و دەگەڕێت

عَلَيْهِمْ بەسەریاندا

وِلْدَانٌ كوڕیژگەی

مُّخَلَّدُونَ نەمر

إِذَا ئەگەر

رَأَيْتَهُمْ ئەوانت دیت

حَسِبْتَهُمْ وایان خەیاڵدەکەی ئەوان

لُؤْلُؤًا مرواری

مَّنثُورًا پڕژێندراون

(١٩)

وَإِذَا و ئەگەر

رَأَيْتَ بینیت

ثَمَّ لێرە و لەوێ (بەهەر كوێێکدا سەیر بکەی)

رَأَيْتَ (دەبینی)

نَعِيمًا خۆشی وناز و نیعمەتێک

وَمُلْكًا وەدەسەڵاتێک (کە بەڕاستی)

كَبِيرًا گەورەیە

(٢٠)

عَالِيَهُمْ بەسەریانەوەیە

ثِيَابُ پۆشاکی

سُندُسٍ تەنکی ئاوریشمێک

خُضْرٌ کە سەوزە

وَإِسْتَبْرَقٌ وپۆشاکێکی ئاوریشمی ئەستوور

وَحُلُّوا و دەڕازێندرێنەوە بە

أَسَاوِرَ بازنی

مِن لە

فِضَّةٍ زیوو

وَسَقَاهُمْ و پێیان دەنۆشی

رَبُّهُمْ پەروەردگاریان

شَرَابًا شەڕابێکی

طَهُورًا پاكکەرەوە

(٢١)

إِنَّ بەڕآستی

هَـٰذَا ئەمە

كَانَ -

لَكُمْ بۆتان

جَزَاءً پاداشت (-بوو)

وَكَانَ -

سَعْيُكُم و هەوڵ و کۆششتان

مَّشْكُورًا سوپاسکراو (-بوو)

(٢٢)

إِنَّا بێگومان ئێمە

نَحْنُ خۆمان

نَزَّلْنَا دامانبەزاند

عَلَيْكَ بەسەرتا

الْقُرْآنَ قورئان

تَنزِيلًا دابەزاندێک (بەڵام دابەزاندنێک)

(٢٣)

فَاصْبِرْ جا ئارام بگرە

لِحُكْمِ لۆ بڕیاری

رَبِّكَ پەروەردیگارت

وَلَاتُطِعْ و گوێمەدە

مِنْهُمْ کەسێک لەوان

آثِمًا تاوانبارێک (بێت)

أَوْ یان

كَفُورًا زۆر ناشوكر و کافر

(٢٤)

وَاذْكُرِ و یادبکەوە

اسْمَ ناوی

رَبِّكَ پەروەردیگارت

بُكْرَةً بەیانیان

وَأَصِيلًا و ئێواران

(٢٥)

وَمِنَ وەلە

اللَّيْلِ شەویشدا

فَاسْجُدْ سوژدە بەرە

لَهُ لۆی

وَسَبِّحْهُ و تەسبیحات بکە (خودا بە بەرز و پیرۆزو بێ عەیبناوبێنە)

لَيْلًا شەوێکی

طَوِيلًا درێژ

(٢٦)

إِنَّ بەڕاستی

هَـٰؤُلَاءِ ئەوانێهێ

يُحِبُّونَ حەزدەکەن لە

الْعَاجِلَةَ دونیا (ئەو شتەی کە ئێستا لەبەردەستانە)

وَيَذَرُونَ وەپشتگوێدەخەن و بەجێدێڵن

وَرَاءَهُمْ لە دوایان

يَوْمًا ڕۆژێکی

ثَقِيلًا قوڕس

(٢٧)

نَّحْنُ ئێمە

خَلَقْنَاهُمْ ئەوانمان بەدیهێناوە

وَشَدَدْنَا و توندوتۆڵمان کرد

أَسْرَهُمْ بەسترانیان

وَإِذَا ئەگەر

شِئْنَا ویستمان

بَدَّلْنَا (ئەوە دەگۆڕین)

أَمْثَالَهُمْ نمونەیان

تَبْدِيلًا گۆڕینێک چۆن گۆڕینێک

(٢٨)

إِنَّ بە دڵنیایی

هَـٰذِهِ ئەمە

تَذْكِرَةٌ بیرخەرەوەیەک

فَمَن جا هەر کەسێک

شَاءَ ویستی

اتَّخَذَ -

إِلَىٰ بەرەو

رَبِّهِ پەروەردگاری

سَبِيلًا ڕیگایەک (-دەگرێتەبەر)

(٢٩)

وَمَا تَشَاءُونَ و ناتانەوێ

إِلَّا ((هیچ) شتێک) مەگەر

أَن کە

يَشَاءَ -

اللَّـهُ خودای پەروەردیگار (-بیەوێت)

إِنَّ بێگومان

اللَّـهَ خودا

كَانَ عَلِيمًا هەر تەواوزاناو

حَكِيمًا تەواودانا (-بووە)

(٣٠)

يُدْخِلُ -

مَن هەر کەسێک

يَشَاءُ بیەوێت -ئەیخاتە

فِي ناو

رَحْمَتِهِ ڕەحمەتی خۆی

وَالظَّالِمِينَ و زاڵمەکانیش

أَعَدَّ (خودا) ئامادەیکردوە

لَهُمْ بۆیان

عَذَابًا عەزاب و ئەشکەنجەیەکی

أَلِيمًا زۆربەئازار و بە ژان

(٣١)
