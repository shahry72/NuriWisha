ٱلۡجَنَّة بەهەشت

ٱلَّتِيٓ ئەوەی

أُورِثۡتُمُوهَا بۆتان ماوەتەوە

بِمَا بەوەی

كُنتُمۡ تَعۡمَلُون پێشان دەتانکرد

(٧٢)

لَكُمۡ بۆتان هەیە

فِيهَا ‌لێی ، تیایدا

فَٰكِهَةٞ میوەی

كَثِيرَةٞ زۆر

مِّنۡهَا لێی

تَأۡكُلُون دەخۆن

(٧٣)

إِنّ بەدڵنیایی

ٱلۡمُجۡرِمِين ستەمکارەکان

فِي لە ، لەناو

عَذَاب عەزاب وئەشکەنجەی

جَهَنَّم دۆزەخ

خَٰلِدُون ماوەن هەتا هەتایێ

(٧٤)

لَايُفَتَّر سوک ناکرێ

عَنۡهُمۡ لەسەریان

وَهُمۡ و ئەوان

فِيه تیایدا

مُبۡلِسُون بێ ئومێد کەوتوون

(٧٥)

وَمَا و-

ظَلَمۡنَٰهُمۡ زوڵم و ستەممان لێ(-نە) کردوون

وَلَٰكِن بەڵام

كَانُوا -

هُم خۆیان

ٱلظَّٰلِمِين ستەمکار (-بوون)

(٧٦)

وَنَادَوۡا و بانگیانکرد

يَٰمَٰلِك ئەی مالک (سەرپەشتیاری دۆزەخ)

لِيَقۡض با کۆتایی بهێنێ

عَلَيۡنَا پێمان (بمانمرێنێ)

رَبُّك پەروەردیگارت

قَال گوتی

إِنَّكُم بەدڵنیایی ئێوە

مَّٰكِثُون ماوەن

(٧٧)

لَقَدۡ و بەدڵنیایی

جِئۡنَٰكُم هێنامان بۆتان

بِٱلۡحَقّ حەق و ڕاستی و ڕەوایی

وَلَٰكِنّ بەڵام

أَكۡثَرَكُمۡ زۆربەتان

لِلۡحَقّ بۆحەق

كَٰرِهُون ناحەز و نەیارن

(٧٨)

أَمۡ یان

أَبۡرَمُوٓا دایانڕشتووە وکۆدەنگبوون لەسەر

أَمۡرٗا کار و فەرمانێک (پیلانێک)

فَإِنَّا بەدڵنیایی ئێمە

مُبۡرِمُون داڕێژەرین

(٧٩)

أَمۡ یان

يَحۡسَبُون وا حساب دەکەن

أَنَّا ئێمە

لَانَسۡمَع نابیسین

سِرَّهُمۡ نهێنیەکانیان

وَنَجۡوَىٰهُم و چپەچپ و قسەی نێوانیان

بَلَى بەڵێ(دەبیسین)

وَرُسُلُنَا و ناردراوەکانمان(فریشتەکان)

لَدَيۡهِمۡ لایان

يَكۡتُبُون دەنووسن

(٨٠)

قُلۡ بڵێ

إِن ئەگەر

كَان بووبێت

لِلرَّحۡمَٰن بۆ خودای زۆربە بەزەیی و میهرەبان

وَلَدٞ کوڕێک

فَأَنَا۠ ئەوە من

أَوَّل یەکەمی

ٱلۡعَٰبِدِين پەرستەکانم

(٨١)

سُبۡحَٰن پاکی و بێگەردی بۆ

رَبّ پەروەردیگار و خاوەنی

ٱلسَّمَٰوَٰت ئاسمانەکان

و ٱلۡأَرۡض و زەوی

رَبّ پەروەردیگار و خاوەنی

ٱلۡعَرۡش عەرش و تەختی پادشایەتی

عَمَّا لەوەی

يَصِفُون ئەوان وەسفی دەکەن

(٨٢)

فَذَرۡهُمۡ جا لێیانگەڕی

يَخُوضُوا بەناو بکەون

وَيَلۡعَبُوا و یاری بکەن

حَتَّى تاکو

يُلَٰقُوا دەگەن بە

يَوۡمَهُم ڕۆژەکەیان

ٱلَّذِي ئەوەی

يُوعَدُون بەڵێنیان پێدەدرێ

(٨٣)

وَهُو و هەر ئەوخۆی

ٱلَّذِي ئەو زاتەیە کە

فِي لە

ٱلسَّمَآء ئاسمان

إِلَٰهٞ خوداو پەرستراوە

وَفِي لە ، لەناو

ٱلۡأَرۡض زەوی

إِلَٰهٞ خوداو پەرستراوە

وَهُو و ئەوە

ٱلۡحَكِيم دانا و تەواو کاربەجێی

ٱلۡعَلِيم تەواو زانا و ئاگادار (ناوێکی پیرۆزی خودایە)

(٨٤)

وَتَبَارَك و فرەپیرۆزی و بەرەکەت بۆ

ٱلَّذِي ئەوەی

لَهُۥ هی ئەوە

مُلۡك پادشایەتی وخاوەنداری

ٱلسَّمَٰوَٰت ئاسمانەکان

و ٱلۡأَرۡض و زەوی

وَمَا و ئەوەی

بَيۡنَهُمَا لەنێوانیشیانە

وَعِندَهُۥ و لای ئەو

عِلۡم زانست و زانیاری

ٱلسَّاعَة ڕۆژی قیامەت

وَإِلَيۡه و هەر بۆلای ئەو

تُرۡجَعُون دەگێردرێنەوە

(٨٥)

وَلَايَمۡلِك و خاوەندار نینە

ٱلَّذِین ئەوانەی

يَدۡعُون ئەوان دەیپەرستن و هاواری بۆ دەکەنە

مِن -

دُونِه بێجگە (-لە) و

ٱلشَّفَٰعَة تکاکاری

إِلَّا بەگەورەدەزانی

مَن کەسێک

شَهِد شایەتی دابێ

بِٱلۡحَقّ بە حەق و ڕاستی و ڕەوایی

وَهُمۡ و ئەوان

يَعۡلَمُون بزانن

(٨٦)

وَلَئِن و ئەگەر

سَأَلۡتَهُم پرسیاریان لێ بکەییت

مَّنۡ کێ

خَلَقَهُمۡ ئەوانی خولقاندوە و دروستکردوە

لَيَقُولُنّ بەدڵنیایی ئەوان دەڵێن

ٱللَّه خودا

فَأَنَّى جا چۆن و بۆکوێ

يُؤۡفَكُون لادەدرێن و وێڵدەکرێن

(٨٧)

وَقِيلِهِۦ و قسەی وی

يَٰرَبّ ئەی پەروەردیگارە

إِنّ بەدڵنیایی

هَٰٓؤُلَآء ئەوانێهێ

قَوۡمٞ قەومێک

لَّا -

يُؤۡمِنُون بڕوا (-نا) هێنن

(٨٨)

فَٱصۡفَحۡ چاوپۆشی بکە

عَنۡهُمۡ لێیان

وَقُلۡ و بڵێ

سَلَٰمٞ سەلام

فَسَوۡف جا لەداهاتوو

يَعۡلَمُون دەزانن

(٨٩)
