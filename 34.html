وَقَال و -

ٱلَّذِین ئەوانەی

كَفَرُوا کوفریانکرد و بێ بڕوابوون (-گوتیان)

لِلۡحَقّ بە حەق و ڕاستی

لَمَّا کاتێ

جَآءَهُمۡ بۆیان هات

إِنۡ هَٰذَآ ئەوە (هیچ) نیە

إِلَّا جگەلە

سِحۡرٞ سیحر و جادوێکی

مُّبِينٞ دیار و ئاشکرا

(٤٣)

وَمَآ و -

ءَاتَيۡنَٰهُم و پێمان (-نە) دان

مِّن (هیچ) لە

كُتُبٖ کیتاب و نامەی ئاسمانی

يَدۡرُسُونَهَا ئەوان بیخوێننەوە

وَمَآ و -

أَرۡسَلۡنَآ ڕەوانمان (-نە) کرد بوو

إِلَيۡهِمۡ بۆیان ، بۆلایان

قَبۡلَك پێش تۆ

مِن (هیچ)

نَّذِيرٖ ترسێنەر و بێدارکەرەوەیەک

(٤٤)

وَكَذَّب و بەدرۆیان دانا و بڕوایان نەهێنا

ٱلَّذِین ئەوانەی

مِن لە

قَبۡلِهِم پێش ئەوان

وَمَا و (قەومی تۆ) نە

بَلَغُوا دە گەیشتنە

مِعۡشَار دەیەکی

مَآ ئەوەی

ءَاتَيۡنَٰهُمۡ بەوانمان دابوو

فَكَذَّبُوا جا بەدرۆیاندانا

رُسُلِي پێغەمبەرنمان

فكَيۡف ئیتر چۆن

كَان بوو

نَكِير ئینکارکردن و سزای (من)

(٤٥)

قُلۡ بڵێ

إِنَّمَآ لەڕاستیدا تەنها

أَعِظُكُم ئامۆژگاریتان دەکەم

بِوَٰحِدَةٍ بە یەک (ئامۆژگاری)

أَن کە

تَقُومُوا هەستنەوە سەرپێیان

لِلَّه بۆ خودای

مَثۡنَى دوو دوو

وَفُرَٰدَى تاک تاک و

ثُمّ پاشان

تَتَفَكَّرُوا بیر بکەنەوە

مَا بِصَاحِبِكُم بە هاوڕێیەکەتان نیە

مِّن (هیچ)

جِنَّةٍ شێتیێکک

إِنۡ هُو ئەوشتێ نیە

إِلَّا جگەلە

نَذِيرٞ ترسێنەرێک

لَّكُم بۆ ئێوە

بَيۡن بەر

يَدَيۡ دەم و دەستی

عَذَابٖ عەزاب وئەشکەنجەیەکی

شَدِيدٖ زۆر توند

(٤٦)

قُلۡ بڵێ

مَا سَأَلۡتُكُم داوام نەکردوە لێتان

مِّنۡ (هیچ)

أَجۡرٖ پاداشتێ

فَهُو ئەوە هەر

لَكُمۡ بۆخۆتانە

إِنۡ -

أَجۡرِي پاداشتم (-لای کەس نیە)

إِلَّا جگەلە

عَلَى لای

ٱللَّه خودا

وَهُو و هەر ئەوە

عَلَى لەسەر

كُلّ هەموو

شَيۡءٖ شتێک

شَهِيدٞ شایەتە

(٤٧)

قُلۡ بڵێ

إِنّ بەدڵنیایی

رَبِّي پەروەردیگارم

يَقۡذِف توند دەهاوێت

بِٱلۡحَقّ حەق و ڕاستی و ڕەوایی

عَلَّٰم ئەوپەڕی زانا

ٱلۡغُيُوب بە پەنهان و شاراوە و نهێنیەکان

(٤٨)

قُلۡ بڵێ

جَآء هات

ٱلۡحَقّ هەق و ڕاستی

وَمَا و-

يُبۡدِئ دەست پێ (-نا) کاتەوە

ٱلۡبَٰطِل ناهەقی و پڕ وپووچی

وَمَا ونا

يُعِيد گێڕێتەوە

(٤٩)

قُلۡ بڵێ

إِن ئەگەر

ضَلَلۡت گومڕابووم و ڕێگام ونکرد

فَإِنَّمَآ ئیتر هەر

أَضِلّ گومڕادەبم

عَلَى لەسەر

نَفۡسِي خودی خۆم

وَإِن و ئەگەر

ٱهۡتَدَيۡت ڕێی هەقم دۆزیەوە و گرتمەبەر

فَبِمَا ئەوە بە (هۆی) ئەوەی

يُوحِيٓ سروش و نیگا دەکات

إِلَيّ بۆ من

رَبِّيٓ پەروەردیگارم

إِنَّهُۥ بەدڵنیایی هەر ئەوە

سَمِيعٞ زۆر بیسەری

قَرِيبٞ زۆرنزیک

(٥٠)

وَلَوۡ وئەگەر

تَرَىٰٓ دەبینی

إِذۡ کاتێ کە

فَزِعُوا ترس دایگرتن

فَلَافَوۡت ئیتر دەربازبوون نیە

وَأُخِذُوا و ڕاپێچکران

مِن لە

مَّكَانٖ شوێنێکی

قَرِيبٖ نزیک

(٥١)

وَقَالُوٓا و گوتیان

ءَامَنَّا بڕوامان هێنا

بِهِۦ پێی

وَأَنَّى و لەکوێ

لَهُم لۆیان هەیە

ٱلتَّنَاوُش دەست بۆ بردن

مِن لە

مَّكَانِۢ جێگایەکی

بَعِيدٖ زۆر دوور

(٥٢)

وَقَدۡ و بەدڵنیایی

كَفَرُوا کوفریانکرد و بێ بڕوابوون

بِهِۦ پێی

مِن لە

قَبۡل پێشا

وَيَقۡذِفُون و پەلار دەهاوێژنە

بِٱلۡغَيۡب غەیب و شاراوە و نهێنیەکان

مِن لە

مَّكَانِۢ جێگایەکی

بَعِيدٖ زۆر دوور

(٥٣)

وَحِيل و جیاکرایەوە

بَيۡنَهُمۡ نێوانیان

وَبَيۡن و نێوان

مَا يَشۡتَهُون ئەوەی حەزی لێدەکەن

كَمَا هەرچۆن

فُعِل واکرا

بِأَشۡيَاعِهِم بەهاوشێوە و هاوسۆزەکانیان

مِّن لە

قَبۡل پێشا

إِنَّهُمۡ بەدڵنیایی ئەوان

كَانُوا پێشان

فِي لە

شَكّٖ گومان و ڕاڕایی

مُّرِيبِۢ زۆر گوماناوی

(٥٤)
