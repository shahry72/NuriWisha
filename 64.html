ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبەبەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


يُسَبِّحُ تەسبیحات دەکات

لِلَّـهِ بۆ خودا

مَا هەرچی

فِي لە

السَّمَاوَاتِ ئاسمانەکانە

وَمَا و هەرچی

فِي لە

الْأَرْضِۖ زەویە

لَهُ هی ئەوە

الْمُلْكُ هەرچی مولک و خاوەنداریە

وَلَهُ و هەر بۆ ئەویشە

الْحَمْدُ هەمو سوپاس و ستایشێک

وَهُوَ وئەویش

عَلَىٰ بەسەر

كُلِّ هەموو

شَيْءٍ شتێک

قَدِيرٌ توانادارە

(١)

هُوَ  

الَّذِي خودایە ئەوەی

خَلَقَكُمْ بەدیهێناون

فَمِنكُمْ ئیتر لەنگۆ هەیە

كَافِرٌ بێ بڕوا و کافرێکە

وَمِنكُم و لەنگۆش هەیە

مُّؤْمِنٌ بڕوادارێک

وَاللَّـهُ و خوداش

بِمَا بەوەی

تَعْمَلُونَ دەیکەن

بَصِيرٌ بینەرە

(٢)

خَلَقَ بەدیهێناوە

السَّمَاوَاتِ ئاسمانەکان

وَالْأَرْضَ و زەوی

بِالْحَقِّ بە حەق

وَصَوَّرَكُمْ و وشێوەی پێداون

فَأَحْسَنَ جا چاک چاکی کردوە

صُوَرَكُمْۖ شێوەکانتان

وَإِلَيْهِ و هەر بۆلای ئەویشە

الْمَصِيرُ داهاتوو و سەرئەنجام

(٣)

يَعْلَمُ دەزانێت

مَا چی

فِي لە

السَّمَاوَاتِ ئاسمانەکان

وَالْأَرْضِ و زەوی دایە

وَيَعْلَمُ و دەشزانێت

مَا چی

تُسِرُّونَ پەنهان دەکەن

وَمَا و چی

تُعْلِنُونَ ئاشکرا دەکەن

وَاللَّـهُ و خوداش

عَلِيمٌ زانایە

بِذَاتِ بە هی ناو

الصُّدُورِ سینگەکان

(٤)

أَلَمْ ئایا نە

يَأْتِكُمْ هاتووە بۆتان

نَبَأُ هەواڵی

الَّذِينَ ئەونەی

كَفَرُوا کوفریان کرد

مِن لە

قَبْلُ پێشدا

فَذَاقُوا جا چەشتیان

وَبَالَ ئۆباڵی

أَمْرِهِمْ کارەکانیان

وَلَهُمْ و بۆیان هەیە

عَذَابٌ عەزاب و ئەشکەنجەیەکی

أَلِيمٌ زۆر بە ئازار

(٥)

ذَٰلِكَ ئەوە ش

بِأَنَّهُ بەهۆی ئەوەی

كَانَت پێشان

تَّأْتِيهِمْ ئەوان بۆیان دەهێنان

رُسُلُهُم پێغەمبەرەکانیان

بِالْبَيِّنَاتِ بەڵگە دیار و ئاشکراکان

فَقَالُوا ئەوسا ئەوان گوتیان

أَ  (ئایا)

بَشَرٌ بەشەر و مرۆڤانێک

يَهْدُونَنَا ڕێنموونیمان دەکەن ؟!

فَكَفَرُوا جا کوفریانکرد

وَتَوَلَّوا و پشتیان هەڵکرد

وَّاسْتَغْنَى -

اللَّـهُ و خوداش (-بێ نیازبوو)

وَاللَّـهُ و خودا

غَنِيٌّ دەوڵەمەندی

حَمِيدٌ زۆر ستایشکراوە

(٦)

زَعَمَ-  

الَّذِينَ ئەوانەی

كَفَرُوا کوفریان کرد و بێ باوەڕبوون (-بەقسەی خۆیان گوتیان)

أَن کەوا

لَّن هەرگیز-

يُبْعَثُوا زیندوو (-نا) کرێنەوە

قُلْ بڵێ

بَلَىٰ بەڵێ

وَرَبِّي و سوێند بە پەروەردیگارم

لَتُبْعَثُنَّ هەرزیندوو دەکرێنەوە

ثُمَّ پاشان

لَتُنَبَّؤُنَّ بەدڵنیایی هەواڵی ڕاستتان پێدەدرێ

بِمَا دەربارەی ئەوەی

عَمِلْتُمْ کردووتانە

وَذَٰلِكَ و ئەوەش

عَلَى بۆ

اللَّـهِ خودا

يَسِيرٌ ئاسانە

(٧)

فَآمِنُوا جا برڕوا بهێنن

بِاللَّـهِ بە خودا

وَرَسُولِهِ و پێغەمبەرەکەی

وَالنُّورِ  

الَّذِي لەگەڵ ئەو رووناکی و نورەی

أَنزَلْنَا دامانبەزاندوە

وَاللَّـهُ و خوداش

بِمَا بەوەی

تَعْمَلُونَ دەیکەن

خَبِيرٌ شارەزایە

(٨)

يَوْمَ ڕۆژێ

يَجْمَعُكُمْ کۆتاندەکاتەوە

لِيَوْمِ بۆ ڕۆژی

الْجَمْعِۖ کۆکردنەوە

ذَٰلِكَ ئەوە

يَوْمُ ڕۆژی

التَّغَابُنِ  

ۗ زیانبارییە

وَمَن و هەر کەسێک

يُؤْمِن بڕوابێنێ

بِاللَّـهِ بە خودا

وَيَعْمَلْ وبکات

صَالِحًا کارێکی چاک

يُكَفِّرْ  (ئەوە خوا) دادەپۆشێ و چاوپۆشی دەکات

عَنْهُ لێی

سَيِّئَاتِهِ خراپە و گوناحەکانی

وَيُدْخِلْهُ و دەیخاتە

جَنَّاتٍ باخاتێک

تَجْرِي تێدەپەڕێ

مِن لە

تَحْتِهَا ژێریدا

الْأَنْهَارُ ڕوبارکان

خَالِدِينَ هەتاهەتایێ

فِيهَا تیایدان

أَبَدًا بە هەمیشەیی

ذَٰلِكَ ئەوەیە

الْفَوْزُ بردنەوەی

الْعَظِيمُ زۆر گەورە و مەزن

(٩)

وَالَّذِينَ و ئەوانەی

كَفَرُوا کوفریانکرددوە

وَكَذَّبُوا و بە درۆیان داناوە و بڕوایان نەکردوە

بِآيَاتِنَا بە بەڵگە دیار و ئاشکراکانمان

أُولَـٰئِكَ ئەوان

أَصْحَابُ هاوەڵ و نیشتەجێی

النَّارِ ئاگرن

خَالِدِينَ هەتاهەتایێ دەمێنن

فِيهَاۖ تیایدا

وَبِئْسَ وخراپترین

الْمَصِيرُ سەرئەنجامە

(١٠)

مَا  

أَصَابَ تووش نایێت

مِن  (هیچ)

مُّصِيبَةٍ بەڵاو و موسیبەتێک

إِلَّا مەگەر

بِإِذْنِ بە ڕێپێدانی

اللَّـهِ  

ۗ خودا

وَمَن و هەرکەسێ

يُؤْمِن بڕوا بێنێ

بِاللَّـهِ بە خودا

يَهْدِ  (ئەوە) ڕاستەڕێی و هیدایەت دەبەخشێ بە -

قَلْبَهُ دڵەکەی

وَاللَّـهُ و خوداش

بِكُلِّ بە هەموو

شَيْءٍ شتێک

عَلِيمٌ زۆر زانا و ئاگادارە

(١١)

وَأَطِيعُوا ملکەچ و گوێڕایەڵ بن بۆ

اللَّـهَ خودا

وَأَطِيعُوا و ملکەچ و گوێڕایه ڵیش بن بۆ

الرَّسُولَ پێغەمبەر

فَإِن جا ئەگەر

تَوَلَّيْتُمْ پشت هەڵکەن

فَإِنَّمَا ئەوە هەر ئەوەندە

عَلَىٰ لەسەر

الْبَلَاغُ ڕاگەیاندنی

الْمُبِينُ دیار و ئاشکرا

(١٢)

اللَّـهُ الله

لَا -

إِلَـٰهَ نیە (هیچ) خوداو پەرستراوێک

إِلَّا تەنها

هُوَ ئەو (نەبێت)

وَعَلَى و هەر بە

اللَّـهِ خودا

فَلْيَتَوَكَّلِ با پشت ببەستن

الْمُؤْمِنُونَ بڕوادارەکان

(١٣)

يَا  

أَيُّهَا هۆ ئەی

الَّذِينَ ئەوانەی

آمَنُوا بڕواتان هێناوە ،

إِنَّ بەدڵنیایی

مِنْ لە

أَزْوَاجِكُمْ خێزانەکانتان

وَأَوْلَادِكُمْ ومنداڵەکانتان

عَدُوًّا  (هەیە) دوژمنێک

وَإِن و ئەگەر

تَعْفُوا ببورن

وَتَصْفَحُوا و چاوپۆشی بکەن

فَإِنَّ ئەوە بەدڵنیایی

اللَّـهَ خودا

غَفُورٌ لێخۆشبوی

رَّحِيمٌ میهرەبابە .

(١٤)

أَمْوَالُكُمْ ماڵ وسامانتان

فِتْنَةٌ فیتنەو تاقیکردنەوەیە

أَجْرٌ پاداشتێکی

عَظِيمٌ زۆر گەورە و مەزن.

مَا ئەوەندەی

اسْتَطَعْتُمْ توانیتان

وَأَطِيعُوا و گوێڕایەڵ بن

وَأَنفِقُوا و ببەخشنەوە

خَيْرًا چاکە

لِّأَنفُسِكُمْ

وَمَن و هەرکەسێ

فَأُولَـٰئِكَ  

هُمُ ئەوە هەر ئەوانن

الْمُفْلِحُونَ سەرکەوتوو و سەرفراز.
