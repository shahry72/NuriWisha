بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ النَّبَأ 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبە بەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


عَمَّ دەربارەی چی

يَتَسَاءَلُونَ پرسیار لە یەکتر دەکەن

(١)

عَنِ دەربارەی

النَّبَإِ ئەو هەواڵەی

الْعَظِيمِ زۆرگەورەیە

(٢)

الَّذِي ئەوەی

هُمْ ئەوان

فِيهِ تیایدا

مُخْتَلِفُونَ جیاوازو ناتەبان و مشتو مڕیانە !

(٣)

كَلَّا نەخێر

سَيَعْلَمُونَ لەداهاتوێکی نزیک دەزانن

(٤)

ثُمَّ دیسان

كَلَّا نەخێر

سَيَعْلَمُونَ لەداهاتوێکی نزیک دەزانن

(٥)

أَلَمْ نَجْعَلِ ئایا -نەمانکردۆتە

الْأَرْضَ -زەوی

مِهَادًا ڕایەخێکی ئامادەکراو

(٦)

وَالْجِبَالَ و شاخەکان (یش مان نەکردوە بە)

أَوْتَادًا چەندین مێخەکێک (داكوتراو)

(٧)

وَخَلَقْنَاكُمْ و ئیوەمان خلق کردوە

أَزْوَاجًا جۆت جۆتێ

(٨)

وَجَعَلْنَا و -مان کردۆتە

نَوْمَكُمْ -نوستنەکەتانمان

سُبَاتًا سڕبوون و سرەوتن و حەسانەوەێ

(٩)

وَجَعَلْنَا و -مان کردۆتە

اللَّيْلَ -شەو

لِبَاسًا پۆشتاکێک

(١٠)

وَجَعَلْنَا و -مان کردۆتە

النَّهَارَ -ڕۆژ مان

مَعَاشًا ژیێنەرێک

(١١)

وَبَنَيْنَا و بونیادمان ناوە

فَوْقَكُمْ لە ژوور سەرتان

سَبْعًا حەوتی

شِدَادًا زۆر توند و قایم

(١٢)

وَجَعَلْنَا و دروستمان کردوە

سِرَاجًا چرایەکی

وَهَّاجًا زۆر ڕۆشەن (وگەرمی بەخش)

(١٣)

وَأَنزَلْنَا و دامانگرتوە

مِنَ لە

الْمُعْصِرَاتِ (هەورەکان)

مَاءً ئاوێکی

ثَجَّاجا بە لێزمە

(١٤)

لِّنُخْرِجَ تا دەربکەین

بِهِ پێی

حَبًّا دانەوێڵە

وَنَبَاتًا و ڕوەک و گژ و گیایێک

(١٥)

وَجَنَّاتٍ و باخاتێکی

أَلْفَافًا چڕوپڕی تێک ئاڵاو

(١٦)

إِنَّ بەراستی

يَوْمَ ڕۆژی

الْفَصْلِ جیاکردەنەوە

كَانَ -

مِيقَاتًا کاتی بۆدیارکراوە(-بوو)

(١٧)

يَوْمَ ئەو ڕۆژەی

يُنفَخُ فوو دەکرێ

فِي لە

الصُّورِ شەیپوور -کەڕەنا

فَتَأْتُونَ ئەوساکە دێن

أَفْوَاجا کۆمەڵ کۆمەڵیک

(١٨)

وَفُتِحَتِ و

السَّمَاءُ ئاسمان

فَكَانَتْ ئەوسا بوو بە

أَبْوَابًا چەندین دەروازە و دەرگایێک

(١٩)

وَسُيِّرَتِ و

الْجِبَالُ شاخەکان

فَكَانَتْ جا بوونە

سَرَابًا سەڕابێک

(٢٠)

إِنَّ جَهَنَّمَ بەڕاستی دۆزەخ

كَانَتْ -

مِرْصَادًا بۆسە(-بوو)

(٢١)

لِّلطَّاغِينَ بۆ سنوور بەزێنان وطاغیەکان

مَآبًا جێگای گەڕانەوەیێک (بوو)

(٢٢)

لَّابِثِينَ ماوەن

فِيهَا تیاید

أَحْقَابًا چەندەها سەدەیێک

(٢٣)

لَّا يَذُوقُونَ نا چێژن

فِيهَا تیایدا

بَرْدًا ساردی و فێنکیێک

وَلَا و نە

شَرَابا خواردنەوەیێک

(٢٤)

إِلَّا تەنها

حَمِيمًا كوڵاوە ئاوێک

وَغَسَّاقًا و کێم و زوخاوێک

(٢٥)

جَزَاءً سزایەکی

وِفَاقًا بەرابەر

(٢٦)

إِنَّهُمْ بەڕاستی ئەوان

كَانُوا لَايَرْجُونَ نەیاندەخواست و بەتەمای -نەبوون

حِسَابًا (-هیچ) حسابێ ک

(٢٧)

وَكَذَّبُوا و باوەڕیان نەهێناو درۆیانکرد

بِآيَاتِنَا بە ئایەتەکانمان

كِذَّابًا درۆکردنێک

(٢٨)

وَكُلَّ و هەموو

شَيْءٍ شتێ

أَحْصَيْنَاهُ سەرژمێرمان کردوە

كِتَابًا بە نوسراوەیێک

(٢٩)

فَذُوقُوا جا بچێژن

فَلَن نَّزِيدَكُمْ ئێمەهەرگیز (چیترتان) بۆ زیادناکەین

إِلَّا جگە لە

عَذَابًا ئەشکەنجە و عەزابێک

(٣٠)

إِنَّ بێ گومان

لِلْمُتَّقِينَ بۆ لەخوا ترسان و پارێزکاران هەیە

مَفَازًا بردنەویێک

(٣١)

حَدَائِقَ باخاتێک

وَأَعْنَابًا و جۆرەها ترێیێک

(٣٢)

وَكَوَاعِبَ و كچانی تازه مه مك کردووی ی

أَتْرَابًا هاوتەمەن وهاوتا لە جوانی وئاکار شیرینیک

(٣٣)

وَكَأْسًا و پەرداخی شەڕابی

دِهَاقًا پڕی لە سەر رژاو

(٣٤)

لَّايَسْمَعُونَ و نابیسن

فِيهَا تیاید

لَغْوًا هیچ قسەیەکی بێهوودە وغەلبەغەلبێک

وَلَا كِذَّابًا و نە هیچ درۆیێک

(٣٥)

جَزَاءً پاداشتێک

مِّن لە

رَّبِّكَ پەروەردیگارتەوە

عَطَاءً بەخشینێک

حِسَابًا (بۆیان) حساب کراوە وبەسیانەک

(٣٦)

رَّبِّ پەروەردگاری

السَّمَاوَاتِ ئاسمانەکان و

وَالْأَرْضِ زەوی

وَمَا و ئەوەی

بَيْنَهُمَا ۖ الرَّحْمَـٰنِ لە نێوانیاندایە

لَا يَمْلِكُونَ نیەتیان

مِنْهُ لێیەوە

خِطَابًا (مۆڵەتی) هیچ وتارێک

(٣٧)

يَوْمَ رۆژێ

يَقُومُ ڕادەوەستێت

الرُّوحُ رۆح (جبریل)

وَالْمَلَائِكَةُ و فریشتەکان

صَفًّا ۖ بەرێز

لَّايَتَكَلَّمُونَ قسە ناکەن

إِلَّا تەنها

مَنْ ئەوکەسەی

أَذِنَ لَهُ رێگای پێدا

الرَّحْمَـٰنُ خودای ئەوپەڕ میهرەبان

وَقَالَ صَوَابًا و ڕاستی (بڵێ)

(٣٨)

ذَٰلِكَ الْيَوْمُ ئا ئەو رۆژە

الْحَقُّ ۖ هەقی تەواوە

فَمَن جا هەرکەسێ

شَاءَ ویستی

اتَّخَذَ دەگرێتەبەر

إِلَىٰ بەرەو

رَبِّهِ پەروەردگاری

مَآبًا ڕێی گەرانەوەیێ ک

(٣٩)

إِنَّا بە ڕاستی ئێمە

أَنذَرْنَاكُمْ هۆشداریمان دانێ لە مەترسی

عَذَابًا عەزاب و جەزرەبەیێک

قَرِيبًا نزیک ،

يَوْمَ رۆژێ

يَنظُرُ 

الْمَرْءُ هەمووکەسێک

مَا چی

قَدَّمَتْ پێشخۆی خستوە (لە کردەوە)

يَدَاهُ هەردوو دەستەکانی

وَيَقُولُ الْكَافِرُ و ئەوڕۆژە کافر دەڵێ

يَالَيْتَنِي ئای خۆزگەم

كُنتُ تُرَابًا خۆڵێ بوام

(٤٠)
