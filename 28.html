مَنۡ کێیە

غَيۡر جگەلە

ٱللَّه الله

يَأۡتِيكُم بهێنێ

بِلَيۡلٖ شەوێک

تَسۡكُنُون ئارام و خامۆش بن

فِيه تیایدا

أَفَلَاتُبۡصِرُون ئایا نابینن و تێڕا نامێنن؟

(٧٢)

وَمِن و لە

رَّحۡمَتِهِۦ رەحمەت و میهرەبانی خۆیەتی

جَعَل کە دایناوە

لَكُم بۆتان

ٱلَّيۡل شەو

وَٱلنَّهَار و ڕۆژگار

لِتَسۡكُنُوا تاکوو خامۆش و کڕ بن

فِيه تیایدا

وَلِتَبۡتَغُوا و بخوازن

مِن لە

فَضۡلِهِۦ زێدە بەخششی ئەو

وَلَعَلَّكُمۡ و بەڵکو ئێوە

تَشۡكُرُون سوپاسگوزاری بکەن

(٧٣)

وَيَوۡم و ئەو ڕۆژەی

يُنَادِيهِمۡ بانگیان دەکات

فَيَقُول جا دەڵێت:

أَيۡن لەکوێیە

شُرَكَآءِي ھاوبەش شەریکەکانم؟!

ٱلَّذِین ئەوانەی

كُنتُمۡ ئێوە پێشان

تَزۡعُمُون (بەدرۆ) دەتانگووت

(٧٤)

وَنَزَعۡنَا و دەرمانهێنا

مِن لە

كُلّ هەموو

أُمَّةٖ ئوممەت و گەلێک

شَهِيدٗا شایەتێک

فَقُلۡنا ئیتر گوتمان

هَاتُوا دەی بهێنن

بُرۡهَٰنَكُمۡ بەڵگە بەهێزەکەتان

فَعَلِمُوٓا ئەوساکە زانیان

أَنّ کە بەڕاستی

ٱلۡحَقّ هەق و ڕاستی

لِلَّه هەر هی خودایە

وَضَلّ و بزر بوو و ون بوو

عَنۡهُم لێیان

مَّا ئەوەی

كَانُوا پێشان

يَفۡتَرُون بە درۆ ئەوان هەڵیان دەبەست

(٧٥)

إِنّ بەدڵنیایی

قَٰرُون قاروون

كَان -

مِن لە

قَوۡم قەومی

مُوسَى مووسا (-بوو)

فَبَغَى جا دەستدرێژی کرد

عَلَيۡهِمۡ لەسەریان

وَءَاتَيۡنَٰه و ئێمە پێمان دا

مِن لە

ٱلۡكُنُوز گەنجینەکان

مَآ ئەوەی

إِنّ بەڕاستی

مَفَاتِحَهُۥ کلیلەکانی

لَتَنُوٓأ بەزەحمەت هەڵدەگیرێ

بِٱلۡعُصۡبَة بە کۆمەڵێکی

أُوْلِي خاوەن

ٱلۡقُوَّة هێزێکی زۆر

إِذۡ ئەوەبوو کە

قَال گوتیان

لَهُۥ پێی

قَوۡمُهُۥ قەومەکەی:

لَاتَفۡرَحۡ دڵخۆش و سەرمەست مەبە

إِنّ بەڕاستی

ٱللَّه خودا

لَايُحِبّ خۆشی ناوێت

ٱلۡفَرِحِين ئەوانەی دڵخۆشی سەرمەستن (بە دونیا)

(٧٦)

وَٱبۡتَغ و بخوازە

فِيمَآ لەوەی

ءَاتَىٰك پێیداوی

ٱللَّه خودا

ٱلدَّار ماڵ و مەنزڵگای

ٱلۡأٓخِرَة (ژیانی) دووایی

وَلَاتَنس و فەرامۆش مەکە

نَصِيبَك بەشی خۆت

مِن لە

ٱلدُّنۡيَا دونیا

وَأَحۡسِن و چاکە بکە

كَمَآ هەروەک

أَحۡسَن -

ٱللَّه خودا (-چاکەی کردوە)

إِلَيۡك بۆتۆ

وَلَاتَبۡغ و مەخوازە

ٱلۡفَسَاد فەساد و خراپەکاری

فِي لە

ٱلۡأَرۡض زەوی دا

إِنّ بەدڵنیایی

ٱللَّه خودا

لَايُحِبّ خۆشی ناوێت

ٱلۡمُفۡسِدِين خراپکار و خراپخوازان

(٧٧)

قَال ئەویش گوتی

إِنَّمَآ لەڕاستیدا

أُوتِيتُهُۥ (ئەوەی) پێم دراوە

عَلَى هەر لەسەر

عِلۡمٍ زانست و زانیارییەکە

عِندِيٓ لای خۆم هەیە

أَو ئایا

لَمۡ يَعۡلَمۡ أَنّ نەیزانیوە کە

ٱللَّه خودا

قَدۡ بەدڵنیایی

أَهۡلَك لەناویبرد و فەوتاندی

مِن لە

قَبۡلِهِۦ پێش ئەو

مِن لە

ٱلۡقُرُون چەندەها خەڵکی سەردەمانێک

مَنۡ ئەوان

هُو بەخۆیان

أَشَدّ توندتر و بەهێزتر (بوون)

مِنۡه لە

قُوَّة هێز و توانادا

وَأَكۡثَر و زۆرتر (-بوو)

جَمۡعٗا کۆکردنەوە و کۆمەڵيان

وَلَايُسۡ‍َٔل و پرسیار ناکرێ

عَن لە

ذُنُوبِهِم گوناهەکانیان

ٱلۡمُجۡرِمُون تاوانبارەکان

(٧٨)

عَلَى بۆ سەر

فِي بە

يُرِيدُون دەیانەویست

ٱلدُّنۡيَا دونیا (-گوتیان)

يَٰلَيۡت ئای خۆزگە

لَنَا بۆ ئێمە (بوایە)

مِثۡل (شتێ) وەک

مَآ ئەوەی

أُوتِي دراوە بە

قَٰرُون قاروون

إِنَّهُۥ بەدڵنیایی ئەو

لَذُو هەرخاوەن

حَظٍّ شانس و بەشێکی

عَظِيمٖ زۆر گەورە و مەزن یە

(٧٩)

وَقَال و -

ٱلَّذِین ئەوانەی

أُوتُوا پێیاندرابوو

ٱلۡعِلۡم زانین و زانست (-گوتیان)

وَيۡلَكُمۡ هەی قوڕتان بەسەر

ثَوَاب پاداشتی

ٱللَّه خودا

خَيۡرٞ چاکتر ە

لِّمَنۡ بۆ ئەوەی

ءَامَن بڕوای هێناوە

و عَمِل و کردویەتی

وَلَايُلَقَّىٰهَآ و ئەوەیان پێ نادرێ

ٱلصَّٰبِرُون ئارامگرەکان

ٱللَّه خودا
