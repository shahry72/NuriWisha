سورة یوسف 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبەبەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)

 


الر ۚ ئەلف لام ڕا

تِلْكَ ئا ئەوەیە

آيَاتُ ئایەت و نیشانە گەورەکانی

الْكِتَابِ ئەو کیتابەی

الْمُبِينِ کە ڕۆن و ڕۆنکەرەوەیە

(١)

إِنَّا بەدڵنیایی ئێمە خۆمان

أَنزَلْنَاهُ دامانبەزاند

قُرْآنًا وەك قورئانێکی

عَرَبِيًّا بەزمانی عەرەبی

لَّعَلَّكُمْ بەلكو

تَعْقِلُونَ تێی بگەن و پێی بگەن

(٢)

نَحْنُ ئێمە

نَقُصُّ چیرۆك دەگێڕینەوە

عَلَيْكَ بەسەرتا

أَحْسَنَ چاكترین

الْقَصَصِ چیڕۆك

بِمَا بەوەی

أَوْحَيْنَا سروشمانکردوە

إِلَيْكَ بۆتۆ

هَـٰذَا ئەو

الْقُرْآنَ قورئانە

وَإِن و ئەگەریش

كُنتَ تۆ-

مِن لە

قَبْلِهِ پێش ئەوەدا

لَمِنَ بەڕاستی لە

الْغَافِلِينَ بێئاگایان -بووی

(٣)

إِذْ کاتێ کە

قَالَ -

يُوسُفُ یوسف -گوتی

لِأَبِيهِ بە باوکی

يَا أَبَتِ بابە گیان

إِنِّي بێگومان من

رَأَيْتُ بینیم (لەخەومدا)

أَحَدَ عَشَرَ یازدە

كَوْكَبًا ئەستێرە

وَالشَّمْسَ و ڕۆژ

وَالْقَمَرَ و مانگ

رَأَيْتُهُمْ ئەوانم بینی

لِي بۆ من!

سَاجِدِينَ لە سوژدەدابوون

(٤)

قَالَ (باوکی) پێی گوت

يَا بُنَيَّ ئەی كوڕی خۆم

لَا تَقْصُصْ مەگێڕەوە

رُؤْيَاكَ چیڕۆکەکەت

عَلَىٰ بەسەر

إِخْوَتِكَ براکانت

فَيَكِيدُوا ئیتر بەفێڵ داڕێژن

لَكَ بۆتۆ

كَيْدًا ۖ پلانیك (بە مەبەستی زەرەربەخش)

إِنَّ بەراستی

الشَّيْطَانَ شەیتان

لِلْإِنسَانِ بۆ مرۆڤ

عَدُوٌّ دوژمنێکی

مُّبِينٌ دیاروڕۆنە

(٥)

وَكَذَٰلِكَ ئا ئەوها

يَجْتَبِيكَ تۆ ‌هەڵدەبژێرێ

رَبُّكَ پەروەردیگار

وَيُعَلِّمُكَ وفێرت دەکات

مِن لە

تَأْوِيلِ لێكدانەوەی

الْأَحَادِيثِ قسەکان (وخەونەکان)

وَيُتِمُّ و کامڵ دەکات

نِعْمَتَهُ ناز و نیعمەتی خۆی

عَلَيْكَ بەسەرت

وَعَلَىٰ و بەسەر

آلِ بنەماڵەی

يَعْقُوبَ یەعقوب

كَمَا هەرچۆن

أَتَمَّهَا (نێعمەتەکانی) کامڵ کردوە

عَلَىٰ بەسەر

أَبَوَيْكَ باوانت

مِن لە

قَبْلُ پێشا

إِبْرَاهِيمَ ئیبراهیم

وَإِسْحَاقَ ۚ و ئیسحاق

إِنَّ بەڕاستی

رَبَّكَ پەروەردیگارت

عَلِيمٌ زۆرزانای

حَكِيمٌ زۆرکاربەجێییە

(٦)

لَّقَدْ بەڕاستی

كَانَ -

فِي هەبوو لە

يُوسُفَ یوسف

وَإِخْوَتِهِ و براکانی

آيَاتٌ نیشانە و بەلگەی زۆر گەورە

لِّلسَّائِلِينَ بۆ داواکار و پرسیارکاران

(٧)

إِذْ کاتێ

قَالُوا گوتیان

لَيُوسُفُ بەڕاستی یوسف

وَأَخُوهُ و براکەی

أَحَبُّ خۆشەویستترن

إِلَىٰ بۆ

أَبِينَا باوكمان

مِنَّا لە ئێمە

وَنَحْنُ لەکاتێ ئێمە

عُصْبَةٌ دەستە و کۆمەڵێکین

إِنَّ بێگومان

أَبَانَا باوكمان

لَفِي هەر لەناو

ضَلَالٍ گومڕاییەکی

مُّبِينٍ زۆر دیار و ئاشكرایە

(٨)

اقْتُلُوا بکوژن

يُوسُفَ یوسفی

أَوِ یان

اطْرَحُوهُ فڕێی بدەنە

أَرْضًا زەوییەك

يَخْلُ چۆڵ دەبێ

لَكُمْ بۆ ئێوە

وَجْهُ ڕووی

أَبِيكُمْ باوكتان

وَتَكُونُوا ئیتر -

مِن لە

بَعْدِهِ دوای ئەو -دەبنە

قَوْمًا کۆمەڵێکی

صَالِحِينَ چاك

(٩)

قَالَ -

قَائِلٌ قسەکەرێك

مِّنْهُمْ لەوان -گوتی

لَا تَقْتُلُوا مەكوژن

يُوسُفَ یوسفی

وَأَلْقُوهُ و فڕێیبدەنە

فِي ناو

غَيَابَتِ الْجُبِّ بیرێکی لێوهەڵچنراو

يَلْتَقِطْهُ هەڵیدەگرنەوە

بَعْضُ هەندێ

السَّيَّارَةِ کاروانچیان

إِن ئەوە ئەگەر

كُنتُمْ ئێوە -

فَاعِلِينَ بکەری ئەوکارە -بووبن

(١٠)

قَالُوا گوتیان

يَا أَبَانَا ئەی باوکی خۆمان

مَا لَكَ ئەوە چییتە

لَا تَأْمَنَّا دڵنیا نابیت لێمان

عَلَىٰ لەسەر

يُوسُفَ یوسف

وَإِنَّا و بەڕاستی ئێمە

لَهُ بۆ وی

لَنَاصِحُونَ تەواو ئامۆژگاریکەرین

(١١)

أَرْسِلْهُ بینێرە

مَعَنَا لەگەڵمان

غَدًا سبەی

يَرْتَعْ هەڵئەپەڕێ

وَيَلْعَبْ و یاریدەکات

وَإِنَّا و بەدڵنیاییش ئێمە

لَهُ بۆ ئەو

لَحَافِظُونَ تەواو پارێزکاریین

(١٢)

قَالَ (باوکیان) گوتی

إِنِّي بەڕاسی من

لَيَحْزُنُنِي زۆر دلگرانم دەکات

أَن کە

تَذْهَبُوا بِهِ ئەوی ببەن

وَأَخَافُ و دەترسم

أَن کە

يَأْكُلَهُ -

الذِّئْبُ گورگ -ئەوی بخوات

وَأَنتُمْ و ئێوەش

عَنْهُ لێی

غَافِلُونَ غافل و بێئاگا بووبن

(١٣)

قَالُوا گوتیان

لَئِنْ (وەللا) ئەگەر

أَكَلَهُ -

الذِّئْبُ گورگ -بیخوات

وَنَحْنُ و ئێمەش

عُصْبَةٌ کۆمەڵ و دەستەیەکی (بەهێزی) ن

إِنَّا (ئەوە) ئێمە

إِذًا کەوابێ

لَّخَاسِرُونَ بەڕاستی خەسارەتمەندیین

(١٤)

فَلَمَّا جا کاتێ

ذَهَبُوا بِهِ ئەویان برد

وَأَجْمَعُوا و کۆدەنگیانکرد

أَن کە

يَجْعَلُوهُ ئەو بخەنە

فِي ناو

غَيَابَتِ الْجُبِّ ۚ تاریکایی بیر

وَأَوْحَيْنَا و سروشمانکرد

إِلَيْهِ بۆی

لَتُنَبِّئَنَّهُم بێگومان تۆ هەواڵیان پێدەدەی

بِأَمْرِهِمْ هَـٰذَا بەو کارەی ئێستایان

وَهُمْ لەکاتێ ئەوان

لَا يَشْعُرُونَ هەست پێناکە و نازانن

(١٥)

وَجَاءُوا ئیتر هاتن بۆ لای

أَبَاهُمْ باوکیان

عِشَاءً عیشایێ

يَبْكُونَ (وا) دەگریێن

(١٦)

قَالُوا گوتیان :

يَا ئەی

أَبَانَا باوکی خۆمان

إِنَّا ئێمە

ذَهَبْنَا ڕۆشتین

نَسْتَبِقُ مسابقە بکەین

وَتَرَكْنَا و ئێمە-

يُوسُفَ یوسف مان

عِندَ لای

مَتَاعِنَا کەلوپەلەکانمان -بەجێهێشت

فَأَكَلَهُ ئەوە بوو-

الذِّئْبُ ۖ گورگ -ئەوی خوارد

وَمَا و-

أَنتَ تۆ

بِمُؤْمِنٍ (هەرگیز) بڕواکەر -نیت

لَّنَا بۆمان

وَلَوْ خۆ ئەگەر

كُنَّا ئێمە-

صَادِقِينَ ڕاستگۆش -بووبین

(١٧)

وَجَاءُوا و هێنایانە

عَلَىٰ سەر

قَمِيصِهِ کراسەکەی

بِدَمٍ خوێنێکی

كَذِبٍ ۚ درۆیین

قَالَ (باوکیان) گوتی :

بَلْ بەڵكوو

سَوَّلَتْ ڵواندووی و ڕازاندوویەتیەوە

لَكُمْ بۆتان

أَنفُسُكُمْ نەفسی خۆتان بۆخۆتان

أَمْرًا ۖ کارێك

فَصَبْرٌ جا خۆڕاگریێکی

جَمِيلٌ ۖ جوان

وَاللَّـهُ و هەرخودایە

الْمُسْتَعَانُ یارمەتی دەر و پشت وپەنابێ

عَلَىٰ لەسەر

مَا ئەوەی

تَصِفُونَ ئێوە وەصفی دەکەن و باسی دەکەن

(١٨)

وَجَاءَتْ و-

سَيَّارَةٌ زۆرڕۆیێك (کاروانێ) -هات

فَأَرْسَلُوا و-

وَارِدَهُمْ ئاوهێنی خۆیان -نارد

فَأَدْلَىٰ جا شۆڕی کردەوە

دَلْوَهُ ۖ دەولکەکەی

قَالَ گوتی:

يَا ئەی

بُشْرَىٰ مزگێنی

هَـٰذَا ئەوە

غُلَامٌ ۚ كوڕێکە

وَأَسَرُّوهُ و شاردیانەوە

بِضَاعَةً ۚ وەك کەل و پەلێ

وَاللَّـهُ و خوداش

عَلِيمٌ زۆر زانایە

بِمَا بەوەی

يَعْمَلُونَ ئەوان دەیکەن

(١٩)

وَشَرَوْهُ و وایانکرد بکڕدرێ (فرۆشتیان)

بِثَمَنٍ بە نرخێکی

بَخْسٍ کەمکراو

دَرَاهِمَ چەند درهەمێکی

مَعْدُودَةٍ ژمێردراو (کەم)

وَكَانُوا و ئەوان-

فِيهِ لەو (لەبۆ یوسف)

مِنَ لە (کەسانێ) -بوون کە

الزَّاهِدِينَ بێ نیاز و زاهید بوون

(٢٠)

وَقَالَ وە-

الَّذِي ئەو کەسەی

اشْتَرَاهُ ئەوی كڕی

مِن لە

مِّصْرَ میصر

لِامْرَأَتِهِ بۆ خێزانەکەی

أَكْرِمِي ڕێز بنێ لە

مَثْوَاهُ جێ و ڕێگای

عَسَىٰ بەڵكوو

أَن ئەوەی کە

يَنفَعَنَا سودمان پێ بگەیەنێ

أَوْ یان

نَتَّخِذَهُ بیکەینە-

وَلَدًا ۚ كوڕێ (-بۆخۆمان)

وَكَذَٰلِكَ و ئاوا

مَكَّنَّا پایەدار و جێگیرمانکرد-

لِيُوسُفَ یوسف

فِي لە

الْأَرْضِ زەوی

وَلِنُعَلِّمَهُ و هەروەها یا فێری بکەین

مِن لە

تَأْوِيلِ لێكدانەوە ویەكلاکردنەوەی

الْأَحَادِيثِ ۚ قسە و باس (و خەون) ەکان

وَاللَّـهُ و خودا

غَالِبٌ زاڵە

عَلَىٰ بەسەر

أَمْرِهِ کار و فەرماانی

وَلَـٰكِنَّ بەڵام

أَكْثَرَ زۆربەی

النَّاسِ خەڵک

لَا -

يَعْلَمُونَ نازانن

(٢١)

وَلَمَّا و کاتێ

بَلَغَ گەیشتە

أَشُدَّهُ ئەوپەڕی هێز و توانایی

آتَيْنَاهُ پێمانبەخشی

حُكْمًا حوكم

وَعِلْمًا ۚ و زانستێک

وَكَذَٰلِكَ و ئاواش

نَجْزِي پاداش دەدەینەوە

الْمُحْسِنِينَ چاکەکاران

(٢٢)

وَلَمَّا کاتێ

بَلَغَ گەیشتە

أَشُدَّهُ ئەوپەڕی هێز وتوانای

آتَيْنَاهُ پێبەخشی

حُكْمًا حوكم

وَعِلْمًا ۚ علمێكمان

وَكَذَٰلِكَ و ئاواش

نَجْزِي -

الْمُحْسِنِينَ چاکەکاران -پاداش دەدەینەوە

(٢٢)

وَرَاوَدَتْهُ -

الَّتِي ئەو (ئافرەتەی)

هُوَ ئەو (یوسف)

فِي لە

بَيْتِهَا ماڵیان (بوو) / (خۆی هێناو برد) ،

عَن (بە ناحەزی)

نَّفْسِهِ نەفسی ئەو (یوسف)

وَغَلَّقَتِ و زۆر مەحکەم-

الْأَبْوَابَ دەرگاکانی -داخست

وَقَالَتْ وگوتی

هَيْتَ -

لَكَ ۚ بۆتۆ -خۆم ئامادەکردوە دە وەرە

قَالَ (یوسف) گوتی

مَعَاذَ پەنا بە

اللَّـهِ ۖ خودا

إِنَّهُ هەر ئەوە

رَبِّي خاوەنم

أَحْسَنَ -

مَثْوَايَ ۖ جێ و ڕێی /چاكکردم ،

إِنَّهُ بەدڵنیایی

لَا يُفْلِحُ -

الظَّالِمُونَ ستەمکاران -سەرناکەون و سەرفراز نابن

(٢٣)

وَلَقَدْ ئەوەبوو

هَمَّتْ (ئافرەتەکە) چوو

بِهِ ۖ لۆی

وَهَمَّ و (یوسف) -

بِهَا لۆی چوو (بوو)

لَوْلَا ئەگەر ئەوەنەبوا

أَن رَّأَىٰ کە بینی

بُرْهَانَ بەڵگەی یەكلاکەری

رَبِّهِ ۚ پەروەردگاری

كَذَٰلِكَ ئاوا

لِنَصْرِفَ تا لادەین

عَنْهُ لێی

السُّوءَ خراپە

وَالْفَحْشَاءَ ۚ و گوناحی گەورەو دزێو و قێزەوەن (زینا)

إِنَّهُ بەدڵنیایی ئەو

مِنْ لە عِبَادِنَا بەندەکانی ئێمەیە

الْمُخْلَصِينَ ئەوانەی دەربازکراو و پاكکراوەن

(٢٤)

وَاسْتَبَقَا و هەردوکیان ڕایانکردە (کێ زووتر بگاتە)

الْبَابَ دەرگا

وَقَدَّتْ و (ئافرەتەکە) -
