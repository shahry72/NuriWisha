(٢٦)

فِي لە

الْأَرْضِ زەوی

مِن لە

شَجَرَةٍ درەخت

أَقْلَامٌ چەندەها قەڵەم بوا

وَالْبَحْرُ وه ئەوەی دەریایە

يَمُدُّهُ (هەر) پێی بدات

مِن لە

بَعْدِهِ دوای ئەویشا

سَبْعَةُ حەوت

أَبْحُرٍ (دەریا)

مَّا نَفِدَتْ دوای نایێت و نابڕێتەوە

كَلِمَاتُ وتەکانی

اللَّـهِ ۗ خودا

إِنَّ بەڕاستی

اللَّـهَ خودا

عَزِيزٌ دەسەڵاتداری باڵادەستی

حَكِيمٌ دانایە

(٢٧)

مَّا خَلْقُكُمْ خولقاندنتان (-شتێک) نیە

وَلَا بَعْثُكُمْ و زیندوکردنەوەشتان (-شتێک) نیە

إِلَّا تەنها

كَنَفْسٍ وەک (خولقاندن و زیندوکردنەوەی) نەفسێک (نەبێت)

وَاحِدَةٍ ۗ یەک نەفس

إِنَّ بەڕاستی

اللَّـهَ خودا

سَمِيعٌ تەواو بیسەری

بَصِيرٌ تەواو بینەرە

أَلَمْ تَرَ ئەدی نەتدیووە

أَنَّ کەوا

اللَّـهَ خودا

يُولِجُ -

اللَّيْلَ شەو -دەخاتە

فِي ناو

النَّهَارِ ڕۆژ

وَيُولِجُ -

النَّهَارَ وڕۆژیش-دەخاتە

فِي ناو

اللَّيْلِ شەو

وَسَخَّرَ وڕامی هێناوە و ڕوخساندوویەتی

الشَّمْسَ خۆر

وَالْقَمَرَ ومانگ

كُلٌّ گشتی

يَجْرِي ڕێ دەکات

إِلَىٰ تاكو

أَجَلٍ ماوە و کاتێک ی

مُّسَمًّى پێشتر دیاریکراو

وَأَنَّ وبەڕاستی

اللَّـهَ خودا

بِمَا بەوەی

تَعْمَلُونَ دەیکەن

خَبِيرٌ تەواوشارەزایە

(٢٩)

ذَٰلِكَ ئەوە

بِأَنَّ بەوەی کە بەڕاستی

اللَّـهَ خودا

هُوَ هەرخۆی

الْحَقُّ حەققە

وَأَنَّ وەبێگومان

مَا هەرچی

يَدْعُونَ ئەوان هاواری لێدەکەن وبانگی دەکەن (دەیپەرستن)

مِن لە

دُونِهِ غەیری ئەو

الْبَاطِلُ بەتاڵ و پوچە

وَأَنَّ و بەڕاستی

اللَّـهَ خودا

هُوَ هەرخۆی

الْعَلِيُّ بەرزی

الْكَبِيرُ گەورەیە

(٣٠)

أَلَمْ تَرَ سەرنجت نەداوە

أَنَّ کەوا

الْفُلْكَ کەشتی

تَجْرِي ڕێ دەکات و دەڕوات

فِي لە ناو

الْبَحْرِ دەریا

بِنِعْمَتِ بەناز و نیعمەتی

اللَّـهِ خودا

لِيُرِيَكُم تاكونیشانتان بدات

مِّنْ هەندێک لە

آيَاتِهِ بەڵگەدیاروگەورەکانی خۆی

إِنَّ بەڕاستی

فِي ذَٰلِكَ لە مانەدا (هەیە)

لَآيَاتٍ چەندەها بەڵگەی گەورە ودیار

لِّكُلِّ بۆگشت

صَبَّارٍ کەسێک زۆر ئارامگربێت

شَكُورٍ (کەسێک) زۆرسوپاسگوزار بێت

(٣١)

وَإِذَا و کاتێک

غَشِيَهُم دایپۆشین

مَّوْجٌ شەپۆلێک

كَالظُّلَلِ وەک هەورەکان

دَعَوُا (ئەوە) هاواری-

اللَّـهَ خودا -دەکەن

مُخْلِصِينَ دڵسۆزانە

لَهُ بۆی

الدِّينَ ملکەچ ن

فَلَمَّا جا هەرکە

نَجَّاهُمْ نەجاتی دان وڕزگاری کردن

إِلَى بۆ

الْبَرِّ وشکانی

فَمِنْهُم ئەوە لەوان (هەیە)

مُّقْتَصِدٌ مام ناوەندییە

وَمَا يَجْحَدُ و نکۆڵی ناکات

بِآيَاتِنَا لە بەڵگەگەورەکانمان

إِلَّا تەنها

كُلُّ هەموو ئەوانەی

خَتَّارٍ زۆرپەیمانەشکێنی

كَفُورٍ زۆر سپڵەی ناشوكرن

(٣٢)

يَا أَيُّهَا هۆ ئەی

النَّاسُ خەلک ینە

اتَّقُوا بترسن و خۆ بپارێزن لە

رَبَّكُمْ پەروەردیگارتان

وَاخْشَوْا وبترسن لە

يَوْمًا ڕۆژێک

لَّا يَجْزِي پاداشت ناداتەوە

وَالِدٌ (هیچ) باوکێک

عَن لەجیان

وَلَدِهِ ڕۆڵەکەی

وَلَا مَوْلُودٌ و (نە هیچ) لەدایکبوێکیش

هُوَ خۆی

جَازٍ پاداشدەرەوەیە

عَن لەجیات

وَالِدِهِ باوکی

شَيْئًا بەهیچ شتێ

إِنَّ بێگومان

وَعْدَ بەڵێنی

اللَّـهِ خودا

حَقٌّ ۖ حەق و ڕاست و دروستە

فَلَا تَغُرَّنَّكُمُ جا با هەڵتان نەخەڵەتێنێ و مەغرورتان نەکات

الْحَيَاةُ ژیانی

الدُّنْيَا دونیا و نزم

وَلَا يَغُرَّنَّكُم وفریوتان نەدات و مەغرورتان نەکات

بِاللَّـهِ بەخودا

الْغَرُورُ هەرچی فریودەر ومەغرورکەرێکە

(٣٣)
