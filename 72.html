سَمِعْنَا گوێمان لێبوو لە

تَعَالَىٰ -

رَبِّنَا پەروەردیگارمان (-بەرز و باڵاترە)

مَااتَّخَذَ بۆخۆی دانەناوە

صَاحِبَةً هیچ دۆست و هاوسەرێک

وَلَا وَلَدًا و نە هیچ ڕۆڵەێکیش

(٣)

وَأَنَّهُ كَانَ و بێگومان جاران

يَقُولُ -

سَفِيهُنَا نەفام و بێ ئاوەزەکەمان

عَلَى دەربارەی

اللَّـهِ خودا

شَطَطًا قسەی ناڕەوا ونابەجێ (كوفریان) (-دەگوت)

(٤)

ظَنَنَّا وامان دەزانی

أَن کە

لَّن تَقُولَ -

الْإِنسُ ئادەمیزاد

وَالْجِنُّ وجنۆکە -هەرگیز ناڵێن (و هەڵنابەستن)

عَلَى بەسەر

اللَّـهِ خودا

كَذِبًا هیچ درۆیەک

(٥)

وَأَنَّهُ و بەڕاستی

كَانَ جاران

رِجَالٌ پیاوانێک

مِّنَ لە

الْإِنسِ مرۆڤەکان

يَعُوذُونَ هانایان دەبرد و پەناگریان دەکرد

بِرِجَالٍ بە پیاوانێک

مِّنَ لە

الْجِنِّ جنۆکەکان

فَزَادُوهُمْ جا ئەوان بۆیان زیادکردن

رَهَقًا ماندوێتی

(٦)

وَأَنَّهُمْ و بێگومان ئەوان

ظَنُّوا وایان زانی

كَمَا هەرچۆن

ظَنَنتُمْ ئێوەش واتان زانی

أَن کە

لَّن يَبْعَثَ -

اللَّـهُ خودای گەورە (-هەرگیز نانێرێت)

أَحَدًا هیچ یەکێک

(٧)

وَأَنَّا و بەڕاستی ئێمە

لَمَسْنَا السَّمَاءَ بە ئاسمان کەوتین

فَوَجَدْنَاهَا ئیتر بینیمان (ئاسمان)

مُلِئَتْ پڕکراوە لە

حَرَسًا پاسەوانانی

شَدِيدًا توند و بەهێز

وَشُهُبًا لەگەل نەیزەکان و (ئەستێرە -ڕاژان)

(٨)

وَأَنَّا و لە ڕاستیدا ئێمە

كُنَّا پێشتر

نَقْعُدُ دادەنیشتین

مِنْهَا لێی

مَقَاعِدَ دانیشتگەی زۆر

لِلسَّمْعِ بۆ بیستن

فَمَن جا ئەوەی

يَسْتَمِعِ گوێبگرێ

الْآنَ ئێستا

يَجِدْ دەبینێت

لە بۆی

شِهَابًا چەندین ئەستێرەرێژێک (کە)

رَّصَدًا ئاڕاستە یکراوە (لەبۆسە دا ئامادەو چاودێرن)

(٩)

وَأَنَّا لەڕاستیدا ئێمە

لَا نَدْرِي نازانین

أَ ئاخۆ

شَرٌّ زیان

أُرِيدَ ویستراوە

بِمَن بۆئەوانەی

فِي لە

الْأَرْضِ زەویدان

أَمْ یان

أَرَادَ بِهِمْ -

رَبُّهُمْ پەروەردگاریان

رَشَدًا ڕێ نیشاندان و هیدایەت ی (-ویستووە بۆیان)

(١٠)

وَأَنَّا و بەدڵنیایی

مِنَّا تێماندا هەیە

الصَّالِحُونَ کەسە چاکەکان

وَمِنَّا و لە ئێمەشدا هەن

دُونَ خوارتر لە

ذَٰلِكَ ئەوان

كُنَّا -پێشتر ئێمە

طَرَائِقَ چەندین ڕێچکە وڕێباز ی

قِدَدًا پارچەپارچە (-بووین)

(١١)

وَأَنَّا و بەدڵنیایی ئێمە

ظَنَنَّا (دڵنیابووین)

أَن کە

لَّن نُّعْجِزَ هەرگیز -

اللَّـهَ خودا

فِي لە

الْأَرْضِ زەویدا (-دۆشداناهێنین و دەستەوەسان ناکەیین)

وَلَن وهەرگیز

نُّعْجِزَهُ دۆشی داناهێنین

هَرَبًا بە هەڵاتنیش

(١٢)

وَأَنَّا و بەڕاستی ئێمە

لَمَّا کاتێ

سَمِعْنَا گوێمان لە -

الْهُدَىٰ ڕێنمونی یەکە (قورئان) (-بوو)

آمَنَّا باوەڕمان هێنا

بِهِ پێی

فَمَن جا هەرکەسێ

يُؤْمِن باوەڕ بهێنێ

بِرَبِّهِ بە پەروەردگاری

فَلَا يَخَافُ ئەوە ناترسێ لە

بَخْسًا هیچ کەمکردن و بێ نرخ کردنێک (بۆ کارەچاکەکانی)

وَلَارَهَقًا و نە هیچ بارکردن و شەکەت کردنێک (بە زیادکردنی تاوان)

(١٣)

وَأَنَّا و بەدڵنیایی ئێمە

مِنَّا تێماندایە

الْمُسْلِمُونَ موسڵمانەکان

وَمِنَّا و تێماندایە

فَمَنْ جا ئەوەی

أَسْلَمَ ملکەچی کردوە (بۆ خودا وموسڵمان بووە)

تَحَرَّوْا گەڕاون ودۆزیتیانەوە

رَشَدًا ڕێیەکی حەق

(١٤)
