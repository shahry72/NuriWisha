أَكۡثَر زۆربەی

ٱلنَّاس خەلک

لَايَعۡلَمُون نازانن

(٢٦)

و لِلَّه هەموو هی خودایە

مُلۡك خاوەنداری و پادشایی

ٱلسَّمَٰوَٰت ئاسمانەکان

و ٱلۡأَرۡض و زەوی

وَيَوۡم و ئەو ڕۆژەی

تَقُوم دادێت و هەڵدەستێ

ٱلسَّاعَة ڕۆژی قیامەت

يَوۡمَئِذٖ ئەوڕۆژە

يَخۡسَر خەسارەتمەند دەبن

ٱلۡمُبۡطِلُون ناحەقان و پڕووپووچ بێژان

(٢٧)

وَتَرَى و دەبینی

كُلّ هەموو

أُمَّةٖ ئوممەت و گەلێک

جَاثِيَة لەسەرچۆکان دانیشتووە

كُلّ هەموو

أُمَّةٖ ئوممەت و گەلێک

تُدۡعَىٰٓ بانگدەکرێ

إِلَى بۆ ، بۆلای

كِتَٰبِهَا کیتاب و نامەی کردارەکانیان

ٱلۡيَوۡم ئیمڕۆ

تُجۡزَوۡن سزادەدرێن و پاداشت دەدرێنەوە بەوەی

مَا كُنتُمۡ تَعۡمَلُون پێشان دەتانکرد

(٢٨)

هَٰذَا ئەوەیە

كِتَٰبُنَا تێنووس و نوسراوی ئێمە

يَنطِق بەدەنگ دێ

عَلَيۡكُم لەسەرتان

بِٱلۡحَقّ بە حەق و ڕاستی و ڕەوایی

إِنَّا بەراستی ئێمە

كُنَّا پێشان

نَسۡتَنسِخ لەبەرماندەگرتەوە و وێنەماندەگرت

مَا كُنتُمۡ تَعۡمَلُون ئەوەی ئێوە دەتانکرد

(٢٩)

فَأَمَّا جا

ٱلَّذِین ئەو(کەس) انەی

ءَامَنُوا بڕوایان هێنا

وَعَمِلُوا و کردوویانە

ٱلصَّٰلِحَٰت کارو کردەوە چاکەکان

فَيُدۡخِلُهُمۡ ئەوە-

رَبُّهُمۡ پەروەردگاریان (-دەیانخاتە)

فِي ناو

رَحۡمَتِهِۦ میهرەبانی خۆی (بەهەشت)

ذَٰلِك ئەویان

هُو (بەڕاستی)

ٱلۡفَوۡز بردنەوەی

ٱلۡمُبِين ڕۆن و ئاشکرایە

(٣٠)

وَأَمَّا و هەرچی

ٱلَّذِین ئەوانەی

كَفَرُوٓا کوفریانکردو بڕوایان نەهێنا

أَفَلَمۡ تَكُنۡ ئەوە نەبوو

ءَايَٰتِي ئایەتەکانم

تُتۡلَى دەخوێندرایەوە

عَلَيۡكُمۡ بەسەرتان ، لەسەرتان

فَٱسۡتَكۡبَرۡتُمۡ ئیتر

وَكُنتُمۡ و...بوون

قَوۡمٗا قەومێ

مُّجۡرِمِين تاوانبار (-بوون)

(٣١)

وَإِذَا وئەگەر

قِيل گوترا

إِنّ بەدڵنیایی

وَعۡد بەڵێنی

ٱللَّه خودا

حَقّٞ هەق و ڕاستە

وَٱلسَّاعَة و ڕۆژی قیامەت

لَارَيۡب بچوکترین گومان نیە

فِيهَا تیایدا

قُلۡتُم گوتتان

مَّا ئێمە نا

نَدۡرِي (نا) زانین

مَا ٱلسَّاعَة چییە ڕۆژی قیامەت

إِن -

نَّظُنّ گومان(-نا) کەین

إِلَّا جگەلە

ظَنّٗا گومانێک

وَمَا -

نَحۡن ئێمە

بِمُسۡتَيۡقِنِين یەقین ودڵنیایمان(نیە)

(٣٢)

وَبَدَا و دەرکەوت

لَهُمۡ بۆیان

سَيِّ‍َٔات خراپەکانی

مَا عَمِلُوا ئەوەی کردییان

وَحَاق و گیربوو

بِهِم لێیان

مَّا ئەوەی

كَانُوا ئەوان پێشان

بِهِۦ -

يَسۡتَهۡزِءُون گەمەو گاڵتەیان دەکرد (-پێی)

(٣٣)

وَقِيل و گوترا

ٱلۡيَوۡم ئیمڕۆ

نَنسَىٰكُمۡ بەجێتان دەهێڵین و لەبیرتان دەکەین

كَمَا هەرچۆن

نَسِيتُمۡ لەبیرتانکرد

لِقَآء پێک گەیشتنی

يَوۡمِكُمۡ ڕۆژەکەتان

هَٰذَا ئەوەیە

وَمَأۡوَىٰكُم و ماڵتان

ٱلنَّار ئاگر

وَمَا ونیە

لَكُم بۆتان

مِّن (هیچ))

نَّٰصِرِين یارمەتیدەر و سەرخەرانێک

(٣٤)

ذَٰلِكُم ئەویان

بِأَنَّكُم بە(هۆی) ئەوەی

ٱتَّخَذۡتُمۡ -

ءَايَٰت ئایەتەکانی قورئان ، نیشانە گەورەکان ، موعجیزاتی

ٱللَّه خوداتان (-کردبووە)

هُزُوٗا گاڵتەجاڕی

وَغَرَّتۡكُم و فریووی دان

ٱلۡحَيَوٰة ژیانی

ٱلدُّنۡيَا دونیا

فَٱلۡيَوۡم جا ئیمڕۆ

لَايُخۡرَجُون دەرناخرێن

مِنۡهَا لەوێ

وَلَاهُمۡ و نە ئەوانیش

يُسۡتَعۡتَبُون گلەییان لێدەکرێ

(٣٥)

فلِلَّه بۆ خودایە

ٱلۡحَمۡد هەرچی سوپاس و ستایشە

رَبّ پەروەردیگار و خاوەنی

ٱلسَّمَٰوَٰت ئاسمانەکان

وَرَبّ و خاوەن و پەروەردگاری

ٱلۡأَرۡض زەوی

رَبّ پەروەردیگار و خاوەنی

ٱلۡعَٰلَمِين جیهانیان

(٣٦)

وَلَه هەر هی ئەوە

ٱلۡكِبۡرِيَآء گەورەیی و بەرزی

فِي لە

ٱلسَّمَٰوَٰت ئاسمانەکان

و ٱلۡأَرۡض و زەوی

وَهُو و ئەوە

ٱلۡعَزِيز ئەوپەڕی دەسەڵاتداری باڵادەست (ناوێکی خودای پیرۆزە)

ٱلۡحَكِيم دانا و تەواو کاربەجێی

(٣٧)
