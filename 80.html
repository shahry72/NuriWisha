بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ عَبَسَ


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبە بەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


عَبَسَ ڕووی گرژ کرد

وَتَوَلَّىٰ و پشتی لێکرد

(١)

أَن بە(هۆی ئە) وەی کە

جَاءَهُ هات بۆلای

الْأَعْمَىٰ نابیناکە

(٢)

وَمَا يُدْرِيكَ جا تۆچوزانی

لَعَلَّهُ بەلكو ئەو

يَزَّكَّىٰ خۆ خاوێن و پاك بکاتەوە

(٣)

أَوْ یان

يَذَّكَّرُ یاد بکاتەوە

فَتَنفَعَهُ ئەوسا

الذِّكْرَىٰ یادکردنەوە و ئامۆژگاری

(٤)

أَمَّا هەرچی

مَنِ ئەوەی

اسْتَغْنَىٰ خۆی بێ نیازگرت و خۆی بەدەوڵەمەند زانی

(٥)

فَأَنتَ ئەوە تۆ

لَهُ بۆی

تَصَدَّىٰ لەڕوو ڕادەوەستی و پێشی پێدەگری

(٦)

وَمَا عَلَيْكَ و چی لە سەر تۆیە ؟ (لەسەر تۆ نیە)

أَلَّا يَزَّكَّىٰ کە خۆ پاك نەکا (بڕوانەهێنێ)

(٧)

وَأَمَّا و هەرچی

مَن ئەوەی

جَاءَكَ هات بۆ لات

يَسْعَىٰ کۆششی دەکرد و چوست دەهات

(٨)

وَهُوَ و لەحاڵێ ئەو

يَخْشَىٰ دەترسا

(٩)

فَأَنتَ ئیتر تۆخۆت

عَنْهُ لێی

تَلَهَّىٰ بێ ئاگا دەکرد

(١٠)

كَلَّا نا ، نابێ!

إِنَّهَا بەڕاستی ئەوە

تَذْكِرَةٌ یادخستنەوە و ئامۆژگارییەکە..

(١١)

فَمَن جا هەرکەسێ

شَاءَ ویستی

ذَكَرَهُ یادی دەکاتەوە و باسی دەکا

(١٢)

فِي لەناو

صُحُفٍ پەڕاوێکی

مُّكَرَّمَةٍ ڕێزلێنراوی

(١٣)

مَّرْفُوعَةٍ بەرزکراوەی

مُّطَهَّرَةٍ پاكڕاگیراو دایە

(١٤)

بِأَيْدِي بەدەستی

سَفَرَةٍ نوسەرانێ (کە)

(١٥)

كِرَامٍ ڕێزداری

بَرَرَةٍ چاکی خواناسی ملکەچن

(١٦)

قُتِلَ بەکوشتچێ !

الْإِنسَانُ مرۆڤ ،

مَا چەندە (چی)

أَكْفَرَهُ کافر وسپڵەیە (ئاوا کافری کردوە)

(١٧)

مِنْ لە

أَيِّ چی

شَيْءٍ شتێ

خَلَقَهُ دروستی کردوە !

(١٨)

مِن لە

نُّطْفَةٍ تنۆکێ

خَلَقَهُ خەلقی کردوە

فَقَدَّرَهُ ئینجا بڕی بۆ داناوە

(١٩)

ثُمَّ پاشان

السَّبِيلَ ڕێگاکەی

يَسَّرَهُ ئاسان کردوە

(٢٠)

ثُمَّ پاشان

أَمَاتَهُ مراندویەتی

فَأَقْبَرَهُ ئیتر لە گۆڕی ناشتوە

(٢١)

ثُمَّ پاشان (خودا)

إِذَا کە

شَاءَ ویستی

أَنشَرَهُ (زیندووی دەکاتەوە و بڵاوی دەکاتەوە)

(٢٢)

كَلَّا نا ، نەخێر!

لَمَّا هێشتا

يَقْضِ بەجێی نەهێناوە

مَا ئەوەی

أَمَرَهُ فەرمانی پێکردوە

(٢٣)

فَلْيَنظُرِ با

الْإِنسَانُ مرۆڤ

إِلَىٰ بۆ

طَعَامِهِ خواردنەکەی

(٢٤)

أَنَّا بێگومان ئێمە

صَبَبْنَا ڕوومانکرد و ڕژاندمان

الْمَاءَ ئاو مان

صَبًّا ڕووکردنێ !

(٢٥)

ثُمَّ پاشان

شَقَقْنَا قڵیشاندومانە

الْأَرْضَ زەوی

شَقًّا قڵیشاندنێ !

(٢٦)

فَأَنبَتْنَا جا ڕواندوومانە

فِيهَا تیایدا

حَبًّا دەنەوێڵەیێ!

(٢٧)

وَعِنَبًا و ترێیێ ..

وَقَضْبًا و سەوزەواتێ ..

(٢٨)

وَزَيْتُونًا و زەیتونێ..

وَنَخْلًا و دارخورمایێ

(٢٩)

وَحَدَائِقَ و باغاتێکی

غُلْبًا چڕ و پڕ

(٣٠)

وَفَاكِهَةً و میوەجاتێ..

وَأَبًّا لەگەڵ گژ و گیایێ..

(٣١)

مَّتَاعًا سود و ڕابواردنێکە

لَّكُمْ بۆخۆتان

وَلِأَنْعَامِكُمْ و بۆ ئاژەڵەکانتان

(٣٢)

فَإِذَا جا کە

جَاءَتِ هات

الصَّاخَّةُ صاخ خە (دەنگێکی زۆر مەزنە قیامەت هەڵدەسێ لەگەڵی)

(٣٣)

يَوْمَ ئەوڕۆژەیە

يَفِرُّ ڕادەکا

الْمَرْءُ (مرۆڤ)

مِنْ لە

أَخِيهِ برای خۆی

(٣٤)

وَأُمِّهِ و (لە) دایکی

وَأَبِيهِ و (لە) باوکی

(٣٥)

وَصَاحِبَتِهِ وە(لە) دۆست و هاوسەرەکەی

وَبَنِيهِ و (لە) كوڕەکانی

(٣٦)

لِكُلِّ (هەیە) بۆ هەر

امْرِئٍ کەسێ

مِّنْهُمْ لەوان

يَوْمَئِذٍ ئەو ڕۆژە

شَأْنٌ وەزع وحاڵێک

يُغْنِيهِ بێ نیازی بکا

(٣٧)

وُجُوهٌ چەندەها روخسار

يَوْمَئِذٍ ئەو ڕۆژە

مُّسْفِرَةٌ بەجۆرێ ڕوون و گەش و جوانن

(٣٨)

ضَاحِكَةٌ دەم بەپێکەنین پێکەنینێ..

مُّسْتَبْشِرَةٌ چاوەوانی موژدەن

(٣٩)

وَوُجُوهٌ و چەندها ڕووش

يَوْمَئِذٍ ئەو ڕۆژە

عَلَيْهَا لەسەرییەتی

غَبَرَةٌ تۆز و خۆڵی

(٤٠)

تَرْهَقُهَا شەکەت و داماوی کردون

قَتَرَةٌ ڕەشاتیێ!

(٤١)

أُولَـٰئِكَ هُمُ بەڕاستی هەر ئەوانن

الْكَفَرَةُ کافرە

الْفَجَرَةُ تاوانبار و فاجرەکان

(٤٢)
