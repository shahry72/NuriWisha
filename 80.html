بۆ زانینی مانای وشەیەك پەنجە بنێ سەری
سورة عَبَسَ

بِسْمِ بەناوی

اللَّـهِ خودا

الرَّحْمَـٰنِ ئەوپەڕی میهرەبانی

الرَّحِيمِ ئەوپەڕی بەخشندە

عَبَسَ ڕووی گرژ كرد

وَتَوَلَّىٰ و پشتی لێكرد

(١)

أَن بە(هۆی ئە) وەی كە

جَاءَهُ هات بۆلای

الْأَعْمَىٰ نابیناكە

(٢)

وَمَا يُدْرِيكَ جا تۆچوزانی

لَعَلَّهُ بەلكو ئەو

يَزَّكَّىٰ خۆ خاوێن و پاك بكاتەوە

(٣)

أَوْ یان

يَذَّكَّرُ یاد بكاتەوە

فَتَنفَعَهُ ئەوسا

الذِّكْرَىٰ یادكردنەوە و ئامۆژگاری

(٤)

أَمَّا هەرچی

مَنِ ئەوەی

اسْتَغْنَىٰ خۆی بێ نیازگرت و خۆی بەدەوڵەمەند زانی

(٥)

فَأَنتَ ئەوە تۆ

لَهُ بۆی

تَصَدَّىٰ لەڕوو ڕادەوەستی و پێشی پێدەگری

(٦)

وَمَا عَلَيْكَ و چی لەسەرتۆیە ؟ ( لەسەرتۆ نیە)

أَلَّا يَزَّكَّىٰ کە خۆ پاك نەكا (بڕوانەهێنێ)

(٧)

وَأَمَّا و هەرچی

مَن ئەوەی

جَاءَكَ هات بۆ لات

يَسْعَىٰ كۆششی دەكرد و چوست دەهات

(٨)

وَهُوَ و لەحاڵێ ئەو

يَخْشَىٰ دەترسا

(٩)

فَأَنتَ ئیتر تۆخۆت

عَنْهُ لێی

تَلَهَّىٰ بێ ئاگا دەكرد

(١٠)

كَلَّا نا ، نابێ!

إِنَّهَا بەڕاستی ئەوە

تَذْكِرَةٌ یادخستنەوە و ئامۆژگارییەكە..

(١١)

فَمَن جا هەركەسێ

شَاءَ ویستی

ذَكَرَهُ یادی دەكاتەوە و باسی دەكا

(١٢)

فِي لەناو

صُحُفٍ پەڕاوێكی

مُّكَرَّمَةٍ ڕێزلێنراوی

(١٣)

مَّرْفُوعَةٍ بەرزكراوەی

مُّطَهَّرَةٍ پاكڕاگیراو دایە

(١٤)

بِأَيْدِي بەدەستی

سَفَرَةٍ نوسەرانێ (كە)

(١٥)

كِرَامٍ ڕێزداری

بَرَرَةٍ چاكی خواناسی ملكەچن

(١٦)

قُتِلَ بەكوشتچێ !

الْإِنسَانُ مرۆڤ،

مَا چەندە (چی)

أَكْفَرَهُ كافر وسپڵەیە (ئاوا كافری كردووە)

(١٧)

مِنْ لە

أَيِّ چی

شَيْءٍ شتێ

خَلَقَهُ دروستی كردووە !

(١٨)

مِن لە

نُّطْفَةٍ تنۆكێ

خَلَقَهُ خەلقی كردووە

فَقَدَّرَهُ ئینجا بڕی بۆ داناوە

(١٩)

ثُمَّ پاشان

السَّبِيلَ ڕێگاكەی

يَسَّرَهُ ئاسان كردووە

(٢٠)

ثُمَّ پاشان

أَمَاتَهُ مراندویەتی

فَأَقْبَرَهُ ئیتر لە گۆڕی ناوە

(٢١)

ثُمَّ پاشان (خودا)

إِذَا كە

شَاءَ ویستی

أَنشَرَهُ (زیندووی دەكاتەوە و بڵاوی دەكاتەوە)

(٢٢)

كَلَّا نا ، نەخێر!

لَمَّا هێشتا

يَقْضِ بەجێی نەهێناوە

مَا ئەوەی

أَمَرَهُ فەرمانی پێكردووە

(٢٣)

فَلْيَنظُرِ با

الْإِنسَانُ مرۆڤ

إِلَىٰ بۆ

طَعَامِهِ خواردنەكەی

(٢٤)

أَنَّا بێگومان ئێمە

صَبَبْنَا ڕوومانكرد و ڕژاندمان

الْمَاءَ ئاو مان

صَبًّا ڕووكردنێ !

(٢٥)

ثُمَّ پاشان

شَقَقْنَا قڵیشاندومانە

الْأَرْضَ زەوی

شَقًّا قڵیشاندنێ !

(٢٦)

فَأَنبَتْنَا جا ڕواندوومانە

فِيهَا تیایدا

حَبًّا دەنەوێڵەیێ!

(٢٧)

وَعِنَبًا و ترێیێ ..

وَقَضْبًا و سەوزەواتێ ..

(٢٨)

وَزَيْتُونًا و زەیتونێ..

وَنَخْلًا و دارخورمایێ

(٢٩)

.. وَحَدَائِقَ و باغاتێكی

غُلْبًا چڕ و پڕ

(٣٠)

وَفَاكِهَةً و میوەجاتێ..

وَأَبًّا لەگەڵ گژ و گیایێ..

(٣١)

مَّتَاعًا سود و ڕابواردنێكە

لَّكُمْ بۆخۆتان

وَلِأَنْعَامِكُمْ و بۆ ئاژەڵەكانتان

(٣٢)

فَإِذَا جا كە

جَاءَتِ هات

الصَّاخَّةُ صاخ خە (دەنگێكی زۆر گەورەیە قیامەت هەڵدەسێ لەگەڵی)

(٣٣)

يَوْمَ ئەوڕۆژەیە

يَفِرُّ ڕادەكا

الْمَرْءُ (مرۆڤ)

مِنْ لە

أَخِيهِ برای خۆی

(٣٤)

وَأُمِّهِ و (لە) دایكی

وَأَبِيهِ و(لە ) باوكی

(٣٥)

وَصَاحِبَتِهِ وە(لە) دۆست و هاوسەرەكەی

وَبَنِيهِ و (لە ) كوڕەكانی

(٣٦)

لِكُلِّ بۆ هەر

امْرِئٍ كەسێ

مِّنْهُمْ لەوان

يَوْمَئِذٍ ئەوڕۆژە (هەیە)

شَأْنٌ وەزع وحاڵێ

يُغْنِيهِ بێ نیازی بكا

(٣٧)

وُجُوهٌ چەندەها ڕوو

يَوْمَئِذٍ ئەو ڕۆژە

مُّسْفِرَةٌ بەجۆرێ ڕوون و گەش وجوانن ..

(٣٨)

ضَاحِكَةٌ دەم بەپێكەنین پێكەنینێ..

مُّسْتَبْشِرَةٌ چاوەوانی موژدەن

(٣٩)

وَوُجُوهٌ و چەندها ڕووش

يَوْمَئِذٍ ئەو ڕۆژە

عَلَيْهَا لەسەرییەتی

غَبَرَةٌ تۆز و خۆڵی

(٤٠)

تَرْهَقُهَا شەكەت و داماوی كردون

قَتَرَةٌ ڕەشاتیێ!

(٤١)

أُولَـٰئِكَ هُمُ ئەوانەن

الْكَفَرَةُ هەرئەوانەن كافری

الْفَجَرَةُ تاوانبارو فاجرەكان

(٤٢)
لێکدانەوە و تەفسیری ئایەتەکان
گەڕانەوە بۆ پێڕست
