عُطِّلَتْ پەک خران (پشتگوێ خران)

(٤)

وَإِذَا و کاتێ

الْوُحُوشُ جانەوەران و دڕندەکان

حُشِرَتْ لە شوێنێ کۆکرانەوە

(٥)

وَإِذَا وەکاتێ

(٦)

النُّفُوسُ نەفسەکان

زُوِّجَتْ جوتكران

(٧)

الْمَوْءُودَةُ كچۆڵەی زیندەبەچاڵکراو

سُئِلَتْ پرسیاری لێکرا

(٨)

قُتِلَتْ كوژرا

(٩)

وَإِذَا و کاتێ

الصُّحُفُ نامەی کردەوەکان

(١٠)

السَّمَاءُ ئاسمان

كُشِطَتْ داماڵێنرا

(١١)

وَإِذَا وەکاتێ

الْجَحِيمُ دۆزەخ

سُعِّرَتْ لەو پەڕی گەرمی جۆش درا

(١٢)

وَإِذَا وەکاتێ

الْجَنَّةُ بەهەشت

أُزْلِفَتْ نزیک كرایەوە

(١٣)

عَلِمَتْ(ئەوکاتە)

نَفْسٌ هەمو) نەفسێ

مَّا چی

أَحْضَرَتْ ئامادەکردوە

(١٤)

فَلَا أُقْسِمُ جا سوێند

بِالْخُنَّسِ بە خۆبزرکەرەوەانی

(١٥)

عَسْعَسَ پاشەکشەی کرد

(١٧)

تَنَفَّسَ هەناسەی دا

(١٨)

رَسُولٍ نێردراوێکی

كَرِيمٍ پایەبەرزە

(١٩)

مَكِينٍ خاوەن مەکانەتێکە..

(٢٠)

أَمِينٍ ئەمین وباوەڕپێکراو

(٢١)

بِمَجْنُونٍ(-هەرگیز) شێت

(٢٢)

رَآهُ ئەوی بینی

بِالْأُفُقِ لە ئاسۆی

الْمُبِينِ ڕۆن و دیار

(٢٣)

شَيْطَانٍ شەیتانێکی

رَّجِيمٍ نەفرین کراو
