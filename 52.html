بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ الطُّور 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبە بەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


وَالطُّورِ سوێند بە کێوی طور

(١)

وَكِتَابٍ وسوێند بە کیتابێکی

مَّسْطُورٍ خەت خەت نوسراو

(٢)

فِي لە

رَقٍّ پێستە پەڕێکی

مَّنشُورٍ بڵاوکراو..

(٣)

وَالْبَيْتِ سوێند بەو ماڵەی

الْمَعْمُورِ ئاوەدانە

(٤)

وَالسَّقْفِ و سوێند بە بانی

الْمَرْفُوعِ بەرز وبڵندکراو

(٥)

وَالْبَحْرِ و سوێند بە دەریای

الْمَسْجُورِ داگیرسێنراو ،

(٦)

إِنَّ بەدڵنیایی

عَذَابَ عەزب و ئەشکەنجەی

رَبِّكَ پەروەردیگارت

لَوَاقِعٌ هەر هاتۆتەجێ

(٧)

مَّا لَهُ نیە بۆی

مِن (هیچ)

دَافِعٍ پاڵنەرولابەرێ

(٨)

يَوْمَ ڕۆژێ

تَمُورُ ڕادەهەژێت

السَّمَاءُ ئاسمان

مَوْرًا ڕاهەژانێ ..

(٩)

وَتَسِيرُ -

الْجِبَالُ و چیاکان -دەڕۆن

سَيْرًا ڕۆیشتنێ..

(١٠)

فَوَيْلٌ جا-

يَوْمَئِذٍ ئەو ڕۆژە -واویلایێ ..

لِّلْمُكَذِّبِينَ بۆ بەدرۆدانەر و بێ بڕواکان

(١١)

الَّذِينَ هُمْ ئەوانەی

فِي لە

خَوْضٍ بەناوکەوتن (ی درۆ وناحەقیا)

يَلْعَبُونَ یاری دەکەن

(١٢)

يَوْمَ ڕۆژێ

يُدَعُّونَ بەتوندی پاڵدەدرێن

إِلَىٰ بۆناو

نَارِ ئاگری

جَهَنَّمَ دۆزەخ

دَعًّا پاڵدرانێ..

(١٣)

هَـٰذِهِ ئەوەیە

النَّارُ الَّتِي ئەو ئاگرەی کە

كُنتُم ئێوە جاران

بِهَا -

تُكَذِّبُونَ بەدرۆ و دادەنا و باوەڕتان نەدەکرد -پێی

(١٤)

أَفَسِحْرٌ ئایا سیحر و جادووە

هَـٰذَا ئەمە؟

أَمْ یان

أَنتُمْ ئێوە

لَا تُبْصِرُونَ نابینن؟

(١٥)

اصْلَوْهَا بڕۆنە ناوی و بسوتێن تیایدا

فَاصْبِرُوا جا ئارام بگرن

أَوْ یان

لَا تَصْبِرُوا ئارام مەگرن

سَوَاءٌ وەک یەکە

عَلَيْكُمْ لەسەرتان ،

إِنَّمَا هەر-

تُجْزَوْنَ پاداشتی (سزای) ئەوە دەدرێنەوە

مَا ئەوەی

كُنتُمْ جاران

تَعْمَلُونَ دەتاکرد

(١٦)

إِنَّ بەڕاستی

الْمُتَّقِينَ پارێزکار و لەخودا ترسان

فِي لە

جَنَّاتٍ باخاتێک

وَنَعِيمٍ و خۆشگوزەرانی و ناز و نیعمەتێکان

(١٧)

فَاكِهِينَ لە ڕابواردندا کەیف خۆشن

بِمَا بەوەی

آتَاهُمْ -

رَبُّهُمْ پەروەردگاریان (-پێیداوون)

وَوَقَاهُمْ و-

رَبُّهُمْ پەروەردگاریان (-پاراستوونی لە)

عَذَابَ عەزابی

الْجَحِيمِ دۆزەخ

(١٨)

كُلُوا بخۆن

وَاشْرَبُوا و بخۆنەوە

هَنِيئًا نۆشی گیانتان بێت

بِمَا بەوەی (بەرامبەر ئەوەی)

كُنتُمْ جاران

تَعْمَلُونَ دەتانکرد .

(١٩)

مُتَّكِئِينَ لێیان پاڵداوەتەوە

عَلَىٰ لەسەر

سُرُرٍ تەخت و قەنەفەی

مَّصْفُوفَةٍ جوان ڕێزکراو ،

وَزَوَّجْنَاهُم و جووتمان کردوون (لێکمان مارەکردوون)

بِحُورٍ لەگەڵ حۆری

عِينٍ چاوگەش و چاوگەورە

(٢٠)

وَالَّذِينَ و ئەوانەی

آمَنُوا بڕوایان هێناوە

وَاتَّبَعَتْهُمْ -

ذُرِّيَّتُهُم و نەوەکانیشیان (-دوایان کەوتوون)

بِإِيمَانٍ بە بڕوا و ئیمانێکەوە

أَلْحَقْنَا -

بِهِمْ (ئەوە) پێیان (-دەگەیەنین))

ذُرِّيَّتَهُمْ نەوەکانیان

وَمَا أَلَتْنَاهُم و(کەم ناکەینەوە)

مِّنْ لە

عَمَلِهِم کاری ئەوان

مِّن (لەهیچ)

شَيْءٍ شتێک ،

كُلُّ هەموو

امْرِئٍ کەسێک

بِمَا بەوەی

كَسَبَ وەدەستی هێناوە

رَهِينٌ بارمەتە و بەستراوە

(٢١)

وَأَمْدَدْنَاهُم وەبەردەوام (پێیان دەدەین)

بِفَاكِهَةٍ لە میوەیێ..

وَلَحْمٍ و گۆشتێ..

مِّمَّا لەوەی

يَشْتَهُونَ دڵیان بۆدەچێت و حەزی لێدەکەن

(٢٢)

يَتَنَازَعُونَ لێک وەردەگرن

فِيهَا تیایدا

كَأْسًا کاسەیێ..

لَّا لَغْوٌ هیچ لەغوەی-

فِيهَا تیادا (-نییە)

وَلَا تَأْثِيمٌ و نە هیچ تاوانێ..

(٢٣)

وَيَطُوفُ و دێن و دەچن

عَلَيْهِمْ بەدەوریاندا

غِلْمَانٌ خزمەت کارانێ ..

لَّهُمْ بۆیان

كَأَنَّهُمْ هەردەڵێی

لُؤْلُؤٌ گەوهەری..

مَّكْنُونٌ پاکی بێگەردی پارێزراون..

(٢٤)

وَأَقْبَلَ وە-

بَعْضُهُمْ هەندێکیان (-ڕوویان کردە)

عَلَىٰ بَعْضٍ (ئەوانی تر)

يَتَسَاءَلُونَ پرسیاریان دەکرد

(٢٥)

قَالُوا گوتیان

إِنَّا بەراستی

كُنَّا ئێمە-

قَبْلُ پێشان

فِي لەناو

أَهْلِنَا کەس و کاری خۆمان

مُشْفِقِينَ ترساوو (-بووین)

(٢٦)

فَمَنَّ جا-

اللَّـهُ خودا (-میننەت و فەزڵی کرد)

عَلَيْنَا لەسەرمان

وَوَقَانَا و ئێمەی پاراست لە

عَذَابَ عەزاب و ئەشکەنجەی

السَّمُومِ (دۆزەخ)

(٢٧)

إِنَّا ئێمە بەڕاستی

كُنَّا مِن قَبْلُ پێشان

نَدْعُوهُ هاوارمان دەکر بۆ خودا و لێی دەپاڕاینەوە ،

إِنَّهُ بەڕاسی ئەوە

هُوَ هەر ئەوە

الْبَرُّ ئەوپەڕی چاکەخوازی

الرَّحِيمُ ئەوپەڕی بەخشندەی میهرەبان

(٢٨)

فَذَكِّرْ ئیتریادبخەرەوە و ئامۆژگاری بکە

فَمَا أَنتَ تۆ-

بِنِعْمَتِ بە فەزڵ و نیعمەتی

رَبِّكَ پەروەردیگارت

بِكَاهِنٍ هیچ فاڵچی (-نیت)

وَلَا مَجْنُونٍ وهیچ شێت (-نیت)

(٢٩)

أَمْ یان

يَقُولُونَ دەڵێن

شَاعِرٌ شاعیرێکە

نَّتَرَبَّصُ بِهِ چاوەڕێ دەکەین-

رَيْبَ (ڕوداوەکانی)

الْمَنُونِ مردن و زەمانەی (تووش بێت و لەناوبچێ)

(٣٠)

قُلْ بڵێ

تَرَبَّصُوا دە چاوەڕوان بن

فَإِنِّي ئەوە بەدڵنیایی منیش

مَعَكُم لەگەڵتان

مِّنَ لە

الْمُتَرَبِّصِينَ چاوەڕوانکارانم

(٣١)

أَمْ یان

تَأْمُرُهُمْ -

أَحْلَامُهُم بیرمەندەکانیان (-فەرمانیان پێدەکا)

بِهَـٰذَا بەم شتە

أَمْ بەڵكو

هُمْ ئەوان

قَوْمٌ قەومێکی

طَاغُونَ لەسنوور دەرچوو ون

(٣٢)

أَمْ یان

يَقُولُونَ دەڵێن

تَقَوَّلَهُ (قسەیەکە) بەدەم خودا هەڵی بەستووە ،

بَل بەڵكو

لَّا يُؤْمِنُونَ بڕوا ناهێنن

(٣٣)

فَلْيَأْتُوا دەبا بهێنن

بِحَدِيثٍ قسەیەک

مِّثْلِهِ وەک وی

إِن ئەگەر

كَانُوا -

صَادِقِينَ ڕاستگۆ (-بووبن)

(٣٤)

أَمْ یان

خُلِقُوا دروستکراون

مِنْ لە

غَيْرِ (هیچ)

شَيْءٍ شتێ (و بەبێ هیچ هۆیەک)

أَمْ یان

هُمُ هەر خۆیان

الْخَالِقُونَ دروستکەر و بەدیهێنەرن

(٣٥)

أَمْ یان

خَلَقُوا ئەوان دروستیانکردوە

السَّمَاوَاتِ ئاسمانەکان

وَالْأَرْضَ و زەوی؟!

بَل بەڵكو

لَّا يُوقِنُونَ ناگەن بە یەقین و دڵنیا نابن

(٣٦)

أَمْ یان

عِندَهُمْ لای وانە

خَزَائِنُ خەزێنەکانی

رَبِّكَ پەروەردیگارت

أَمْ یان

هُمُ ئەوانن

الْمُصَيْطِرُونَ جلەوبەدەستان (لەم بوونەوەرە؟)

(٣٧)

أَمْ یان

لَهُمْ ئەوان هەیتیان

سُلَّمٌ گەلێک پەیژە

يَسْتَمِعُونَ کە گوێدەگرن

فِيهِ تییایدا؟ ،

فَلْيَأْتِ دەبا بهێنێ

مُسْتَمِعُهُم گوێگرەکەیان

بِسُلْطَانٍ بەڵگە یەکی

مُّبِينٍ دیار و ئاشكرا

(٣٨)

أَمْ یان

لَهُ -

الْبَنَاتُ كچان (-هی خودان)

وَلَكُمُ و-

الْبَنُونَ كوڕانیش (-هی ئێوەن؟) !

(٣٩)

أَمْ یان

تَسْأَلُهُمْ پرسیاروداوای-

أَجْرًا كرێ و پارەیەکیان (-لێدەکەی؟)

فَهُم جا ئەوان

مِّن بەهۆی (داواکردنی)

مَّغْرَمٍ بەرامبەرێک

مُّثْقَلُونَ بارگران بوون!

(٤٠)

أَمْ  یان

عِندَهُمُ  -

الْغَيْبُ  نهێنی و پەنهانیەکانی غەیبیان (-لایە)

فَهُمْ  ئیتر ئەوانن

يَكْتُبُونَ  دەنووسن؟!

(٤١)

أَمْ  یان

يُرِيدُونَ  -

كَيْدًا  پیلان و تەڵەکەیەکیان (-دەوێت و خەریکین؟)

فَالَّذِينَ  ئەوە ئەوانەی

كَفَرُوا  بێ بڕوا و کافر بوون

هُمُ  خۆیان

الْمَكِيدُونَ  تێکەوتەی پیلانگێڕین

(٤٢)

أَمْ  یان

لَهُمْ  -

إِلَـٰهٌ  خودایەکیان ((-هەیە))

غَيْرُ  جگە

اللَّـهِ  خودای پەروەردیگار؟! ،

سُبْحَانَ  پاکی و پیرۆزی بۆ

اللَّـهِ  خودای پەروەردیگار

عَمَّا  لەوەی

يُشْرِكُونَ  هاوەڵ و شەریکی بۆ دادەنێن ،

(٤٣)

وَإِن  و ئەگەر

يَرَوْا  ببینن

كِسْفًا  زۆر پارچە

مِّنَ  لە

السَّمَاءِ  ئاسمانەوە

سَاقِطًا  وا لەکەوتن دایە

يَقُولُوا  دەڵێن

سَحَابٌ  پەڵە هەورێکی

مَّرْكُومٌ  کەڵەکەبووە

(٤٤)

فَذَرْهُمْ  جا وازیان لێ بهێنە

حَتَّىٰ  تاكوو

يُلَاقُوا  دەگەن و توشی-

يَوْمَهُمُ  ڕۆژی خۆیان (-دێن)

الَّذِي  ئەوەی

فِيهِ  تیاییدا

يُصْعَقُونَ  دەمرن ولات و پات دەبن ،

(٤٥)

يَوْمَ  ئەو ڕۆژەی

لَا يُغْنِي  سود نابەخشێ

عَنْهُمْ  پێیان

كَيْدُهُمْ  پیلان و فێڵیان

شَيْئًا  هیچ

وَلَا هُمْ  و نە ئەوانیش

يُنصَرُونَ  پارمەتی دەدرێن سەردەخرێن (ڕزگاردەکرێن)

(٤٦)

وَإِنَّ  و بەڕاستی

لِلَّذِينَ  بۆ ئەوانەی

ظَلَمُوا  ستەم و كوفریان کردوە

عَذَابًا  عەزاب و ئەشکەنجەیەک هەیە

دُونَ  نزیکتر و خوارتر

ذَٰلِكَ  لەوە

وَلَـٰكِنَّ  بەڵام

أَكْثَرَهُمْ  زۆربەیان

لَا يَعْلَمُونَ  نازانن

(٤٧)

وَاصْبِرْ  و سەبر و ئارام بگرە

لِحُكْمِ  بۆ حوكم و بڕیاری

رَبِّكَ  پەروەردیگارت

فَإِنَّكَ  ئیتر تۆ

بِأَعْيُنِنَا  لەبەرچاو و لەژێر چاودێری ئێمەی

وَسَبِّحْ  و تەسبیحات بکە

بِحَمْدِ  هاوکات بە ستایشکردنی

رَبِّكَ  پەروەردیگارت

حِينَ  کاتێ

تَقُومُ  هەڵدەستی و ڕادەوەستی

(٤٨)

وَمِنَ  و لە

اللَّيْلِ  شەویش دا

فَسَبِّحْهُ  هەر تەسبیح بکە و خودا بە پاك و پیرۆز بگرە

وَإِدْبَارَ  و لەدوای پاشەکشی و ون بوونی

النُّجُومِ  ئەستێرەکان یش

(٤٩)
