سُورَةُ الْحَاقَّةُ 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕیبە بەزەیی و میهرەبان (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


الْحَاقَّةُ ئەل حاااق قە

(١)

مَا چی یە

الْحَاقَّةُ حاااق قە

(٢)

وَمَا أَدْرَاكَ و تۆ چوزانی چی یە

الْحَاقَّةُ حاققە

(٣)

كَذَّبَتْ -

ثَمُودُ ثەمود

وَعَادٌ و عاد

بِالْقَارِعَةِ قاریعە یان (-بەدرۆدانا)

(٤)

فَأَمَّا هەرچی

ثَمُودُ ثەموودە

فَأُهْلِكُوا ئەوە لەناوبران

بِالطَّاغِيَةِ بە لەڕاددە بەدەر ەکە

(٥)

وَأَمَّا و ئەمما

عَادٌ عاد

فَأُهْلِكُوا ئەوەلەناوبران

بِرِيحٍ بە بایێک ی

صَرْصَرٍ هەرەبەهێزی سارد و سەخت بەسڕەسڕ ی

عَاتِيَةٍ توندی سەرکێش وبەرپێنەگیراو

(٦)

سَخَّرَهَا تەسخیری کردبوو وناردبووی

عَلَيْهِمْ بەسەریانا

سَبْعَ حەوت

لَيَالٍ شەوان

وَثَمَانِيَةَ و هەشت

أَيَّامٍ ڕۆژان

حُسُومًا بێ وچان و لە سەر یەک و نەپچڕاوە (وبەیاكجار لەناوبەر)

فَتَرَى جا -

الْقَوْمَ ئەو قەومە (-دەبینی)

فِيهَا تیایدا

صَرْعَىٰ مردەی بێ گیان کەوتوو(بوون)

كَأَنَّهُمْ ئەوان هەروەک (بڵێی)

أَعْجَازُ قەدی

نَخْلٍ دارخورما ی

خَاوِيَةٍ ناو پوکی لەعارد کەوتوو بن

(٧)

فَهَلْ جا ئایا

تَرَىٰ دەبینی

لَهُم لەوان

مِّن (هیچ)

بَاقِيَةٍ ماوەبێ (نەفەوتاوێ)

(٨)

وَجَاءَ -

فِرْعَوْنُ و فیرعەون

وَمَن قَبْلَهُ وئەوانی پێش ئەویش

وَالْمُؤْتَفِكَاتُ و سەرو ژێرکراوەکانیش

بِالْخَاطِئَةِ خراپە (تاوانە) گەورەکەیان (شیرك) -بەرپا کرد

(٩)

فَعَصَوْا ئەوەبوو لایاندا ولاملیان کرد لە فەرمانی

رَسُولَ پەیامبەری

رَبِّهِمْ پەروەردگاریان

فَأَخَذَهُمْ ئیتر (خودا) گرتنی

أَخْذَةً گرتنێکی

رَّابِيَةً زۆر سەختی سەرەوزیاد

(١٠)

إِنَّا بێگومان ئێمە

لَمَّا کاتێک

طَغَى -

الْمَاءُ ئاو -لەسنووردەرچوو(بوە تۆفان)

حَمَلْنَاكُمْ ئێوەمان هەڵگرت

فِي لەناو

الْجَارِيَةِ (کەشتیە) بەڕێ کەوتوەکە

(١١)

لِنَجْعَلَهَا تا (کەشتیەکە) -

لَكُمْ بۆتان (-بکەینە)

تَذْكِرَةً بیرخەرەوە و پەندێ

وَتَعِيَهَا وەلەگوێی بگرێ (و تێی بگات)

أُذُنٌ گوێیەک ی

وَاعِيَةٌ گوێگری تێگەیشتوو

(١٢)

فَإِذَا جا هەرکە

نُفِخَ فووکرا

فِي لە

الصُّورِ کەڕەنا

نَفْخَةٌ فوێک

وَاحِدَةٌ یەک فوو

(١٣)

وَحُمِلَتِ و هەڵگیران

الْأَرْضُ زەوی

وَالْجِبَالُ و چیاکان

فَدُكَّتَا ئیتر پێکدادران و هەپرون هەپرون بوون

دَكَّةً پيكدادران و تەخت بوونێک

وَاحِدَةً بە جارێک

(١٤)

فَيَوْمَئِذٍ ئا ئەو ڕۆژە

وَقَعَتِ -

الْوَاقِعَةُ ڕوداوەکە (قیامەت) (-ڕویدا و بەرپا بوو)

(١٥)

وَانشَقَّتِ و -

السَّمَاءُ ئاسمان (-شەقی تێکەوت و لێک ترازا)

فَهِيَ ئیتر ئەو (ئاسمانە)

يَوْمَئِذٍ لەو ڕۆژەدا

وَاهِيَةٌ شل و شۆڵ وخاوە

(١٦)

وَالْمَلَكُ و فریشتەکان

عَلَىٰ لە

أَرْجَائِهَا دەوروبەری دان

وَيَحْمِلُ و-

عَرْشَ عەرشی

رَبِّكَ پەروەردیگارت

فَوْقَهُمْ لەسەریانەوە (-هەڵدەگرێت)

يَوْمَئِذٍ ئەوڕۆژە

ثَمَانِيَةٌ هەشت ێک

(١٧)

يَوْمَئِذٍ ئەوڕۆژە

تُعْرَضُونَ پیشان دەدرێن

لَا تَخْفَىٰ پەنهان نابێت

مِنكُمْ لە(بارەی) ئێوە

خَافِيَةٌ هیچ پەنهان و شاراوەیێک (بچوكترین شاراوەش)

(١٨)

فَأَمَّا مَنْ جا هەرکەسەی

أُوتِيَ پێی درا

كِتَابَهُ نوسراوی کردەوەکانی

بِيَمِينِهِ بە ڕاستەیەوە

فَيَقُولُ ئەوە دەڵێت

هَاؤُمُ ئەها بگرن

اقْرَءُوا بخوێننەوە

كِتَابِيَهْ کیتابەکەم

(١٩)

إِنِّي من بەڕاستی

ظَنَنتُ هەر وام دەزانی

أَنِّي مُلَاقٍ کەمن هەردەگەم بە

حِسَابِيَهْ حیسابەکەم

(٢٠)

فَهُوَ جا ئەو

فِي لەناو

عِيشَةٍ ژینێکی

رَّاضِيَةٍ (پڕ لە ڕەزامەندی دایە)

(٢١)

فِي لە ناو

جَنَّةٍ بەهەشتێکی

عَالِيَةٍ بەرز و بڵندە

(٢٢)

قُطُوفُهَا (بەرو میوەکانی)

دَانِيَةٌ نزمکراوە و نزیکەدەستە

(٢٣)

كُلُوا دە بخۆن

وَاشْرَبُوا و بخۆنەوە

هَنِيئًا نۆشی گیانتان(بێ)

بِمَا بەوەی

أَسْلَفْتُمْ پێشوو کردبووتان

فِي لە

الْأَيَّامِ ڕۆژەکانی

الْخَالِيَةِ بەسەرچوو

(٢٤)

وَأَمَّا هەرچی

مَنْ ئەوکەسەی

أُوتِيَ پێی درا

كِتَابَهُ کیتابەکەی

بِشِمَالِهِ بە چەپانی

فَيَقُولُ ئەوە دەڵێ

يَالَيْتَنِي خۆزگەم

لَمْ أُوتَ پێم نەدرابوا

كِتَابِيَهْ کیتابەکەم

(٢٥)

وَلَمْ أَدْرِ و نەشم زنیبوا

مَا چی یە

حِسَابِيَهْ حیسابەکەم

(٢٦)

يَالَيْتَهَا خۆزگە ئەو(مردنەیی کە مردم) یان

كَانَتِ الْقَاضِيَةَ لەناوبردە و کۆتاییم بوا

(٢٧)

مَاأَغْنَىٰ عَنِّي چیسودی بوو لۆم

مَالِيَهْ ۜ ماڵ و سامانەکەم

(٢٨)

هَلَكَ لەناوچوو

عَنِّي لێم

سُلْطَانِيَهْ دەسەڵاتەکەم

(٢٩)

خُذُوهُ دە ڕای بپێچن

فَغُلُّوهُ ئینجا دەست بەمل بیبەستن

(٣٠)

ثُمَّ پاشان

الْجَحِيمَ (هەر) لە دۆزەخ ی

صَلُّوهُ بخەن وبیسوتێنن

(٣١)

ثُمَّ پاشانیش

فِي لە

سِلْسِلَةٍ زنجیرێک ی

ذَرْعُهَا درێژی

سَبْعُونَ حەفتا

ذِرَاعًا باڵ ی

فَاسْلُكُوهُ ئەو ببەستنەوە

(٣٢)

إِنَّهُ كَانَ ئەو بێگومان پێشان

لَا يُؤْمِنُ ئیمانی نەدەهێنا

بِاللَّـهِ بەو خودایەی ی

الْعَظِيمِ (کە) زۆر زۆر مەزنە

(٣٣)

وَلَا يَحُضُّ و هانی (کەسی) نەدەدا

عَلَىٰ بۆ

طَعَامِ ناندانی

الْمِسْكِينِ هەژار

(٣٤)

فَلَيْسَ لَهُ جا ئەو نیەتی

الْيَوْمَ ئیمڕۆ

هَاهُنَا ئا لێرە

حَمِيمٌ (هیچ) دۆستێکی (گەرم وگوڕ)

(٣٥)

وَلَا و (نیەتی) هیچ

طَعَامٌ خواردنێک یش

إِلَّا مەگین

مِنْ لە

غِسْلِينٍ کێماو (ی زامداران)

(٣٦)

لَّا يَأْكُلُهُ (ئەو زاماوە) ناخوات

إِلَّا تەنها

الْخَاطِئُونَ ئەوانەی خەتاکارن

(٣٧)

فَلَا أُقْسِمُ ئیتر سوێند نەبێ

بِمَا بە ئەوەی

تُبْصِرُونَ ئێوە دەیبینن

(٣٨)

وَمَا و ئەوەیش کە

لَاتُبْصِرُونَ ئێوە نایبینن

(٣٩)

إِنَّهُ بەدڵنیایی ئەوە

لَقَوْلُ هەر گوفتاری

رَسُولٍ نێردراوێک و پەیامبەرێک ی

كَرِيمٍ زۆر بە ڕێزو پلەبەرزە

(٤٠)

وَمَا هُوَ و ئەوە -

بِقَوْلِ (بەهیچ جۆرێ) قسەی

شَاعِرٍ شاعیرێ (-نییە)

قَلِيلًا مَّا بە کەمی

تُؤْمِنُونَ بڕوا دەکەن

(٤١)

وَلَا و -

بِقَوْلِ (هیچ) قسەی

كَاهِنٍ فاڵگرەوەیەکیش (-نیە)

قَلِيلًا مَّا بە کەمی

تَذَكَّرُونَ یاد دەکەنەوە و پەند وەردەگرن

(٤٢)

تَنزِيلٌ دابەزێنراوێکە

مِّن لە

رَّبِّ خودای پەروەردگاری

الْعَالَمِينَ هەموو جیهانیان

(٤٣)

وَلَوْ ئەگەر بهاتایە

تَقَوَّلَ هەڵیبەستبا

عَلَيْنَا بەدەم ئێمەوە

بَعْضَ هەندێ

الْأَقَاوِيلِ قسەڵۆکان

(٤٤)

لَأَخَذْنَا هەر -

مِنْهُ لەوی مان (-دەبرد و ڕاپێچمان دەکرد)

بِالْيَمِينِ بە ڕاستەیەوە

(٤٥)

ثُمَّ پاشان

لَقَطَعْنَا هەر -

مِنْهُ لەوی

الْوَتِينَ شاخوێنبەرمان (-دەپچڕاند)

(٤٦)

فَمَا جا -

مِنكُم --

مِّنْ أَحَدٍ هیچ کەسێ (-نیە --لەئێوە)

عَنْهُ -

حَاجِزِينَ بەربەستەبن (-لێی)

(٤٧)

وَإِنَّهُ و ئەوەش بێگومان

لَتَذْكِرَةٌ هەر یادکەرەوە و ئامۆژگاریەکە

لِّلْمُتَّقِينَ بۆ لەخوا ترس و پارێزکاران

(٤٨)

وَإِنَّا و بە دڵنیایی ئێمە

لَنَعْلَمُ هەردەزانین

أَنَّ کە

مِنكُم لەئێوە هەن

مُّكَذِّبِينَ بەدرۆ دانەر و بڕوانەکەرانێک
