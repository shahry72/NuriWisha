أَوۡیان

أَرَاد-

شُكُورٗاسوپاسگوزاری (-ویست بی)

(٦٢)

وَعِبَادو بەندەکانی

ٱلرَّحۡمَٰنخودای بەبەزەیی و میهرەبان

ٱلَّذِینئەوانەی

يَمۡشُون-

عَلَىلەسەر

ٱلۡأَرۡضزەوی (-دەڕۆن)

هَوۡنٗابە ئارامی و هێمنی

وَإِذَاوئەگەر

خَاطَبَهُم-

ٱلۡجَٰهِلُونجاهل و نەفامەکان (-قسەی نەفامانەیان ئاڕاستەکردن)

قَالُوادەڵێن

سَلَٰمٗاسەلامو لێبێ (بەخراپ وەڵامیان نادەنەوە)

(٦٣)

وَ ٱلَّذِینئەوانەی

يَبِيتُونشەوبەسەردەبەن

لرَبِّهِملۆ پەروەردیگاریان

سُجَّدٗابە سوژدە بردن

وَقِيَٰمٗاو بەپێ وەستانەوە (لەنوێژا)

(٦٤)

وَ ٱلَّذِینئەوانەی

يَقُولُوندەڵێن

رَبَّنَائەی پەروەردیگاریمان

ٱصۡرِفۡدوورخەرەوە و لادە

عَنَّالێمان

عَذَابعەزاب و ئەشکەنجەیی

جَهَنَّمدۆزەخ

إِنّبەڕاستی

عَذَابَهَاعەزاب و ئەشکەنجەی ئەو

كَانبوو

غَرَامًابەردەوام و پێوەنوساوە

(٦٥)

إِنَّهَابەدڵنیایی ئەو

سَآءَتۡخراپترین

مُسۡتَقَرّٗاجێگای جیگیر کەوتن

وَمُقَامٗاو هەلسانەوە و مانەوەیە

(٦٦)

وَ ٱلَّذِینو ئەوانەی

إِذَآکاتێ

أَنفَقُوا(مەسرەف دەکەن)

لَمۡ يُسۡرِفُوائیسراف و زێدەڕەوی ناکەن

وَلَمۡ يَقۡتُرُواو بەخیلی و تەنگیش ناکەن

وَكَانبەلکو (مەسەرەفیان)-

بَيۡنلە نێوان

ذَٰلِكئەودا

قَوَامٗامام ناوەندییە

(٦٧)

وَ ٱلَّذِینئەوانەی

لَايَدۡعُونهاوارنەکەن و ناپەرستن

مَعلەگەڵ

ٱللَّهخودا

إِلَٰهًاپەرستراوێک

ءَاخَردیکە

وَلَايَقۡتُلُونو ناکوژن

ٱلنَّفۡسئەو نەفسە و کەسە

ٱلَّتِيئەوەی

حَرَّم-

ٱللَّهخودا (-قەدەغەی کردووە)

إِلَّائیللا

بِٱلۡحَقّبە حەق و ڕەوایی

وَلَايَزۡنُونو زینا ناکەن

وَمَنو هەرکەسێ

يَفۡعَلۡ-

ذَٰلِكئەوە (-بکات)

يَلۡقتوشی دێت و دوچاری

أَثَامٗاتاوانان (-دەبێت)

(٦٨)

يُضَٰعَفۡچەندجارە و چەند قات دەکرێ

لَهبۆئەو

ٱلۡعَذَابعەزاب و ئەشکەنجە

يَوۡمڕۆژی

ٱلۡقِيَٰمَةقیامەت (هەلسانەوە لەگۆڕەکان و بەپاوەستان لە مەیدانی مەحشەر)

وَيَخۡلُدۡوهەتاهەتا دەمێنێ

فِيهِۦتیایدا

مُهَانًاشەرمەزار و ڕیسواکراو

(٦٩)

إِلَّائیللا

مَنئەوەی

تَابپەشیمان بویەوە و گەڕایەوە

وَءَامَنو بڕوای هێنا

و عَمِلو کردی

عَمَلٗاکردەوەیەکی

صَٰلِحٗاچاک و باش

فَأُوْلَٰٓئِكئەوە ئەوان

يُبَدِّلدەگۆڕێ

ٱللَّهخودا

سَيِّ‍َٔاتِهِمۡخراپەکانیان

حَسَنَٰتٖ(بە) چاکان

وَكَانو هەمیشە -

ٱللَّهخودا

غَفُورٗائەوپەڕی لێبوردەی

رَّحِيمٗائەوپەڕی بەبەزەیی (-بووە)

(٧٠)

وَمَنو هەرکەسێ

تَابپەشیمان بویەوە و گەڕایەوە

و عَمِلو کردی

صَٰلِحٗاکاری چاک

فَإِنَّهُۥئیتر ئەوە

يَتُوبتۆبە دەکات و دەگەڕێتەوە

إِلَىبۆ ، بۆلای

ٱللَّهخودا

مَتَابٗاتەواوی گەڕانەوە و تۆبە

(٧١)

وَ ٱلَّذِینو ئەوانەی

لَايَشۡهَدُوننابنە ئامادە و شایەت لە

ٱلزُّورناهەقی و ناڕەوایی

وَإِذَاوکاتێ

مَرُّوادەڕۆن

بِٱللَّغۡوبەلای لەغوە و قسەی بەتاڵ و بێسود

مَرُّواتێدەپەڕن

كِرَامٗابەڕیز و بەهادارییەوە

(٧٢)

وَ ٱلَّذِینئەوانەی

إِذَاکاتێ

ذُكِّرُوایاد دەخرێنەوە و ئامۆژگاری دەکرێن

بِ‍َٔايَٰتبە ئایەتەکانی

رَبِّهِمپەروەردیگاریان

لَمۡ يَخِرُّواناکەونە

عَلَيۡهَاسەری

صُمّٗاکەڕ

وَعُمۡيَانٗاو کوێرانە

(٧٣)

وَ ٱلَّذِینو ئەوانەی

يَقُولُوندەڵێن

رَبَّنَائەی پەروەردیگاریمان

هَبۡببەخشە

لَنَاپێمان

مِنۡلە

أَزۡوَٰجناهاوسەرانمان

وَذُرِّيَّٰتِنَاو مناڵ و نەوەکانمان

قُرَّةئۆقرەی

أَعۡيُنٖچاوان

وَٱجۡعَلۡنَاو -

لِلۡمُتَّقِينبۆ لەخوداترس و پارێزکاران (-بمانکە)

إِمَامًاپێشەوایەک

(٧٤)

أُوْلَٰٓئِكئەوان

يُجۡزَوۡنپاداشت دەدرێنەوە و پێیان دەدرێ

ٱلۡغُرۡفَةماڵ و ژووری (تایبەت)

بِمَابەوەی

صَبَرُوادانیان بەخۆداگرت و ئارامیانگرت

وَيُلَقَّوۡنو پێشوازی دەکرێن

فِيهَاتیایدا

تَحِيَّةبە سڵاو

وَسَلَٰمًاو سەلام

(٧٥)

فِيهَاتیایدا
