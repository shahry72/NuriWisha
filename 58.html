وَلَا أَوْلَادُهُم و نە ڕۆڵەکانیشیان -سودنابەخشێ پێیان (ودەربازیان ناکات)

مِّنَ لە

اللَّـهِ خودا

شَيْئًا هیچ

أُولَـٰئِكَ ئەوانە

أَصْحَابُ برادەر و دۆستانی

النَّارِ ۖ ئاگر ن

هُمْ ئەوان

فِيهَا تیایدا

خَالِدُونَ ماوەن هەتا هەتایێ

(١٧)

يَوْمَ ئەوڕٶژەی

يَبْعَثُهُمُ -

اللَّـهُ خودا -زیندوویان دەکاتەوە و هەڵیان دەسێنتەوە

جَمِيعًا گشتیان

فَيَحْلِفُونَ جا سوێند دەخۆن

لَهُ بۆی

كَمَا هەرچۆن

يَحْلِفُونَ سوێن دەخۆن

لَكُمْ ۖ بۆتان

وَيَحْسَبُونَ و وادەزانن

أَنَّهُمْ کەوا ئەوان بەدڵنییایی

عَلَىٰ لەسەر

شَيْءٍ شتێکن (لە حەق)

أَلَا دەی بزانن

إِنَّهُمْ هُمُ بەدڵنیایی هەر ئەوانن

الْكَاذِبُونَ درۆزنەکان

(١٨)

اسْتَحْوَذَ عَلَيْهِمُ -

الشَّيْطَانُ شەیطان -دستی گرتووە بەسەریاندا

فَأَنسَاهُمْ جا لەبیری بردونەتەوە

ذِكْرَ یادی

اللَّـهِ خودا

أُولَـٰئِكَ ئەوانە

حِزْبُ تاقم و دەستەی

الشَّيْطَانِ شەیطان ن

أَلَا دە بزانن

إِنَّ کە بێگومان

حِزْبَ دەستەو تاقمی

الشَّيْطَانِ شەیطان

هُمُ هەر ئەوانن

الْخَاسِرُونَ زیانمەندان

(١٩)

إِنَّ بەڕاستی

الَّذِينَ ئەونەی

يُحَادُّونَ دوژمنداری و ڕکابەری دەکەن دژ بە-

اللَّـهَ خودا

وَرَسُولَهُ وپێغەمبەرەکەی

أُولَـٰئِكَ ئەوانە

فِي لە

الْأَذَلِّينَ نزمترین و ڕیسواترینی (خەلقن)

(٢٠)

كَتَبَ نوسیویەتی

اللَّـهُ خودا

لَأَغْلِبَنَّ هەر بەدڵنیایی و بێگومان سەردەکەوین

أَنَا خۆم

وَرُسُلِي وپەیامبەرانم

إِنَّ بەڕاستی

اللَّـهَ خودا

قَوِيٌّ بەهێز و توانای

عَزِيزٌ دەسەڵاتدر و باڵادەستە

(٢١)

لَّا تَجِدُ نابینی

قَوْمًا (هیچ) کۆمەڵێک

يُؤْمِنُونَ بڕوایان هەبێت

بِاللَّـهِ بە خودا

وَالْيَوْمِ وڕۆژی

الْآخِرِ دوایی

يُوَادُّونَ خۆشیان بوێت ودۆستداری بکەن لەگەڵ

مَنْ ئەوکەسەی

حَادَّ دوژمنایەتی و ڕکابەری-

اللَّـهَ خودا

وَرَسُولَهُ پێغەمبەرەکەی -کردوە

وَلَوْ هەرچەند

كَانُوا -

آبَاءَهُمْ باوکیان

أَوْ یان

أَبْنَاءَهُمْ كوڕیان

أَوْ یان

إِخْوَانَهُمْ برایان

أَوْ یان

عَشِيرَتَهُمْ تیرە وهۆزیان(-بووبن)

أُولَـٰئِكَ ئەوان

كَتَبَ (خودا) -

فِي لەناو

قُلُوبِهِمُ دڵیاندا -نوسیویەتی (وچەسپاندوویەتی)

الْإِيمَانَ ئیمان وباوەڕ

وَأَيَّدَهُم وپاڵپشتی کردوون

بِرُوحٍ بە گیانێک

مِّنْهُ ۖ لەلایەن خۆی

وَيُدْخِلُهُمْ ودەیانخاتە

جَنَّاتٍ باغ باغاتێک

تَجْرِي تێدەپەڕێت

مِن لە

تَحْتِهَا ژێریاندا

الْأَنْهَارُ ڕوبارەکان

خَالِدِينَ نەمرن و هەتاهەتا ماوەن

فِيهَا تیایدا

رَضِيَ -

اللَّـهُ خودا -ڕازیبووە

عَنْهُمْ لێیان

وَرَضُوا (ئەوانیش) ڕازینە

عَنْهُ لێی

أُولَـٰئِكَ ئەوانە

حِزْبُ دەستە و تاقم و

 لایەنگرانی

اللَّـهِ خودان

أَلَا ئەها بزانن کە

إِنَّ بەدڵنیایی

حِزْبَ دەستەو تاقم ولایەنگرانی

اللَّـهِ خودا

هُمُ هەر ئەوان

الْمُفْلِحُونَ سەرکەوتوو و سەرفرازن

(٢٢)
