يَاسِين یاسین

(١٣٠)

إِنَّا بەدڵنیایی ، بەراستی ئێمە

كَذَٰلِك ئاوا

نَجۡزِي پاداشت دەدەینەوە ، سزادەدەین

ٱلۡمُحۡسِنِين چاکەکارەکان

(١٣١)

إِنَّهُۥ بەدڵنیایی ئەوە

مِنۡ لە

عِبَادِنَا بەندەکانی ئێمە

ٱلۡمُؤۡمِنِين بڕوادارەکان

(١٣٢)

وَإِنّ و بەدڵنیایی

لُوطٗا پێغەمبەر لووط

لَّمِن هەر لە

ٱلۡمُرۡسَلِين نێردراو و پەیامبەرەکان

(١٣٣)

إِذۡ کە

نَجَّيۡنَٰه ئەومان دەربازکرد

وَأَهۡلَهُۥٓ لەگەڵ کەس و کارەکەی

أَجۡمَعِين گشتیان

(١٣٤)

إِلَّا جگەلە

عَجُوزٗا پیرێژنێک

فِي لە ، لەناو

ٱلۡغَٰبِرِين بەجێماوان

(١٣٥)

ثُمّ پاشان

دَمَّرۡنَا تەفروتونامانکرد

ٱلۡأٓخَرِين ئەوانەی دیکە

(١٣٦)

وَإِنَّكُمۡ و بەدڵنیایی ئێوە

لَتَمُرُّون هەر تێدەپەڕن

عَلَيۡهِم بەسەریاندا

مُّصۡبِحِين بەرەبەیانیان

(١٣٧)

وَبِٱلَّيۡل و بە شەوانیش

أَفَلَا تَعۡقِلُون ئایا تێناگەن و پێناگەن

(١٣٨)

وَإِنّ و بەدڵنیایی

يُونُس یوونس

لَمِن هەر لە

ٱلۡمُرۡسَلِين نێردراو و پەیامبەرانە

(١٣٩)

إِذۡ کاتێ کە

أَبَق هەڵات و ڕایکرد

إِلَى بۆ ، بۆلای

ٱلۡفُلۡك کەشتیە

ٱلۡمَشۡحُون بارکراوەکە

(١٤٠)

فَسَاهَم جا بەشدار بوو لە تیرو پشکەکە

فَكَان ئیتر-

مِن لە

ٱلۡمُدۡحَضِين دۆڕاوەکان(-بوو)

(١٤١)

فَٱلۡتَقَمَه ئەوەبوو قووتی دا

ٱلۡحُوت حووتەکە

وَهُو لەو کاتەی ئەو-

مُلِيمٞ گلەی لێکراو(-بوو)

(١٤٢)

فلَوۡلَآ ئەگەر -

أَنَّهُۥ بەڕاستی ئەو

كَان --

مِن لە

ٱلۡمُسَبِّحِين تەسبیح کاران -نە--بووا

(١٤٣)

لَلَبِث ئەوە بەدڵنیایی دەمایەوە

فِي لە ، لەناو

بَطۡنِهِۦٓ سکی

إِلَى هەتا

يَوۡم ئەو ڕۆژەی

يُبۡعَثُون زیندوودەکرانەوە

(١٤٤)

فَنَبَذۡنَٰه جا فڕێماندا

بِٱلۡعَرَآء لە چۆڵایی

وَهُو و ئەویش

سَقِيمٞ نەخۆش بوو وەک فڕوجێکی بێ پەڕ

(١٤٥)

وَأَنۢبَتۡنَا و ڕواندمان

عَلَيۡه بەسەری ، لەسەری

شَجَرَة دار و درەختێ

مِّن لە

يَقۡطِينٖ کودی زەرد

(١٤٦)

وَأَرۡسَلۡنَٰه و هەناردمان

إِلَى بۆ ، بۆلای

مِاْئَة سەد

أَلۡفٍ هەزارێ

أَوۡ یان

يَزِيدُون زیاتر دەبوون

(١٤٧)

فَ‍َٔامَنُوا ئیتر بڕوایان هینا

فَمَتَّعۡنَٰهُمۡ ئینجا ئێمە ئەوانمان ژیاند وڕایانبوێرد

إِلَى هەتا

حِينٖ ماوەیەک

(١٤٨)

فَٱسۡتَفۡتِهِمۡ جا پرسیاریان لێبکە

أَلِرَبِّك ئایا بۆ پەروەردیگارت هەیە

ٱلۡبَنَات کچان

وَلَهُم و بۆخۆشیان

ٱلۡبَنُون کوڕان؟!

(١٤٩)

أَمۡ یان

خَلَقۡنَا -

ٱلۡمَلَٰٓئِكَة فریشتەکانمان -دروستکردوە

إِنَٰثٗا مێ

وَهُمۡ و ئەوان یش

شَٰهِدُون ئامادە و شایەت ن

(١٥٠)

أَلَآ دەبزانن

إِنَّهُم بەدڵنیایی ئەوان

مِّنۡ لە

إِفۡكِهِمۡ درۆ و هەڵبەستراویان

لَيَقُولُون هەر دەڵێن

(١٥١)

وَلَد -

ٱللَّه خودا -مناڵی بوو

وَإِنَّهُمۡ و بەڕاستی ئەوان

لَكَٰذِبُون هەر درۆ دەکەن و درۆزنن

(١٥٢)

أَصۡطَفَى ئایا هەڵیبژاردووە؟!

ٱلۡبَنَات کچان

عَلَى بەسەر

ٱلۡبَنِين کوڕان

(١٥٣)

مَا لَكُمۡ ئەوەچیتانە؟

كَيۡف چۆن

تَحۡكُمُون بریار دەدەن

(١٥٤)

أَفَلَا تَذَكَّرُون ئایا بیرناکەنەوە و پەند وەرناگرن

(١٥٥)

أَمۡ یان

لَكُمۡ هەتانە

سُلۡطَٰنٞ بەلگە و دەسەڵاتێ

مُّبِينٞ دیار و ئاشکرا

(١٥٦)

فَأۡتُوا دەبێنن

بِكِتَٰبِكُمۡ نوسراوەکەتان

إِن ئەگەر

كُنتُمۡ ئیوە-

صَٰدِقِين ڕاستگۆ -بوون

(١٥٧)

وَجَعَلُوا دایاننا

بَيۡنَهُۥ نێوان ئەو

وَبَيۡن و نێوان

ٱلۡجِنَّة جنۆکان

نَسَبٗا پەیوەندی خێزانی

وَلَقَدۡ و بێگومان

عَلِمَت -

ٱلۡجِنَّة جنۆکەکان -زانیویان

إِنَّهُمۡ کە بەدڵنیایی ئەوان

لَمُحۡضَرُون هەر ئامادە دەکرێن

(١٥٨)

سُبۡحَٰن پاکی و بێگەردی بۆ

ٱللَّه خودا

عَمَّا لەوەی

يَصِفُون ئەوان وەسفی دەکەن

(١٥٩)

إِلَّا جگەلە

عِبَاد بەندەکانی

ٱللَّه خودا

ٱلۡمُخۡلَصِين پاککراو و رزگارکراوەکان

(١٦٠)

فَإِنَّكُمۡ بەڕاستی ئێوە

وَمَا و ئەوەی

تَعۡبُدُون ئێوە دەیپەرستن

(١٦١)

مَآ -

أَنتُمۡ ئێوە

عَلَيۡه لەسەری

بِفَٰتِنِين مەفتون و گومڕا -ناکەن

(١٦٢)

إِلَّا مەگەر

مَنۡ ئەوکەسەی

هُو خۆی

صَال دەچێت و دەگات بۆ

ٱلۡجَحِيم دۆزەخ

(١٦٣)

وَمَا و نیە

مِنَّآ لە ئێمەوە

إِلَّا ئیللا

لَهُۥ بۆی هەیە

مَقَامٞ جێی و پایەکی

مَّعۡلُومٞ زانراوی دیاریکراو

(١٦٤)

وَإِنَّا و بەدڵنیایی ئێمە

لَنَحۡن هەر خۆمانین

ٱلصَّآفُّون ڕێزگرتوەکان

(١٦٥)

وَإِنَّا و بەدڵنیایی

لَنَحۡن هەر ئێمەین

ٱلۡمُسَبِّحُون تەسبیح کارەکان

(١٦٦)

وَإِن و هەرچەند

كَانُوا ئەوان پێشان

لَيَقُولُون هەر دەیانگوت

(١٦٧)

لَوۡ ئەگەر

أَنّ بەڕاستی

عِندَنَا لامان هەبا

ذِكۡرٗا باس و خواسێک کیتابێک

مِّن لە

ٱلۡأَوَّلِين پێشینەکان

(١٦٨)

لَكُنَّا بەدڵنیایی ئێمە دەبووینە

عِبَاد بەندە ی

ٱللَّه خودا

ٱلۡمُخۡلَصِين ئەوانەی پاککراو و رزگارکراون

(١٦٩)

فَكَفَرُوا جا بڕوایان نەهێنا و کوفریان کرد

بِهِۦ پێی

فَسَوۡف ئیتر لەداهاتوو

يَعۡلَمُون دەزانن

(١٧٠)

وَلَقَدۡ و بەدڵنیایی

سَبَقَتۡ پێشتر-

كَلِمَتُنَا وتە و بەڵێنی ئێمە -دەڕچووبوو

لِعِبَادِنَا بۆ بەندەکان ئێمەی

ٱلۡمُرۡسَلِين نێردراو و پەیامبەرەکان

(١٧١)

إِنَّهُمۡ بەدڵنیایی ئەوان

لَهُم هەر خۆیان

ٱلۡمَنصُورُون سەرخراو و سەرکەوتوو ن

(١٧٢)

وَإِنّ و بێگومان

جُندَنَا سەربازەکانمان

لَهُم هەر خۆیانن

ٱلۡغَٰلِبُون سەرکەوتوون

(١٧٣)

فَتَوَلّ جا پشت هەلکە

عَنۡهُمۡ لێیان

حَتَّى تاکو

حِينٖ ماوەیەک

(١٧٤)

وَأَبۡصِرۡهُمۡ و تەماشایان بکە و بیانبینە

فَسَوۡف جا لەداهاتوو

يُبۡصِرُون ئەوان دەبینن

(١٧٥)

أَفَبِعَذَابِنَا ئایا لە هاتنی عەزاب و ئەشکەنجەدانی ئێمە

يَسۡتَعۡجِلُون ئەوان پەلەپەل دەکەن ودەڵێن با بێت

(١٧٦)

فَإِذَا جاکاتێ

نَزَل دابەزی

بِسَاحَتِهِمۡ لەگۆڕەپانی ئەوان

فَسَآء جا یەکجار خراپ بوو

صَبَاح بەربەیانی

ٱلۡمُنذَرِين ترسێنراوان و بێدارکراوەکان

(١٧٧)

وَتَوَلّ و پشت هەڵکە

عَنۡهُمۡ لێیان

حَتَّى تاکو

حِينٖ ماوەیەک

(١٧٨)

وَأَبۡصِرۡ و ببینە و تەماشا بکە

فَسَوۡف ئەوان لەداهاتوو

يُبۡصِرُون دەبینن

(١٧٩)

سُبۡحَٰن پاکی و بێگەردی بۆ

رَبِّك پەروەردیگارت

رَبّ پەروەردیگار و خاوەنی

ٱلۡعِزَّة دەسەڵاتی باڵا ولەبن نەهاتوو

عَمَّا لەوەی

يَصِفُون ئەوان وەسفی دەکەن

(١٨٠)

وَسَلَٰم و سەلام

عَلَى لەسەر

ٱلۡمُرۡسَلِين نێردراو و پەیامبەرەکان

(١٨١)

وَٱلۡحَمۡد و هەموو سوپاس و ستایشێ

لِلَّه هەر بۆ خودای

رَبّ پەروەردیگار و خاوەنی

ٱلۡعَٰلَمِين جیهانیان

(١٨٢)
