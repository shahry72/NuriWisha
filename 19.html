سورة مَرْيَمَ  


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی بەبەزەیی و میهرەبان (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


كهيعص کاف ها یا عەین صاد

(١)

ذِكْرُ باسی

رَحْمَتِ بەزەیی و میهرەبانی

رَبِّكَ پەروەردیگات

عَبْدَهُ (بۆ) بەندەی خۆی

زَكَرِيَّا زەکەرییا

(٢)

إِذْ کاتێک

نَادَىٰ هاواری کردە

رَبَّهُ پەروەردگاری خۆی

نِدَاءً هاوارێکی

خَفِيًّا كز و شاراوە

(٣)

قَالَ گوتی

رَبِّ پەروەردیگارە

إِنِّي بەڕاستی من

وَهَنَ بێهێز بوە

الْعَظْمُ هەرچی ئێسکە

مِنِّي لێم

وَاشْتَعَلَ وەداگیرساوە

الرَّأْسُ سەر (م)

شَيْبًا بە موی سپی

وَلَمْ أَكُن -وقەت وانەبووە

بِدُعَائِكَ بە پاڕانەوەم لێت

رَبِّ ئەی پەروەردیگارم

شَقِيًّا پەرێشانی نا موراد -بووبم

(٤)

وَإِنِّي وەمن بەڕاستی

خِفْتُ ترسام (لە)

الْمَوَالِيَ کەس وکار (ەکانم)

مِن لە

وَرَائِي دوای خۆم

وَكَانَتِ -

امْرَأَتِي وەژنەکەم

عَاقِرًا نەزۆک (-بوو)

فَهَبْ جا ببەخشە

لِي پێم

مِن لە

لَّدُنكَ لایەن خۆتەوە

وَلِيًّا دۆست ونزیکێک

(٥)

يَرِثُنِي میراتگرم بێت

وَيَرِثُ ومیراتگری بکات

مِنْ لە

آلِ بنەماڵە (و شوێنکەوتەی)

يَعْقُوبَ یەعقوب ۖ

وَاجْعَلْهُ -

رَبِّ و پەروەردیگارم -بیکەیتە

رَضِيًّا (کەسێکی لێ) ڕازیبوو

(٦)

يَا ئەی

زَكَرِيَّا زەکەرییا

إِنَّا بەدڵنیایی ئێمە

نُبَشِّرُكَ مزگێنیت پێ دەدەین

بِغُلَامٍ بە كوڕێک

اسْمُهُ ناوی

يَحْيَىٰ یەحیایە

لَمْ نَجْعَل دامان نەناوە

لَّهُ بۆی

مِن لە

قَبْلُ پێشدا

سَمِيًّا (هیچ) ناولێنراوێک (بەوناوە)

(٧)

قَالَ گوتی

رَبِّ پەروەردیگارە

أَنَّىٰ چۆن

يَكُونُ دەبێت

لِي بۆم

غُلَامٌ كوڕێک

وَ (لەکاتێ)

كَانَتِ -

امْرَأَتِي ژنەکەم

عَاقِرًا نەزۆک (-بوو)

وَقَدْ و وا (من)

بَلَغْتُ -

مِنَ لە

الْكِبَرِ گەورەیی و تەمن زۆری -گەیشتوومەتە

عِتِيًّا ئەوپەڕی

(٨)

قَالَ گوتی

كَذَٰلِكَ ئاوها

قَالَ -

رَبُّكَ پەروەردیگارت -گوتی

هُوَ ئەو (کارە)

عَلَيَّ هَيِّنٌ لام سوك و ئاسانە

وَقَدْ و بێگومان وبە تەحقیق

خَلَقْتُكَ تۆم خولقاندوە

مِن لە

قَبْلُ پێشدا

وَلَمْ تَكُ -

شَيْئًا و(هیچ)یش -نەبوویت

(٩)

قَالَ گوتی

رَبِّ پەروەردیگارە

اجْعَل دانێ

لِّي بۆم

آيَةً بەڵگە ونیشانەیەکی دیار ۚ

قَالَ گوتی

آيَتُكَ بەڵگە ونیشانەکەت (ئەوەیە)

أَلَّا تُكَلِّمَ کە نادوێنی

النَّاسَ خەڵک

ثَلَاثَ سێ

لَيَالٍ شەوان

سَوِيًّا بە تەواوی

(١٠)

فَخَرَجَ ئەوسا دەرچوو

عَلَىٰ بۆ سەر

قَوْمِهِ نەتەوەکەی

مِنَ لە

الْمِحْرَابِ میحرابەکە

فَأَوْحَىٰ ویەکسەر ئیشارەتی کرد

إِلَيْهِمْ بۆیان

أَن کە

سَبِّحُوا تەسبیحات بکەن (خودا بە پاك و بەرزی ڕاگرن)

بُكْرَةً بە بەیانیان

وَعَشِيًّا وئێواران

(١١)

يَا ئەی

يَحْيَىٰ یەحیا

خُذِ -

الْكِتَابَ کیتابەکە

بِقُوَّةٍ ۖ بەتوندی -وەرگرە

وَآتَيْنَاهُ و پێماندا

الْحُكْمَ حوكم وداوەری

صَبِيًّا لە مناڵیەوە

(١٢)

وَحَنَانًا وەسۆز و میهرەبانیەک

مِّن لە

لَّدُنَّا لایەن خۆمانەوە

وَزَكَاةً ۖ و پاکیەک

وَكَانَ -وە

تَقِيًّا زۆر لەخوا ترس و پارێزکار (-بوو)

(١٣)

وَبَرًّا وەزۆر چاك بوو

بِوَالِدَيْهِ بۆهەردوو باوانی

وَلَمْ يَكُن و -

جَبَّارًا خۆبە زلزانی ناچارکار

عَصِيًّا لاسار وسەرپێچیکار -نەبوو

(١٤)

وَسَلَامٌ وەسەلامێکی (خودای)

عَلَيْهِ لەسەرە

يَوْمَ ڕۆژێ

وُلِدَ لەدایکبووە

وَيَوْمَ وەڕۆژێ

يَمُوتُ دەمرێت

وَيَوْمَ وەڕۆژێ

يُبْعَثُ هەڵدەسێنرێتەوە

حَيًّا بەزیندوویی (زیندوو دەکرێتەوە)

(١٥)

وَاذْكُرْ و باس بکە

فِي لە ناو

الْكِتَابِ ئەوکیتابە (قورئان)

مَرْيَمَ (باسی) مەریەم

إِذِ کاتێ

انتَبَذَتْ -

مِنْ لە

أَهْلِهَا کەسوکاری

مَكَانًا لەشوێنێکی

شَرْقِيًّا ڕۆژهەڵاتی -دوورگیربوو (کەنارگیربوو)

(١٦)

فَاتَّخَذَتْ ئینجا داینا

مِن دُونِهِمْ بێجگە لەوان (نێوان خۆی و ئەوان)

حِجَابًا پەردە ونێوانگرێک

فَأَرْسَلْنَا ئەوسا ناردمان

إِلَيْهَا بۆی

رُوحَنَا (جبریل)

فَتَمَثَّلَ -

لَهَا (ئەویش) بۆی -هاتەسەر شێوەی

بَشَرًا مرۆڤێکی

سَوِيًّا ڕێک وتەواو

(١٧)

قَالَتْ (مەریەم) گوتی

إِنِّي بەدڵنیایی من

أَعُوذُ پەنادەخوازم

بِالرَّحْمَـٰنِ بەخودای میهرەبان

مِنكَ لەتۆ

إِن ئەگەرهاتوو

كُنتَ -

تَقِيًّا زۆرلەخوداترس وپارێزکار -بیت

(١٨)

قَالَ گوتی

إِنَّمَا هەر ئەوەندە

أَنَا من

رَسُولُ پەیامبەرێکی

رَبِّكِ پەروەردیگارتم

لِأَهَبَ تاكو ببەخشم

لَكِ پێت

غُلَامًا كوڕێکی

زَكِيًّا پاك

(١٩)

قَالَتْ گوتی

أَنَّىٰ چۆن

يَكُونُ دەبێت

لِي بۆم

غُلَامٌ كوڕێک

وَلَمْ يَمْسَسْنِي لەکاتێک لێک کەوتن ڕوینەداوە لە نێوان من و

بَشَرٌ (هیچ) مرۆڤێک

وَلَمْ أَكُ و -

بَغِيًّا داواکاری داوێن پیس -نەبوومە

(٢٠)

قَالَ گوتی

كَذَٰلِكِ ئاوای

قَالَ گوت

رَبُّكِ پەروەردیگارت

هُوَ ئەو (کارە)

عَلَيَّ لەسەرم

هَيِّنٌ ۖ زۆر ئاسانە

وَلِنَجْعَلَهُ وتاكوو بیکەینە

آيَةً بەڵگەیەکی گەورەی زۆر دیار

لِّلنَّاسِ بۆخەڵک

وَرَحْمَةً و رەحمەتێکیش

مِّنَّا ۚ لەئێمەوە وە

وَكَانَ -

أَمْرًا وەکارێکی

مَّقْضِيًّا بڕاوە و کۆتاهاتوو (-بوو)

(٢١)

فَحَمَلَتْهُ ئەوەبوو هەڵیگرت (سکی بوو)

فَانتَبَذَتْ بِهِ جا بەدووری گرت

مَكَانًا لەشوێنێکی

قَصِيًّا دوور

(٢٢)

فَأَجَاءَهَا ئەوسا -

الْمَخَاضُ ژانی لەدایک بوونەکە (-مەریەمی) هێنا

إِلَىٰ بۆلای

جِذْعِ قەدی

النَّخْلَةِ دارخورماکە

قَالَتْ گوتی

يَا ئەی

لَيْتَنِي خۆزگەم

مِتُّ مرد بام

قَبْلَ پێش

هَـٰذَا ئەمە

وَكُنتُ -

نَسْيًا و شتێکی کەم بایەخی (کە ئاسان لەبیردەکرێت)

مَّنسِيًّا لەبیرکراو -بووام

(٢٣)

فَنَادَاهَا ئەوسا -

مِن لە

تَحْتِهَا ژێریەو -بانگی کرد

جَعَلَ -

سَرِيًّا جۆگەلەیەک

النَّخْلَةِ دارخورماکە

تُسَاقِطْ دادەکەوێنی

عَلَيْكِ بەسەرتدا

رُطَبًا روطابێکی (خورمایەکێ)

جَنِيًّا تازەلێکراوە

وَاشْرَبِي و بخۆرەوە

فَإِمَّا جائەگەر

تَرَيِنَّ بینیت

مِنَ لە

أَحَدًا هەر کەسێک

فَقُولِي ئەوە پێی بڵێ

مِّنكُمْ لەنگۆ (کەسێک)

إِلَّا ئیللا

وَارِدُهَا ۚ دێت بەسەریدا

كَانَ عَلَىٰ لەسەر

رَبِّكَ پەروەردیگارت

حَتْمًا واجبێکی

مَّقْضِيًّا بڕاوەیە ودێتەجێ

(٧١)

ثُمَّ پاشان

نُنَجِّي بەدڵنیایی دەرباز دەکەین

الَّذِينَ ئەوانەی

اتَّقَوا لەخواترس و پارێزکاربوون

وَّنَذَرُ وبەجێ دەهێلین

الظَّالِمِينَ ستەمکارەکان

فِيهَا تیاییدا

جِثِيًّا لەسەرچۆکان

(٧٢)

وَإِذَا و کاتێ

تُتْلَىٰ دەخوێندرێتەوە

عَلَيْهِمْ بەسەریاندا

آيَاتُنَا ئایاتەکانمان

بَيِّنَاتٍ ڕۆشن و دیارن

قَالَ -

الَّذِينَ ئەوانەی

كَفَرُوا کافربوون

لِلَّذِينَ بەوانەی

آمَنُوا بڕوایان هێناو -دەڵێن

أَيُّ کامەی

الْفَرِيقَيْنِ دوو کۆمەڵ و تاقمەکە

خَيْرٌ چاكترن

مَّقَامًا لەجێ و ڕێی ژیاندا

وَأَحْسَنُ وباشترن

نَدِيًّا لە دانیشتگایان

(٧٣)

وَكَمْ و چەند مان

أَهْلَكْنَا لەناوبرد

قَبْلَهُم پێش ئەوان

مِّن لە

قَرْنٍ (میللەتان)

هُمْ ئەوان

أَحْسَنُ باشتربوون

أَثَاثًا لە کەل و پەلی ماڵ ودارایی

وَرِئْيًا و ڕەنگ وڕوخساری بینراو

(٧٤)

قُلْ بڵێ

مَن کێ

كَانَ -

فِي لەناو

الضَّلَالَةِ گومڕایی -دابوو

فَلْيَمْدُدْ لَهُ با هەر درێژکاتەوە و پێی بدات

الرَّحْمَـٰنُ خودای میهرەبان

مَدًّا ۚ درێزكرنەوە و پێدانێک

حَتَّىٰ هەتاكوو

إِذَا کە

رَأَوْا بینیان

مَا ئەوەی

يُوعَدُونَ بەڵێنیان پێدەدرێ

إِمَّا جا چ

الْعَذَابَ عەزاب وئەشکەنجە

وَإِمَّا و یا

السَّاعَةَ (ڕۆژی قیامەت)

فَسَيَعْلَمُونَ ئینجا بەڕاستی دەزانن

مَنْ هُوَ کێ یە

شَرٌّ خراپترە

مَّكَانًا شوێن و جێگای

وَأَضْعَفُ و بێهێزترە

جُندًا سوپاو لەشكری

(٧٥)

وَيَزِيدُ و زیاد دەکات

اللَّـهُ خودا بۆ

الَّذِينَ ئەوانەی

اهْتَدَوْا ڕێنمونی وهیدایەتیان وەرگرتووە

هُدًى ۗ هیدایەت وڕێنموونی (زیاتر لەوەی پێشو)

وَالْبَاقِيَاتُ و ئەوانەی ماونەتەوە لە

الصَّالِحَاتُ چاکەکان

خَيْرٌ باشترە

عِندَ لەلای

رَبِّكَ پەروەردیگارت

ثَوَابًا لە پاداشت

وَخَيْرٌ و باشترە

مَّرَدًّا لە گەڕانەوە وسەرئەنجامدا

(٧٦)

أَفَرَأَيْتَ جا ئایا دیوتە

الَّذِي ئەوەی

كَفَرَ بڕاوای نەکردوە و كوفری کردوە

بِآيَاتِنَا بە ئایات وبەڵگەدیارەکانمان

وَقَالَ و ووتوویەتی

لَأُوتَيَنَّ بەدڵنیایی (سوێند بێ) هەر پێمدەدرێ

مَالًا ماڵ و سامانێک

وَوَلَدًا و كوڕێک (ی زۆر)

(٧٧)

أَطَّلَعَ ئایا سەیری -

الْغَيْبَ غەیب و پەنهانی -کردوە (زانیویەتی)

أَمِ یان

اتَّخَذَ وەریگرتووە

عِندَ لە لای

الرَّحْمَـٰنِ خودای میهرەبانەوە

عَهْدًا بەڵێنێک

(٧٨)

كَلَّا ۚ نا نەخێر

سَنَكْتُبُ بەدڵنیایی دەنوسین

مَا ئەوەی

يَقُولُ دەیڵێت

وَنَمُدُّ ودرێژدەکەینەوە

لَهُ بۆی

مِنَ لە

الْعَذَابِ عەزاب وئەشکەنجە

مَدًّا درێژکردنەوەیەک

(٧٩)

وَنَرِثُهُ وبۆی دەهێلینەوە

مَا ئەوەی

يَقُولُ دەیڵێت

وَيَأْتِينَا وبۆمان دێت

فَرْدًا بەتەنیایی

(٨٠)

وَاتَّخَذُوا و دایاننا

مِن لە

دُونِ غەیری

اللَّـهِ خودا (الله)

آلِهَةً چەندین پەرستراو و خودای درۆین

لِّيَكُونُوا تاكوو -

لَهُمْ بۆیان -ببێتە

عِزًّا پشت وپەنا

(٨١)

كَلَّا ۚ نا نەخێر

سَيَكْفُرُونَ بەدڵنیایی دوایی ئینکاری وكوفر دەکەن

بِعِبَادَتِهِمْ بە پەرستنیان

وَيَكُونُونَ -

عَلَيْهِمْ وبەسەریاندا -دەبنە

ضِدًّا دژ و بەرپەرچ دەرەوە

(٨٢)

أَلَمْ تَرَ ئایا نەتدیوە و نەتزانیوە

أَنَّا کە بەدڵنیایی ئێمە

أَرْسَلْنَا ناردوومانە

الشَّيَاطِينَ شەیطانەکان

عَلَى بۆ سەر

الْكَافِرِينَ کافر و یەکخوانەپەرستان

تَؤُزُّهُمْ دەیان ورژێنن و دەیانبزوێنن

أَزًّا (چۆن) بزواندنێک

(٨٣)

فَلَا تَعْجَلْ جا پەلە مەکە

عَلَيْهِمْ ۖ لێیان

إِنَّمَا هەر ئەوەندەیە

نَعُدُّ ئێمە ژماردە دەکەین

لَهُمْ بۆیان

عَدًّا ژماردنێک

(٨٤)

يَوْمَ ڕۆژێ

نَحْشُرُ کۆدەکەینەوە

الْمُتَّقِينَ لەخواترس وپارێزکارەکان

إِلَى بۆلای

الرَّحْمَـٰنِ خودای میهرەبان

وَفْدًا بە کۆمەڵی ڕێزلێنراو

(٨٥)

وَنَسُوقُ ودەهاژۆین و وەڕێدێخین

الْمُجْرِمِينَ تاوانکارەکان

إِلَىٰ بەرەو

جَهَنَّمَ دۆزەخ

وِرْدًا بە تینوویی (دێن)

(٨٦)

لَّايَمْلِكُونَ نیەتیان

الشَّفَاعَةَ (هیچ) تکاکاریەک

إِلَّا مەگەر

مَنِ کەسێک

اتَّخَذَ -

عِندَ لەلای

الرَّحْمَـٰنِ خودای میهرەبانەوە

عَهْدًا بەڵێن و پەیمانێکی -وەریگرتبێت

(٨٧)

وَقَالُوا وەگوتیان

اتَّخَذَ -

الرَّحْمَـٰنُ خودای میهرەبان

وَلَدًا ڕۆڵەیەکی -داناوە بۆخۆی

(٨٨)

لَّقَدْ سوێند بەخودا

جِئْتُمْ هێناوتانە

شَيْئًا شتێکی

إِدًّا زۆر نابەجێ

(٨٩)

تَكَادُ خەریکە

السَّمَاوَاتُ ئاسمانەکان

يَتَفَطَّرْنَ دەرزدەرز دەبن

مِنْهُ لێیەوە

وَتَنشَقُّ ولەت لەت دەبێت

الْأَرْضُ زەوی

وَتَخِرُّ وخڕەی دێت ودەکەوێتەخوار

الْجِبَالُ چیاکان

هَدًّا لە داڕماندا

(٩٠)

أَن (بەهۆی) ئەوەی

دَعَوْا بانگیانکرد و دایاننا

لِلرَّحْمَـٰنِ بۆخودای میهرەبان

وَلَدًا ڕۆڵەیەک

(٩١)

وَمَا يَنبَغِي و ناگونجێ و ناشێت

لِلرَّحْمَـٰنِ بۆ خودای میهرەبان

أَن کە

يَتَّخِذَ دانێت (بۆ خۆی)

وَلَدًا ڕۆڵەیەک

(٩٢)

إِن -

كُلُّ هەرچی

مَن ئەوی

فِي لەناو

السَّمَاوَاتِ ئاسمانەکان

وَالْأَرْضِ و زەوی دان

إِلَّا ئیللا

آتِي هاتووە بۆ لای

الرَّحْمَـٰنِ خودای میهرەبان

عَبْدًا بەبەندەتی

(٩٣)

لَّقَدْ بەدڵنیایی و بێگومان

أَحْصَاهُمْ سەرژمێری کردوون

وَعَدَّهُمْ و ژماردوونی

عَدًّا ژماردنێک

الْقِيَامَةِ قیامەت

فَرْدًا بە تاك تاکی

إِنَّ بەدڵنیایی

الَّذِينَ ئەوانەی

الصَّالِحَاتِ چاکەکان
