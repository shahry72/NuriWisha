مَّحِيصٖ پەنایێک

(٤٨)

لَّا -

يَسۡ‍َٔم بێزار (-نا) بێ

ٱلۡإِنسَٰن مرۆڤ

مِن لە

دُعَآء دوعا و پاڕانەوەی

ٱلۡخَيۡر خێر و چاک

وَإِن و ئەگەر

مَّسَّه پێی بکەوێ

ٱلشَّرّ خراپە و شەڕ

فَيَ‍ُٔوسٞ ئەوە بێ هیوا و

قَنُوطٞ بێ ئومیدکەوتووە

(٤٩)

وَلَئِنۡ و ئەگەر

أَذَقۡنَٰه پێی بچێژین

رَحۡمَة بەزەیی و میهرەبانیێ

مِّنَّا لە لایەن خۆمانەوە

مِنۢ لە

بَعۡد دوای

ضَرَّآء زیان و زەرەرێک

مَسَّتۡه پێی کەوتبێ

لَيَقُولَنّ ئەوە بەڵنیایی دەڵێ

هَٰذَا ئەوە

لِي بۆ منە (لەخۆمەوەیە)

وَمَآ و-

أَظُنّ گومان (-نا) کەم

ٱلسَّاعَة ڕۆژی قیامەت

قَآئِمَة هەڵساو بێت

وَلَئِن و ئەگەر

رُّجِعۡت بگێردرێمەوە

إِلَى بۆ ، بۆلای

رَبِّيٓ پەروەردیگارم

إِنّ بەدڵنیایی

لِي بۆمن (دەبێ)

عِندَهُۥ لای ئەو

لَلۡحُسۡنَى هەرە چاکترین شت (بەهەشت)

فَلَنُنَبِّئَنّ بەدڵنیایی هەواڵ دەدەینە

ٱلَّذِین ئەوانەی

كَفَرُوا کوفریانکرد و بێ بڕوابوون

بِمَا بەوەی

عَمِلُوا کردوویانە

وَلَنُذِيقَنَّهُم و بێگومان پێیان دەچیژین

مِّنۡ لە

عَذَابٍ عەزاب وئەشکەنجەیەکی

غَلِيظٖ زبر و تووند

(٥٠)

وَإِذَا وئەگەر

أَنۡعَمۡنَا نیعمەتمان باراندە

عَلَى سەر

ٱلۡإِنسَٰن مرۆڤ

أَعۡرَض پشت هەڵدەکا

وَنَ‍َٔا بِجَانِبِهِۦ و شانی بادەدا

وَإِذَا وئەگەر

مَسَّه پێی بکەوێ

ٱلشَّرّ خراپە و شەڕێ

فَذُو ئەوە خاوەن

دُعَآءٍ دوعا و پاڕانەوەیەکی

عَرِيضٖ پان و درێژە

(٥١)

قُلۡ بڵێ

أَرَءَيۡتُمۡ بینیوتانە و تێڕاماون

إِن ئەگەر

كَان -

مِنۡ لە

عِند لای

ٱللَّه خودا (-بێت)

ثُمّ ئینجا

كَفَرۡتُم کوفرتان کرد

بِهِۦ پێی

مَنۡ و کێ

أَضَلّ وێڵ و ڕێگای بزرترە

مِمَّنۡ لەوەی

هُو (بەڕاستی)

فِي لە ، لەناو

شِقَاقِۢ دژایەتی و دووبەرەکی

بَعِيدٖ زۆر دوور

(٥٢)

سَنُرِيهِمۡ لەداهاتوو نیشانیان دەدەین

ءَايَٰتِنَا ئایەتەکانمان (بەلگە و نیشانە و موعجیزات)

فِي لە ، لەناو

ٱلۡأٓفَاق ئاسۆیەکان

وَفِيٓ و لە

أَنفُسِهِمۡ نەفس و زاتی خۆیان

حَتَّى تاکو

يَتَبَيَّن ڕۆن دەبێتەوە

لَهُمۡ بۆیان

أَنَّه کەوا بەڕاستی

ٱلۡحَقّ هەق و ڕاستیە

أَو ئایا

لَمۡ يَكۡف بەس نەبووە

بِرَبِّك پەروەردیگارت

أَنَّهُۥ بەدڵنیایی ئەو

عَلَى لەسەر

كُلّ هەموو

شَيۡءٖ شتێک

شَهِيد شایەتە

(٥٣)

أَلَآ دەبزانن

إِنَّهُمۡ بەدڵنیایی ئەوان

فِي لەناو

مِرۡيَةٖ ڕاڕایی و گومانن

مِّن لە

لِّقَآء پێگەیشتن بە

رَبِّهِم پەروەردگاریان

أَلَآ دەبزانن

إِنَّهُۥ بەدڵنیایی ئەو

بِكُلّ بە هەموو

شَيۡءٖ شتێک

مُّحِيطُۢ تەواو ئاگادارە و دەوری داوە

(٥٤)
