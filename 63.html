بۆ زانینی مانای وشەیەک پەنجە بنێ سەری ، بۆ مانای ئایەتەکەش پەنجە بنێ بە ژمارەی ئایەتەکە

سُورَةُ الْمُنَافِقُونَ 


بِسۡم بەناوی

ٱللَّه خودا

ٱلرَّحۡمَٰن ئەوپەڕی میهرەبان وبە بەزەیی (ناوێکی پیرۆزی خودایە)

ٱلرَّحِيم ئەوپەڕی بەخشندە (ناوێکی پیرۆزی خودایە)


إِذَا ئەگەر

جَاءَكَ هاتنە لات

الْمُنَافِقُونَ مونافیق و دووڕووەکان

قَالُوا گوتیان

نَشْهَدُ شایەتی دەدەین

إِنَّكَ بەڕاستی تۆ

لَرَسُولُ گومان نیە پێغەمبەری

اللَّـهِ 

ۗ خودای ،

وَاللَّـهُ و خوداش

يَعْلَمُ دەزانێت

إِنَّكَ بێگومان تۆ

لَرَسُولُهُ بەڕاستی پێغەمبەری ئەوی

وَاللَّـهُ و خوداش

يَشْهَدُ شایەتی دەدات

إِنَّ کە

الْمُنَافِقِينَ مونافیق و دووڕوەکان

لَكَاذِبُونَ بەدڵنیایی درۆزنن

(١)

اتَّخَذُوا -

أَيْمَانَهُمْ سوێندەکانیان (-کردۆتە)

جُنَّةً قەلغانێک

فَصَدُّوا جا ڕێگریان کردوە و لەمپەریان داناوە

عَن لە

سَبِيلِ ڕێی

اللَّـهِ خودا

إِنَّهُمْ بەڕاستی ئەوان

سَاءَ خراپترن شتە

مَا ئەوەی

كَانُوا 

يَعْمَلُونَ دەیانکرد

(٢)

ذَٰلِكَ ئەوە

بِأَنَّهُمْ بەهۆی ئەوەی ئەوان

آمَنُوا بڕوایان هێنا

ثُمَّ پاشان

كَفَرُوا کوفریان کرد

فَطُبِعَ ئینجا مۆری نا

عَلَىٰ بەسەر

قُلُوبِهِمْ دڵەکانیان

فَهُمْ ئیتر ئەوان

لَا -

يَفْقَهُونَ باش تێناگەن

(٣)

وَإِذَا و ئەگەر

رَأَيْتَهُمْ بیان بینی

تُعْجِبُكَ سەرسامت دەکات

أَجْسَامُهُمْ لەش و قەڵافەتەکانیان

وَإِن و ئەگەر

يَقُولُوا (بڵێن)

تَسْمَعْ گوێ دەگری

لِقَوْلِهِمْ بۆ قسەکانیان

كَأَنَّهُمْ ئەوان هەروەک

خُشُبٌ چەندین تەختەداری

مُّسَنَّدَةٌ هەڵپەسێردراون

يَحْسَبُونَ وا دەزانن

كُلَّ هەرچی

صَيْحَةٍ بانگ و هاوارێکە

عَلَيْهِمْ بۆ سەر ئەوانە

هُمُ ئا ئەوان

الْعَدُوُّ دوژمنی تەواون

فَاحْذَرْهُمْ جا وریایان بە

قَاتَلَهُمُ -

اللَّـهُ خودا -بیانکوژێت

أَنَّىٰ لە هەرکوێی (چۆن)

يُؤْفَكُونَ وەردەگێڕدرێن و لادەدەن (لە حەق)

(٤)

وَإِذَا و کاتێ

قِيلَ دەوترێ

لَهُمْ پێیان

تَعَالَوْا وەرن

يَسْتَغْفِرْ داوای لێخۆشبوون بکات

لَكُمْ بۆتان

رَسُولُ پێغەمبەری

اللَّـهِ خودا

لَوَّوْا بادەدەن

رُءُوسَهُمْ سەرەکانیان

وَرَأَيْتَهُمْ ودەیانبینی

يَصُدُّونَ ڕێگری و بەرهەڵستی دەکەن

وَهُم لەگەڵ ئەوەی خۆیان

مُّسْتَكْبِرُونَ بەگەورەدەزانن و لووتبەرزن

(٥)

سَوَاءٌ وەک یەکە

عَلَيْهِمْ بۆیان

أَسْتَغْفَرْتَ داوای لێخۆش بوونت کردبێ

لَهُمْ بۆیان

أَمْ یان

لَمْ 

تَسْتَغْفِرْ داوای لێخۆشبوون نەکردبێ

لَهُمْ بۆیان

لَن 

يَغْفِرَ (ئەوە)هەرگیز خۆشنابێت

اللَّـهُ خودا

لَهُمْ لێیان

إِنَّ بەدڵنیایی

اللَّـهَ خودا

لَا -

يَهْدِي هیدایەت و ڕاستەڕێ ناکات

الْقَوْمَ ئەو قەومەی

الْفَاسِقِينَ فاسق و لە سنوردەرچوون

(٦)

هُمُ 

الَّذِينَ ئەوە ئەوانن

يَقُولُونَ دەڵێن

لَا -

تُنفِقُوا مەبەخشن و مەسرەف مەکەن

عَلَىٰ لەسەر

مَنْ (ئەوانەی)

عِندَ لای

رَسُولِ پێغەمبەری

اللَّـهِ خودان

حَتَّىٰ هەتا

يَنفَضُّوا 

ۗ واز بهێنن و چۆڵێ بکەن

وَلِلَّـهِ و هەر هی خودایە

خَزَائِنُ خەزینەکانی

السَّمَاوَاتِ ئاسمانەکان

وَالْأَرْضِ و زەوی

وَلَـٰكِنَّ بەڵام

الْمُنَافِقِينَ مونافیق ودووڕووەکان

لَا -

يَفْقَهُونَ باش حاڵی نابن

(٧)

يَقُولُونَ دەڵێن

لَئِن ئەگەر هاتوو

رَّجَعْنَا گەڕاینەوە

إِلَى بۆ

الْمَدِينَةِ شاری مەدینە

لَيُخْرِجَنَّ -

الْأَعَزُّ خاوەنی دەسەڵات و بەعیزەتتر -بێگومان دەرداکات

مِنْهَا لێی

الْأَذَلَّ ئەوەی بێدەسەڵات و زەلیل ترە

وَلِلَّـهِ و هەر هی خودایە

الْعِزَّةُ هەرچی دەسەڵات و عیززەتە

وَلِرَسُولِهِ و هی پێغەمبەرەکەیەتی

وَلِلْمُؤْمِنِينَ وەهی بڕوادارانە

وَلَـٰكِنَّ بەڵام

الْمُنَافِقِينَ مونافیق و دووڕووەکان

لَا -

يَعْلَمُونَ نازانن

(٨)

يَا 

أَيُّهَا هۆ ئەی

الَّذِينَ ئەوانەی

آمَنُوا بڕواتان هێناوە

لَا -

تُلْهِكُمْ بێئاگاو نەکات

أَمْوَالُكُمْ ماڵ و سامانتان

وَلَا و نە

أَوْلَادُكُمْ ڕۆڵەکانیشتان

عَن لە

ذِكْرِ یاد و زیکری

اللَّـهِ خودا

وَمَن وهەرکەسێ

يَفْعَلْ -

ذَٰلِكَ ئاوا -بکات

فَأُولَـٰئِكَ 

هُمُ ئەوە بەڕاستی ئەوانن

الْخَاسِرُونَ خەسارەت مەند و دۆڕاوەکان

(٩)

وَأَنفِقُوا و ببەخشنەوە

مِن 

مَّا لەوەی

رَزَقْنَاكُم پێمانداو

مِّن لە

قَبْلِ پێش ئەوەی

أَن کە

يَأْتِيَ بێتەلای

أَحَدَكُمُ یەکێکتان

الْمَوْتُ مردن

فَيَقُولَ جا بڵێت

رَبِّ پەروەردیگارە

لَوْلَا 

أَخَّرْتَنِي بۆ دوام ناخەیت

إِلَىٰ تا

أَجَلٍ ماوە و کاتێکی

قَرِيبٍ نزیک

فَأَصَّدَّقَ ئەوسا خێر و صەدەقە دەکەم

وَأَكُن و-

مِّنَ لە

الصَّالِحِينَ چاکەکاران -دەبن

(١٠)

وَلَن 

يُؤَخِّرَ وهەرگیز دوا ناخات

اللَّـهُ خودا

نَفْسًا هیچ کەسێک

إِذَا کاتێک

جَاءَ هات

أَجَلُهَا ئەجەل و کاتی (مردنی)

وَاللَّـهُ و خوداش

خَبِيرٌ زانا و شارەزای تەواوە

بِمَا بەوەی

تَعْمَلُونَ دەیکەن

(١١)
