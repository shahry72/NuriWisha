بَعۡضَهُۥ هەندێکی

قُلۡ بڵێ

إِنَّمَآ لەڕاستیدا من هەر

أُمِرۡت فەرمانم پێکراوە

أَنۡ کە

أَعۡبُد بپەرستم

ٱللَّه خودا

وَلَآ أُشۡرِك و هاوەڵ و شەریک دانەنێم

بِهِۦٓ لۆی

إِلَيۡه بۆ لای ئەو

أَدۡعُوا بانگاوازی دەکەم

وَإِلَيۡه و هەر بۆلای ئەوە

مَ‍َٔاب گەڕانەوەم

(٣٦)

وَكَذَٰلِك و ئاوا

أَنزَلۡنَٰه ئەومان دامانبەزاند

حُكۡمًا مەحکەم و ڕێک و پێک

عَرَبِيّٗا (بەزمانی) عەرەبی یە

وَلَئِن وئەگەربێتوو

ٱتَّبَعۡت بکەوتایە دووای

أَهۡوَآءَهُم حەز و ئارەزووبازی ئەوان

بَعۡد دوای ئەوەی کە

مَاجَآءَك بۆت هات

مِن لە

ٱلۡعِلۡم زانین و زانستی تەواو

مَا لَك (ئیتر) نیە بۆ تۆ

مِن لە

ٱللَّه خوداوە

مِن (هیچ)

وَلِيّٖ دۆستێک

وَلَاوَاقٖ و نە (هیچ) پارێرێزەرێک یش

(٣٧)

وَلَقَدۡ و بەدڵنیایی

أَرۡسَلۡنَا ئێمە ناردمان

رُسُلٗا پەیامبەرانێک

مِّن لە

قَبۡلِك پێش تۆ

وَجَعَلۡنَا واماندانا

لَهُمۡ بۆیان

أَزۡوَٰج هاوسەرانێک

وَذُرِّيَّة و نەوە ش

وَمَا و-

كَان (-نە) بووە

لِرَسُولٍ بۆ پەیامبەرێک

أَن کە

يَأۡتِي بهێنێت

بِ‍َٔايَةٍ ئایەت وموعجیزەیەک

إِلَّا مەگەر تەنها

بِإِذۡن بەئیزن و ڕێپێدانی

ٱللَّه خودا

لِكُلّ بۆهەموو

أَجَلٖ کات و ماوەیەکی دانراو

كِتَابٞ کیتاب و نوسراوێ (هەیە)

(٣٨)

يَمۡحُوا لادەبا و ئەیسڕێتەوە

ٱللَّه خودا

مَا يَشَآء ئەوەی بیەوێت

وَيُثۡبِت و جێگیر دەکات

وَعِندَهُۥ لای خۆیەتی

أُمّ ئەسڵی

ٱلۡكِتَٰب کیتاب (ە ئاسمانیەکان لە لەوحی مەحفوظ)

(٣٩)

وَإِن و ئەگەر

مَّا -

نُرِيَنَّك پێت نیشاندەدەین

بَعۡض هەندێک

ٱلَّذِي لە ئەوەی

نَعِدُهُمۡ بەڵێنیان پێدەدین

أَوۡ یان

نَتَوَفَّيَنَّك بتمرێنین و تەواوت بکەیین

فَإِنَّمَا ئیتر هەر ئەوەت

عَلَيۡك لەسەرە

ٱلۡبَلَٰغ ڕاگەیاندن (و بەس)

وَعَلَيۡنَا و لەسەرئێمەش ە

ٱلۡحِسَاب لێپێچینەو حیسابکردن

(٤٠)

أَو ئایا

لَمۡ يَرَوۡا نەیان بینیووە

أَنَّا (کە) ئێمە

نَأۡتِي دێین بۆ

ٱلۡأَرۡض زەوی

نَنقُصُهَا کەم ئەکەینەوە

مِنۡ لە

أَطۡرَافِهَا لایەکانی

وَٱللَّه و خودا ش

يَحۡكُم حوکم و فەرمانڕەوایی دەکا

لَامُعَقِّب بەدواچووێک نیە

لِحُكۡمِهِۦ بۆ حوکم و فەرمانڕەوایی ئەو

وَهُو و هەر ئەوە

سَرِيع خێرایە

ٱلۡحِسَاب لێپێچینەوە و حساب ی

(٤١)

وَقَدۡ و بێگومان

مَكَر پیلانیان داڕشت

ٱلَّذِین ئەوانەی

مِن لە

قَبۡلِهِم پێش ئەوان (بوون)

فلِلَّه جا هەر بۆ خودایە

ٱلۡمَكۡر پیلان

جَمِيعٗا هەمووی

يَعۡلَم ئەو دەزانێ

مَا تَكۡسِب چی دەستەبەر دەکات

كُلّ هەموو

نَفۡسٖ نەفس و کەسێک

وَسَيَعۡلَم و لەوەولا دەزانن

ٱلۡكُفَّٰر کافر و بێبڕواکان

لِمَنۡ بۆکێیە

عُقۡبَى سەرئەنجام و عاقبەتی

ٱلدَّار ماڵ و مەنزڵگا

(٤٢)

وَيَقُول و دەڵێن

ٱلَّذِین ئەوانەی

كَفَرُوا کوفریانکرد و بێ بڕوابوون

لَسۡت تۆ-

مُرۡسَلٗا نێردراو (-نیت)

قُلۡ بڵێ

كَفَى بەسسە

بٱللَّه کەخودا

شَهِيدَۢا شایەت بێ

بَيۡنِي نێوان من

وَبَيۡنَكُمۡ نێوان ئێوەش

وَمَنۡ و ئەوەیشی

عِندَهُۥ لایەتی

عِلۡم زانست و زانیاری

ٱلۡكِتَٰب نوسراو و کیتاب (ی ئاسمانی)

(٤٣)
